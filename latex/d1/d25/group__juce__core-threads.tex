\doxysubsubsection{\texorpdfstring{ScopedLock}{ScopedLock}}
{\footnotesize\ttfamily using \mbox{\hyperlink{group__juce__core-threads_gadd0b55041ea549cc9b1dfae71670e543}{Scoped\+Lock}} =  \mbox{\hyperlink{class_critical_section_a4d398019b977a4c2151c6f1e2a06210b}{Critical\+Section\+::\+Scoped\+Lock\+Type}}}

Automatically locks and unlocks a \mbox{\hyperlink{class_critical_section}{Critical\+Section}} object.

You can use a Scoped\+Lock as a local variable to provide RAII-\/based locking of a \mbox{\hyperlink{class_critical_section}{Critical\+Section}}.

e.\+g.

\DoxyCodeLine{}
\DoxyCodeLine{    \textcolor{comment}{// assuming that this example function will be called by multiple threads}}
\DoxyCodeLine{    \textcolor{keywordtype}{void} foo()}
\DoxyCodeLine{    \{}
\DoxyCodeLine{        \textcolor{keyword}{const} \mbox{\hyperlink{class_generic_scoped_lock}{ScopedLock}} myScopedLock (objectLock);}
\DoxyCodeLine{}
\DoxyCodeLine{        \textcolor{comment}{// objectLock is now locked..}}
\DoxyCodeLine{}
\DoxyCodeLine{        ...do some thread-\/safe work here...}
\DoxyCodeLine{}
\DoxyCodeLine{        \textcolor{comment}{// ..and objectLock gets unlocked here, as myScopedLock goes out of}}

\DoxyCodeLine{    \}}
\DoxyCodeLine{\};}

\end{DoxyCode}

\doxysubsubsection{\texorpdfstring{ScopedUnlock}{ScopedUnlock}}
{\footnotesize\ttfamily using \mbox{\hyperlink{group__juce__core-threads_ga1a3fa8167a53f3d021763968a03282ea}{Scoped\+Unlock}} =  \mbox{\hyperlink{class_critical_section_afc7e83573aaf997f40d3c61a02823ea4}{Critical\+Section\+::\+Scoped\+Unlock\+Type}}}

Automatically unlocks and re-\/locks a \mbox{\hyperlink{class_critical_section}{Critical\+Section}} object.

This is the reverse of a Scoped\+Lock object -\/ instead of locking the critical section for the lifetime of this object, it unlocks it.

Make sure you don\textquotesingle{}t try to unlock critical sections that aren\textquotesingle{}t actually locked!

e.\+g. 
\begin{DoxyCode}{0}
\DoxyCodeLine{\textcolor{keyword}{struct }MyObject}
\DoxyCodeLine{\{}
\DoxyCodeLine{    \mbox{\hyperlink{class_critical_section}{CriticalSection}} objectLock;}
\DoxyCodeLine{}
\DoxyCodeLine{    \textcolor{keywordtype}{void} foo()}
\DoxyCodeLine{    \{}
\DoxyCodeLine{        \{}
\DoxyCodeLine{            \textcolor{keyword}{const} \mbox{\hyperlink{class_generic_scoped_lock}{ScopedLock}} myScopedLock (objectLock);}
\DoxyCodeLine{}
\DoxyCodeLine{            \textcolor{comment}{// objectLock is now locked..}}
\DoxyCodeLine{}
\DoxyCodeLine{            \{}
\DoxyCodeLine{                \mbox{\hyperlink{class_generic_scoped_unlock}{ScopedUnlock}} myUnlocker (objectLock);}
\DoxyCodeLine{}

\DoxyCodeLine{            \}}
\DoxyCodeLine{}
\DoxyCodeLine{            \textcolor{comment}{// ..and now locked again..}}
\DoxyCodeLine{        \}}
\DoxyCodeLine{}

\DoxyCodeLine{    \}}
\DoxyCodeLine{\};}
