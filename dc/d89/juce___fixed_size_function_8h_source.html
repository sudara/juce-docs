36 template <typename Ret, typename... Args>
37 struct Vtable
38 {
39 using Storage = void*;
40
41 using Move = void (*) (Storage, Storage);
42 using Call = Ret (*) (Storage, Args...);
43 using Clear = void (*) (Storage);
44
45 constexpr Vtable (Move moveIn, Call callIn, Clear clearIn) noexcept
46 : move (moveIn), call (callIn), clear (clearIn) {}
47
48 Move move = nullptr;
49 Call call = nullptr;
50 Clear clear = nullptr;
51 };

53 template <typename Fn>
54 void move (void* from, void* to)
55 {
56 new (to) Fn (std::move (*reinterpret_cast<Fn*> (from)));
57 }

59 template <typename Fn, typename Ret, typename... Args>
60 typename std::enable_if<std::is_same<Ret, void>::value, Ret>::type call (void* s, Args... args)
61 {
62 (*reinterpret_cast<Fn*> (s)) (args...);
63 }
64
65 template <typename Fn, typename Ret, typename... Args>
66 typename std::enable_if<! std::is_same<Ret, void>::value, Ret>::type call (void* s, Args... args)
67 {
68 return (*reinterpret_cast<Fn*> (s)) (std::forward<Args> (args)...);
69 }
70
71 template <typename Fn>
72 void clear (void* s)
73 {
74 auto& fn = *reinterpret_cast<Fn*> (s);
75 fn.~Fn();
76 // I know this looks insane, for some reason MSVC 14 sometimes thinks fn is unreferenced
77 ignoreUnused (fn);
78 }

81 constexpr Vtable<Ret, Args...> makeVtable()
82 {
83 return { move <Fn>, call <Fn, Ret, Args...>, clear<Fn> };
84 }

128 template <typename Callable,
129 typename Fn = Decay<Callable>,
130 IntIfValidConversion<Callable> = 0>
131 FixedSizeFunction (Callable&& callable)
132 {
133 static_assert (sizeof (Fn) <= len,
134 "The requested function cannot fit in this FixedSizeFunction");
135 static_assert (alignof (Fn) <= alignof (Storage),
136 "FixedSizeFunction cannot accommodate the requested alignment requirements");
137
138 static constexpr auto vtableForCallable = detail::makeVtable<Fn, Ret, Args...>();
139 vtable = &vtableForCallable;
140
141 auto* ptr = new (&storage) Fn (std::forward<Callable> (callable));
142 jassertquiet ((void*) ptr == (void*) &storage);
143 }
