33extern bool offerKeyMessageToJUCEWindow (MSG&);
34
35static HHOOK mouseWheelHook = nullptr, keyboardHook = nullptr;
36static int numHookUsers = 0;
37
38struct WindowsHooks
39{
40 WindowsHooks()
41 {
42 if (numHookUsers++ == 0)
43 {
44 mouseWheelHook = SetWindowsHookEx (WH_MOUSE, mouseWheelHookCallback,
45 (HINSTANCE) juce::Process::getCurrentModuleInstanceHandle(),
46 GetCurrentThreadId());
47
48 keyboardHook = SetWindowsHookEx (WH_GETMESSAGE, keyboardHookCallback,
49 (HINSTANCE) juce::Process::getCurrentModuleInstanceHandle(),
50 GetCurrentThreadId());
51 }
52 }
53
54 ~WindowsHooks()
55 {
56 if (--numHookUsers == 0)
57 {
58 if (mouseWheelHook != nullptr)
59 {
60 UnhookWindowsHookEx (mouseWheelHook);
61 mouseWheelHook = nullptr;
62 }
63
64 if (keyboardHook != nullptr)
65 {
66 UnhookWindowsHookEx (keyboardHook);
67 keyboardHook = nullptr;
68 }
69 }
70 }
71
72 static LRESULT CALLBACK mouseWheelHookCallback (int nCode, WPARAM wParam, LPARAM lParam)
73 {
74 if (nCode >= 0 && wParam == WM_MOUSEWHEEL)
75 {
76 // using a local copy of this struct to support old mingw libraries
77 struct MOUSEHOOKSTRUCTEX_ : public MOUSEHOOKSTRUCT { DWORD mouseData; };
78
79 auto& hs = *(MOUSEHOOKSTRUCTEX_*) lParam;
80
81 if (auto* comp = Desktop::getInstance().findComponentAt ({ hs.pt.x, hs.pt.y }))
82 if (comp->getWindowHandle() != nullptr)
83 return PostMessage ((HWND) comp->getWindowHandle(), WM_MOUSEWHEEL,
84 hs.mouseData & 0xffff0000, (hs.pt.x & 0xffff) | (hs.pt.y << 16));
85 }
86
87 return CallNextHookEx (mouseWheelHook, nCode, wParam, lParam);
88 }
89
90 static LRESULT CALLBACK keyboardHookCallback (int nCode, WPARAM wParam, LPARAM lParam)
91 {
92 MSG& msg = *(MSG*) lParam;
93
94 if (nCode == HC_ACTION && wParam == PM_REMOVE
95 && offerKeyMessageToJUCEWindow (msg))
96 {
97 zerostruct (msg);
98 msg.message = WM_USER;
99 return 1;
100 }
101
102 return CallNextHookEx (keyboardHook, nCode, wParam, lParam);
103 }
104};
