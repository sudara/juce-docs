68 template <typename ProcessContext>
69 void process (const ProcessContext& context) noexcept
70 {
71 const auto& inputBlock = context.getInputBlock();
72 auto& outputBlock = context.getOutputBlock();
73 const auto numChannels = outputBlock.getNumChannels();
74 const auto numSamples = outputBlock.getNumSamples();
75
76 jassert (inputBlock.getNumChannels() == numChannels);
77 jassert (inputBlock.getNumSamples() == numSamples);
78
79 if (context.isBypassed)
80 {
81 outputBlock.copyFrom (inputBlock);
82 return;
83 }
84
85 for (size_t channel = 0; channel < numChannels; ++channel)
86 {
87 auto* inputSamples = inputBlock .getChannelPointer (channel);
88 auto* outputSamples = outputBlock.getChannelPointer (channel);
89
90 for (size_t i = 0; i < numSamples; ++i)
91 outputSamples[i] = processSample ((int) channel, inputSamples[i]);
92 }
93 }
