155 float getNextSample() noexcept
156 {
157 if (state == State::idle)
158 return 0.0f;
159
160 if (state == State::attack)
161 {
162 envelopeVal += attackRate;
163
164 if (envelopeVal >= 1.0f)
165 {
166 envelopeVal = 1.0f;
167 goToNextState();
168 }
169 }
170 else if (state == State::decay)
171 {
172 envelopeVal -= decayRate;
173
174 if (envelopeVal <= parameters.sustain)
175 {
176 envelopeVal = parameters.sustain;
177 goToNextState();
178 }
179 }
180 else if (state == State::sustain)
181 {
182 envelopeVal = parameters.sustain;
183 }
184 else if (state == State::release)
185 {
186 envelopeVal -= releaseRate;
187
188 if (envelopeVal <= 0.0f)
189 goToNextState();
190 }
191
192 return envelopeVal;
193 }

200 template <typename FloatType>
201 void applyEnvelopeToBuffer (AudioBuffer<FloatType>& buffer, int startSample, int numSamples)
202 {
203 jassert (startSample + numSamples <= buffer.getNumSamples());
204
205 if (state == State::idle)
206 {
207 buffer.clear (startSample, numSamples);
208 return;
209 }
210
211 if (state == State::sustain)
212 {
213 buffer.applyGain (startSample, numSamples, parameters.sustain);
214 return;
215 }
216
217 auto numChannels = buffer.getNumChannels();
218
219 while (--numSamples >= 0)
220 {
221 auto env = getNextSample();
222
223 for (int i = 0; i < numChannels; ++i)
224 buffer.getWritePointer (i)[startSample] *= env;
225
226 ++startSample;
227 }
228 }

231 //==============================================================================
232 void recalculateRates() noexcept
233 {
234 auto getRate = [] (float distance, float timeInSeconds, double sr)
235 {
236 return timeInSeconds > 0.0f ? (float) (distance / (timeInSeconds * sr)) : -1.0f;
237 };
238
239 attackRate = getRate (1.0f, parameters.attack, sampleRate);
240 decayRate = getRate (1.0f - parameters.sustain, parameters.decay, sampleRate);
241 releaseRate = getRate (parameters.sustain, parameters.release, sampleRate);
242
243 if ((state == State::attack && attackRate <= 0.0f)
244 || (state == State::decay && (decayRate <= 0.0f || envelopeVal <= parameters.sustain))
245 || (state == State::release && releaseRate <= 0.0f))
246 {
247 goToNextState();
248 }
249 }
250
251 void goToNextState() noexcept
252 {
253 if (state == State::attack)
254 state = (decayRate > 0.0f ? State::decay : State::sustain);
255 else if (state == State::decay)
256 state = State::sustain;
257 else if (state == State::release)
258 reset();
259 }
