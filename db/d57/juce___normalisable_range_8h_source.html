125 ValueType convertTo0to1 (ValueType v) const noexcept
126 {
127 if (convertTo0To1Function != nullptr)
128 return clampTo0To1 (convertTo0To1Function (start, end, v));
129
130 auto proportion = clampTo0To1 ((v - start) / (end - start));
131
132 if (skew == static_cast<ValueType> (1))
133 return proportion;
134
135 if (! symmetricSkew)
136 return std::pow (proportion, skew);
137
138 auto distanceFromMiddle = static_cast<ValueType> (2) * proportion - static_cast<ValueType> (1);
139
140 return (static_cast<ValueType> (1) + std::pow (std::abs (distanceFromMiddle), skew)
141 * (distanceFromMiddle < ValueType() ? static_cast<ValueType> (-1)
142 : static_cast<ValueType> (1)))
143 / static_cast<ValueType> (2);
144 }
145
149 ValueType convertFrom0to1 (ValueType proportion) const noexcept
150 {
151 proportion = clampTo0To1 (proportion);
152
153 if (convertFrom0To1Function != nullptr)
154 return convertFrom0To1Function (start, end, proportion);
155
156 if (! symmetricSkew)
157 {
158 if (skew != static_cast<ValueType> (1) && proportion > ValueType())
159 proportion = std::exp (std::log (proportion) / skew);
160
161 return start + (end - start) * proportion;
162 }
163
164 auto distanceFromMiddle = static_cast<ValueType> (2) * proportion - static_cast<ValueType> (1);
165
166 if (skew != static_cast<ValueType> (1) && distanceFromMiddle != static_cast<ValueType> (0))
167 distanceFromMiddle = std::exp (std::log (std::abs (distanceFromMiddle)) / skew)
168 * (distanceFromMiddle < ValueType() ? static_cast<ValueType> (-1)
169 : static_cast<ValueType> (1));
170
171 return start + (end - start) / static_cast<ValueType> (2) * (static_cast<ValueType> (1) + distanceFromMiddle);
172 }
173
177 ValueType snapToLegalValue (ValueType v) const noexcept
178 {
179 if (snapToLegalValueFunction != nullptr)
180 return snapToLegalValueFunction (start, end, v);
181
182 if (interval > ValueType())
183 v = start + interval * std::floor ((v - start) / interval + static_cast<ValueType> (0.5));
184
185 return (v <= start || end <= start) ? start : (v >= end ? end : v);
186 }
