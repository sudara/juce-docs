84 template <typename ProcessContext>
85 void process (const ProcessContext& context) noexcept
86 {
87 const auto& inputBlock = context.getInputBlock();
88 auto& outputBlock = context.getOutputBlock();
89 const auto numChannels = outputBlock.getNumChannels();
90 const auto numSamples = outputBlock.getNumSamples();
91
92 jassert (inputBlock.getNumChannels() == numChannels);
93 jassert (inputBlock.getNumChannels() == lastOutput.size());
94 jassert (inputBlock.getNumSamples() == numSamples);
95
96 if (context.isBypassed)
97 {
98 outputBlock.copyFrom (inputBlock);
99 return;
100 }
101
102 int numSamplesDown = 0;
103 auto counter = updateCounter;
104
105 for (size_t i = 0; i < numSamples; ++i)
106 {
107 if (counter == 0)
108 numSamplesDown++;
109
110 counter++;
111
112 if (counter == maxUpdateCounter)
113 counter = 0;
114 }
115
116 if (numSamplesDown > 0)
117 {
118 auto freqBlock = AudioBlock<SampleType>(bufferFrequency).getSubBlock (0, (size_t) numSamplesDown);
119 auto contextFreq = ProcessContextReplacing<SampleType> (freqBlock);
120 freqBlock.clear();
121
122 osc.process (contextFreq);
123 freqBlock.multiplyBy (oscVolume);
124 }
125
126 auto* freqSamples = bufferFrequency.getWritePointer (0);
127
128 for (int i = 0; i < numSamplesDown; ++i)
129 {
130 auto lfo = jlimit (static_cast<SampleType> (0.0),
131 static_cast<SampleType> (1.0),
132 freqSamples[i] + normCentreFrequency);
133
134 freqSamples[i] = mapToLog10 (lfo, static_cast<SampleType> (20.0),
135 static_cast<SampleType> (jmin (20000.0, 0.49 * sampleRate)));
136 }
137
138 auto currentFrequency = filters[0]->getCutoffFrequency();
139 dryWet.pushDrySamples (inputBlock);
140
141 for (size_t channel = 0; channel < numChannels; ++channel)
142 {
143 counter = updateCounter;
144 int k = 0;
145
146 auto* inputSamples = inputBlock .getChannelPointer (channel);
147 auto* outputSamples = outputBlock.getChannelPointer (channel);
148
149 for (size_t i = 0; i < numSamples; ++i)
150 {
151 auto input = inputSamples[i];
152 auto output = input - lastOutput[channel];
153
154 if (i == 0 && counter != 0)
155 for (int n = 0; n < numStages; ++n)
156 filters[n]->setCutoffFrequency (currentFrequency);
157
158 if (counter == 0)
159 {
160 for (int n = 0; n < numStages; ++n)
161 filters[n]->setCutoffFrequency (freqSamples[k]);
162
163 k++;
164 }
165
166 for (int n = 0; n < numStages; ++n)
167 output = filters[n]->processSample ((int) channel, output);
168
169 outputSamples[i] = output;
170 lastOutput[channel] = output * feedbackVolume[channel].getNextValue();
171
172 counter++;
173
174 if (counter == maxUpdateCounter)
175 counter = 0;
176 }
177 }
178
179 dryWet.mixWetSamples (outputBlock);
180 updateCounter = (updateCounter + (int) numSamples) % maxUpdateCounter;
181 }
182
183private:
184 //==============================================================================
185 void update();
186
187 //==============================================================================
188 Oscillator<SampleType> osc;
189 OwnedArray<FirstOrderTPTFilter<SampleType>> filters;
190 SmoothedValue<SampleType, ValueSmoothingTypes::Linear> oscVolume;
191 std::vector<SmoothedValue<SampleType, ValueSmoothingTypes::Linear>> feedbackVolume { 2 };
192 DryWetMixer<SampleType> dryWet;
193 std::vector<SampleType> lastOutput { 2 };
194 AudioBuffer<SampleType> bufferFrequency;
195 SampleType normCentreFrequency = 0.5;
196 double sampleRate = 44100.0;
197
198 int updateCounter = 0;
199 static constexpr int maxUpdateCounter = 4;
200
201 SampleType rate = 1.0, depth = 0.5, feedback = 0.0, mix = 0.5;
202 SampleType centreFrequency = 1300.0;
203 static constexpr int numStages = 6;
204};
