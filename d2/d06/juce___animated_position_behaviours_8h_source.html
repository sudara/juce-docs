54 void setFriction (double newFriction) noexcept

61 void setMinimumVelocity (double newMinimumVelocityToUse) noexcept
62 {
63 minimumVelocity = newMinimumVelocityToUse;

65
70 void releasedWithVelocity (double /*position*/, double releaseVelocity) noexcept
71 {
72 velocity = releaseVelocity;
73 }
74
78 double getNextPosition (double oldPos, double elapsedSeconds) noexcept
79 {
80 velocity *= damping;
81
82 if (std::abs (velocity) < minimumVelocity)
83 velocity = 0;
84

86 }
87
91 bool isStopped (double /*position*/) const noexcept

113 struct SnapToPageBoundaries
114 {
115 SnapToPageBoundaries() = default;
116
121 void releasedWithVelocity (double position, double releaseVelocity) noexcept
122 {
123 targetSnapPosition = std::floor (position + 0.5);
124
125 if (releaseVelocity > 1.0 && targetSnapPosition < position) ++targetSnapPosition;
126 if (releaseVelocity < -1.0 && targetSnapPosition > position) --targetSnapPosition;
127 }
128
132 double getNextPosition (double oldPos, double elapsedSeconds) const noexcept
133 {
134 if (isStopped (oldPos))
135 return targetSnapPosition;
136
137 const double snapSpeed = 10.0;
138 const double velocity = (targetSnapPosition - oldPos) * snapSpeed;
139 const double newPos = oldPos + velocity * elapsedSeconds;
140
141 return isStopped (newPos) ? targetSnapPosition : newPos;
142 }
143
147 bool isStopped (double position) const noexcept
148 {
149 return std::abs (targetSnapPosition - position) < 0.001;
150 }
151
152 private:
153 double targetSnapPosition = 0.0;
154 };
