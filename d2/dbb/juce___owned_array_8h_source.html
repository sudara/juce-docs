559 template <typename ElementComparator>
560 int indexOfSorted (ElementComparator& comparator, const ObjectClass* objectToLookFor) const noexcept
561 {
562 // If you pass in an object with a static compareElements() method, this
563 // avoids getting warning messages about the parameter being unused
564 ignoreUnused (comparator);
565
566 const ScopedLockType lock (getLock());
567 int s = 0, e = values.size();
568
569 while (s < e)
570 {
571 if (comparator.compareElements (objectToLookFor, values[s]) == 0)
572 return s;
573
574 auto halfway = (s + e) / 2;
575
576 if (halfway == s)
577 break;
578
579 if (comparator.compareElements (objectToLookFor, values[halfway]) >= 0)
580 s = halfway;
581 else
582 e = halfway;
583 }
584
585 return -1;

599 void remove (int indexToRemove, bool deleteObject = true)
600 {
601 std::unique_ptr<ObjectClass> toDelete;
602
603 {
604 const ScopedLockType lock (getLock());
605
606 if (isPositiveAndBelow (indexToRemove, values.size()))
607 {
608 auto** e = values.begin() + indexToRemove;
609
610 if (deleteObject)
611 toDelete.reset (*e);
612
613 values.removeElements (indexToRemove, 1);
614 }

617 if ((values.size() << 1) < values.capacity())

683 void removeRange (int startIndex, int numberToRemove, bool deleteObjects = true)
684 {
685 const ScopedLockType lock (getLock());
686 auto endIndex = jlimit (0, values.size(), startIndex + numberToRemove);
687 startIndex = jlimit (0, values.size(), startIndex);
688 numberToRemove = endIndex - startIndex;
689
690 if (numberToRemove > 0)
691 {
692 Array<ObjectClass*> objectsToDelete;
693
694 if (deleteObjects)
695 objectsToDelete.addArray (values.begin() + startIndex, numberToRemove);
696
697 values.removeElements (startIndex, numberToRemove);
698
699 for (auto& o : objectsToDelete)
700 ContainerDeletePolicy<ObjectClass>::destroy (o);
701
702 if ((values.size() << 1) < values.capacity())
703 minimiseStorageOverheads();
704 }
705 }
