55 inline ~ScopedPointer() { reset(); }
56
57 ScopedPointer& operator= (ScopedPointer& objectToTransferFrom)
58 {
59 if (this != objectToTransferFrom.getAddress())
60 {
61 // Two ScopedPointers should never be able to refer to the same object - if
62 // this happens, you must have done something dodgy!
63 jassert (object == nullptr || object != objectToTransferFrom.object);
64 reset (objectToTransferFrom.release());
65 }
66
67 return *this;
68 }
69
70 ScopedPointer& operator= (ObjectType* newObjectToTakePossessionOf)
71 {
72 reset (newObjectToTakePossessionOf);
73 return *this;
74 }
75
76 ScopedPointer (ScopedPointer&& other) noexcept : object (other.object)
77 {
78 other.object = nullptr;
79 }
80
81 ScopedPointer& operator= (ScopedPointer&& other) noexcept
82 {
83 reset (other.release());
84 return *this;
85 }

88 inline operator ObjectType*() const noexcept { return object; }
89 inline ObjectType* get() const noexcept { return object; }
90 inline ObjectType& operator*() const noexcept { return *object; }
91 inline ObjectType* operator->() const noexcept { return object; }
92
93 void reset()
94 {
95 auto* oldObject = object;
96 object = {};
97 ContainerDeletePolicy<ObjectType>::destroy (oldObject);
98 }
99
100 void reset (ObjectType* newObject)
101 {
102 if (object != newObject)
103 {
104 auto* oldObject = object;
105 object = newObject;
106 ContainerDeletePolicy<ObjectType>::destroy (oldObject);
107 }
108 else
109 {
110 // You're trying to reset this ScopedPointer to itself! This will work here as ScopedPointer does an equality check
111 // but be aware that std::unique_ptr won't do this and you could end up with some nasty, subtle bugs!
112 jassert (newObject == nullptr);
113 }
114 }
115
116 void reset (ScopedPointer& newObject)
117 {
118 reset (newObject.release());
119 }
120
121 ObjectType* release() noexcept { auto* o = object; object = {}; return o; }
122

124 void swapWith (ScopedPointer<ObjectType>& other) noexcept
125 {
126 // Two ScopedPointers should never be able to refer to the same object - if
127 // this happens, you must have done something dodgy!
128 jassert (object != other.object || this == other.getAddress() || object == nullptr);
129
130 std::swap (object, other.object);
131 }
