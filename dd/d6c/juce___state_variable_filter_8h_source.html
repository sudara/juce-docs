111 template <typename ProcessContext>
112 void process (const ProcessContext& context) noexcept
113 {
114 static_assert (std::is_same<typename ProcessContext::SampleType, SampleType>::value,
115 "The sample-type of the filter must match the sample-type supplied to this process callback");
116
117 if (context.isBypassed)
118 processInternal<true, ProcessContext> (context);
119 else
120 processInternal<false, ProcessContext> (context);
121 }

125 SampleType JUCE_VECTOR_CALLTYPE processSample (SampleType sample) noexcept
126 {
127 switch (parameters->type)
128 {
129 case Parameters<NumericType>::Type::lowPass: return processLoop<false, Parameters<NumericType>::Type::lowPass> (sample, *parameters); break;
130 case Parameters<NumericType>::Type::bandPass: return processLoop<false, Parameters<NumericType>::Type::bandPass> (sample, *parameters); break;
131 case Parameters<NumericType>::Type::highPass: return processLoop<false, Parameters<NumericType>::Type::highPass> (sample, *parameters); break;
132 default: jassertfalse;
133 }
134
135 return SampleType{0};
136 }

139 //==============================================================================
140 template <bool isBypassed, typename Parameters<NumericType>::Type type>
141 SampleType JUCE_VECTOR_CALLTYPE processLoop (SampleType sample, Parameters<NumericType>& state) noexcept
142 {
143 y[2] = (sample - s1 * state.R2 - s1 * state.g - s2) * state.h;
144
145 y[1] = y[2] * state.g + s1;
146 s1 = y[2] * state.g + y[1];
147
148 y[0] = y[1] * state.g + s2;
149 s2 = y[1] * state.g + y[0];
150
151 return isBypassed ? sample : y[static_cast<size_t> (type)];
152 }

154 template <bool isBypassed, typename Parameters<NumericType>::Type type>
155 void processBlock (const SampleType* input, SampleType* output, size_t n) noexcept
156 {
157 auto state = *parameters;
158
159 for (size_t i = 0 ; i < n; ++i)
160 output[i] = processLoop<isBypassed, type> (input[i], state);
161
162 #if JUCE_DSP_ENABLE_SNAP_TO_ZERO
163 snapToZero();
164 #endif
165
166 *parameters = state;
167 }

169 template <bool isBypassed, typename ProcessContext>
170 void processInternal (const ProcessContext& context) noexcept
171 {
172 auto&& inputBlock = context.getInputBlock();
173 auto&& outputBlock = context.getOutputBlock();
174
175 // This class can only process mono signals. Use the ProcessorDuplicator class
176 // to apply this filter on a multi-channel audio stream.
177 jassert (inputBlock.getNumChannels() == 1);
178 jassert (outputBlock.getNumChannels() == 1);
179
180 auto n = inputBlock.getNumSamples();
181 auto* src = inputBlock .getChannelPointer (0);
182 auto* dst = outputBlock.getChannelPointer (0);
183
184 switch (parameters->type)
185 {
186 case Parameters<NumericType>::Type::lowPass: processBlock<isBypassed, Parameters<NumericType>::Type::lowPass> (src, dst, n); break;
187 case Parameters<NumericType>::Type::bandPass: processBlock<isBypassed, Parameters<NumericType>::Type::bandPass> (src, dst, n); break;
188 case Parameters<NumericType>::Type::highPass: processBlock<isBypassed, Parameters<NumericType>::Type::highPass> (src, dst, n); break;
189 default: jassertfalse;
190 }
191 }

230 void setCutOffFrequency (double sampleRate, NumericType frequency,
231 NumericType resonance = static_cast<NumericType> (1.0 / MathConstants<double>::sqrt2)) noexcept
232 {
233 jassert (sampleRate > 0);
234 jassert (resonance > NumericType (0));
235 jassert (frequency > NumericType (0) && frequency <= NumericType (sampleRate * 0.5));
236
237 g = static_cast<NumericType> (std::tan (MathConstants<double>::pi * frequency / sampleRate));
238 R2 = static_cast<NumericType> (1.0 / resonance);
239 h = static_cast<NumericType> (1.0 / (1.0 + R2 * g + g * g));
240 }
241
242 //==============================================================================
246 using Ptr = ReferenceCountedObjectPtr<Parameters>;
247
248 //==============================================================================
249 Parameters() = default;
250 Parameters (const Parameters& o) : g (o.g), R2 (o.R2), h (o.h) {}
251 Parameters& operator= (const Parameters& o) noexcept { g = o.g; R2 = o.R2; h = o.h; return *this; }
252
253 //==============================================================================
254 NumericType g = static_cast<NumericType> (std::tan (MathConstants<double>::pi * 200.0 / 44100.0));
255 NumericType R2 = static_cast<NumericType> (MathConstants<double>::sqrt2);
256 NumericType h = static_cast<NumericType> (1.0 / (1.0 + R2 * g + g * g));
257 };
