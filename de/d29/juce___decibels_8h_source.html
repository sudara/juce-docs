56 template <typename Type>
57 static Type gainToDecibels (Type gain,
58 Type minusInfinityDb = Type (defaultMinusInfinitydB))
59 {
60 return gain > Type() ? jmax (minusInfinityDb, static_cast<Type> (std::log10 (gain)) * Type (20.0))

72 template <typename Type>
73 static String toString (Type decibels,
74 int decimalPlaces = 2,
75 Type minusInfinityDb = Type (defaultMinusInfinitydB),
76 bool shouldIncludeSuffix = true,
77 StringRef customMinusInfinityString = {})
78 {
79 String s;
80 s.preallocateBytes (20);
81
82 if (decibels <= minusInfinityDb)
83 {
84 if (customMinusInfinityString.isEmpty())
85 s << "-INF";
86 else
87 s << customMinusInfinityString;
88 }
89 else
90 {
91 if (decibels >= Type())
92 s << '+';
93
94 if (decimalPlaces <= 0)
95 s << roundToInt (decibels);
96 else
97 s << String (decibels, decimalPlaces);
98 }
99
100 if (shouldIncludeSuffix)
101 s << " dB";
102
103 return s;
104 }
