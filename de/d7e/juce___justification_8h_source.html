78 template <typename ValueType>
79 void applyToRectangle (ValueType& x, ValueType& y, ValueType w, ValueType h,
80 ValueType spaceX, ValueType spaceY, ValueType spaceW, ValueType spaceH) const noexcept
81 {
82 x = spaceX;
83 if ((flags & horizontallyCentred) != 0) x += (spaceW - w) / (ValueType) 2;
84 else if ((flags & right) != 0) x += spaceW - w;
85
86 y = spaceY;
87 if ((flags & verticallyCentred) != 0) y += (spaceH - h) / (ValueType) 2;
88 else if ((flags & bottom) != 0) y += spaceH - h;
89 }
90
93 template <typename ValueType>
94 const Rectangle<ValueType> appliedToRectangle (const Rectangle<ValueType>& areaToAdjust,
95 const Rectangle<ValueType>& targetSpace) const noexcept
96 {
97 ValueType x = areaToAdjust.getX(), y = areaToAdjust.getY();
98 applyToRectangle (x, y, areaToAdjust.getWidth(), areaToAdjust.getHeight(),
99 targetSpace.getX(), targetSpace.getY(), targetSpace.getWidth(), targetSpace.getHeight());
100 return areaToAdjust.withPosition (x, y);
101 }
