53private:
54 //==============================================================================
55 typedef std::map<HeavyweightLeakedObjectDetector<OwnerClass>*, String> BacktraceMap;
56
57 //==============================================================================
58 struct BacktraceMapHolder
59 {
60 BacktraceMapHolder() = default;
61
62 ~BacktraceMapHolder()
63 {
64 if (map.size() > 0)
65 {
66 DBG ("*** Leaked objects detected: " << map.size() << " instance(s) of class " << getLeakedObjectClassName());
67 DBG (getFormattedBacktracesString());
68
77 jassertfalse;
78 }
79 }
80
81 String getFormattedBacktracesString() const
82 {
83 String str;
84
85 int counter = 1;
86 for (auto& bt : map)
87 {
88 str << "\nBacktrace " << String (counter++) << "\n"
89 << "-----------------------------------------------------------------" << "\n"
90 << bt.second;
91 }
92
93 return str;
94 }
95
96 BacktraceMap map;
97 };
98
99 static BacktraceMap& getBacktraceMap()
100 {
101 static BacktraceMapHolder holder;
102 return holder.map;
103 }
104
105 static const char* getLeakedObjectClassName()
106 {
107 return OwnerClass::getLeakedObjectClassName();
108 }
109};
