68public:

75 void incReferenceCount() noexcept

79
83 void decReferenceCount() noexcept
84 {
85 jassert (getReferenceCount() > 0);
86
87 if (--refCount == 0)
88 delete this;
89 }
90
95 bool decReferenceCountWithoutDeleting() noexcept
96 {
97 jassert (getReferenceCount() > 0);

99 }

105protected:

248{

256
258 ReferenceCountedObjectPtr (decltype (nullptr)) noexcept {}
259
263 ReferenceCountedObjectPtr (ReferencedType* refCountedObject) noexcept

265 {

267 }
268
272 ReferenceCountedObjectPtr (ReferencedType& refCountedObject) noexcept
273 : referencedObject (&refCountedObject)
274 {

276 }
277
281 ReferenceCountedObjectPtr (const ReferenceCountedObjectPtr& other) noexcept
282 : referencedObject (other.referencedObject)
283 {
284 incIfNotNull (referencedObject);
285 }
286
288 ReferenceCountedObjectPtr (ReferenceCountedObjectPtr&& other) noexcept
289 : referencedObject (other.referencedObject)

291 other.referencedObject = nullptr;
292 }
293
297 template <typename Convertible>
298 ReferenceCountedObjectPtr (const ReferenceCountedObjectPtr<Convertible>& other) noexcept
299 : referencedObject (other.get())
300 {
301 incIfNotNull (referencedObject);
302 }
303
308 ReferenceCountedObjectPtr& operator= (const ReferenceCountedObjectPtr& other)
309 {
310 return operator= (other.referencedObject);
311 }
312
317 template <typename Convertible>
318 ReferenceCountedObjectPtr& operator= (const ReferenceCountedObjectPtr<Convertible>& other)
319 {
320 return operator= (other.get());
321 }
322
328 ReferenceCountedObjectPtr& operator= (ReferencedType* newObject)
329 {
330 if (newObject != nullptr)
331 return operator= (*newObject);
332
333 reset();
334 return *this;
335 }
336
342 ReferenceCountedObjectPtr& operator= (ReferencedType& newObject)
343 {
344 if (referencedObject != &newObject)
345 {
346 newObject.incReferenceCount();
347 auto* oldObject = referencedObject;
348 referencedObject = &newObject;
349 decIfNotNull (oldObject);
350 }
351
352 return *this;
353 }
354
356 ReferenceCountedObjectPtr& operator= (decltype (nullptr))
357 {
358 reset();
359 return *this;
360 }
361
363 ReferenceCountedObjectPtr& operator= (ReferenceCountedObjectPtr&& other) noexcept
364 {
365 std::swap (referencedObject, other.referencedObject);
366 return *this;
367 }
368
373 ~ReferenceCountedObjectPtr()
374 {
375 decIfNotNull (referencedObject);
376 }
377
378 //==============================================================================
382 ReferencedType* get() const noexcept { return referencedObject; }
383
385 void reset() noexcept
386 {
387 auto oldObject = referencedObject; // need to null the pointer before deleting the object
388 referencedObject = nullptr; // in case this ptr is itself deleted as a side-effect
389 decIfNotNull (oldObject); // of the destructor
390 }
391
392 // the -> operator is called on the referenced object
393 ReferencedType* operator->() const noexcept
394 {
395 jassert (referencedObject != nullptr); // null pointer method call!
396 return referencedObject;
397 }
398
402 ReferencedType& operator*() const noexcept { jassert (referencedObject != nullptr); return *referencedObject; }
403
405 bool operator== (decltype (nullptr)) const noexcept { return referencedObject == nullptr; }
407 bool operator!= (decltype (nullptr)) const noexcept { return referencedObject != nullptr; }
408
410 bool operator== (const ObjectType* other) const noexcept { return referencedObject == other; }
412 bool operator== (const ReferenceCountedObjectPtr& other) const noexcept { return referencedObject == other.get(); }
414 bool operator!= (const ObjectType* other) const noexcept { return referencedObject != other; }
416 bool operator!= (const ReferenceCountedObjectPtr& other) const noexcept { return referencedObject != other.get(); }
417
418 #if JUCE_STRICT_REFCOUNTEDPOINTER
420 explicit operator bool() const noexcept { return referencedObject != nullptr; }
421
422 #else
429 operator ReferencedType*() const noexcept { return referencedObject; }
430 #endif
431
432 #ifndef DOXYGEN
433 [[deprecated ("Use the get method instead.")]]
434 ReferencedType* getObject() const { return get(); }
435 #endif
436
437private:
438 //==============================================================================
439 ReferencedType* referencedObject = nullptr;
440
441 static void incIfNotNull (ReferencedType* o) noexcept
442 {
443 if (o != nullptr)
444 o->incReferenceCount();
445 }
446
447 static void decIfNotNull (ReferencedType* o) noexcept
448 {
449 if (o != nullptr && o->decReferenceCountWithoutDeleting())
450 ContainerDeletePolicy<ReferencedType>::destroy (o);
451 }
