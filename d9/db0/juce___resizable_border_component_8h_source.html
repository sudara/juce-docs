143 template <typename ValueType>
144 Rectangle<ValueType> resizeRectangleBy (Rectangle<ValueType> original,
145 const Point<ValueType>& distance) const noexcept
146 {
147 if (isDraggingWholeObject())
148 return original + distance;
149
150 if (isDraggingLeftEdge()) original.setLeft (jmin (original.getRight(), original.getX() + distance.x));
151 if (isDraggingRightEdge()) original.setWidth (jmax (ValueType(), original.getWidth() + distance.x));
152 if (isDraggingTopEdge()) original.setTop (jmin (original.getBottom(), original.getY() + distance.y));
153 if (isDraggingBottomEdge()) original.setHeight (jmax (ValueType(), original.getHeight() + distance.y));

156 }
