41 public:
42 void alloc (AudioProcessor& processor)
43 {
44 const int numInputBuses = AudioUnitHelpers::getBusCount (processor, true);
45 const int numOutputBuses = AudioUnitHelpers::getBusCount (processor, false);
46
47 initializeChannelMapArray (processor, true, numInputBuses);
48 initializeChannelMapArray (processor, false, numOutputBuses);
49
50 for (int busIdx = 0; busIdx < numInputBuses; ++busIdx)
51 fillLayoutChannelMaps (processor, true, busIdx);
52
53 for (int busIdx = 0; busIdx < numOutputBuses; ++busIdx)
54 fillLayoutChannelMaps (processor, false, busIdx);
55 }

61 outputLayoutMapPtrStorage.free();
62 inputLayoutMapStorage.free();
63 outputLayoutMapStorage.free();
64 }

76 void initializeChannelMapArray (AudioProcessor& processor, bool isInput, const int numBuses)
77 {
78 HeapBlock<int*>& layoutMapPtrStorage = isInput ? inputLayoutMapPtrStorage : outputLayoutMapPtrStorage;
79 HeapBlock<int>& layoutMapStorage = isInput ? inputLayoutMapStorage : outputLayoutMapStorage;
80 int**& layoutMap = isInput ? inputLayoutMap : outputLayoutMap;
81
82 const int totalInChannels = processor.getTotalNumInputChannels();
83 const int totalOutChannels = processor.getTotalNumOutputChannels();
84
85 layoutMapPtrStorage.calloc (static_cast<size_t> (numBuses));
86 layoutMapStorage.calloc (static_cast<size_t> (isInput ? totalInChannels : totalOutChannels));
87
88 layoutMap = layoutMapPtrStorage.get();
89
90 int ch = 0;
91 for (int busIdx = 0; busIdx < numBuses; ++busIdx)
92 {
93 layoutMap[busIdx] = layoutMapStorage.get() + ch;
94 ch += processor.getChannelCountOfBus (isInput, busIdx);
95 }
96 }

98 void fillLayoutChannelMaps (AudioProcessor& processor, bool isInput, int busNr)
99 {
100 int* layoutMap = (isInput ? inputLayoutMap : outputLayoutMap)[busNr];
101 auto channelFormat = processor.getChannelLayoutOfBus (isInput, busNr);
102 AudioChannelLayout coreAudioLayout;
103
104 zerostruct (coreAudioLayout);
105 coreAudioLayout.mChannelLayoutTag = CoreAudioLayouts::toCoreAudio (channelFormat);
106
107 const int numChannels = channelFormat.size();
108 auto coreAudioChannels = CoreAudioLayouts::getCoreAudioLayoutChannels (coreAudioLayout);
109
110 for (int i = 0; i < numChannels; ++i)
111 layoutMap[i] = coreAudioChannels.indexOf (channelFormat.getTypeOfChannel (i));
112 }
113 };

172 void set (int bus, AudioBufferList& bufferList, const int* channelMap) noexcept
173 {
174 if (bufferList.mNumberBuffers <= 0 || ! isPositiveAndBelow (bus, inputBusOffsets.size() - 1))
175 return;
176
177 const auto n = (UInt32) (bufferList.mBuffers[0].mDataByteSize / (bufferList.mBuffers[0].mNumberChannels * sizeof (float)));
178 const auto isInterleaved = isAudioBufferInterleaved (bufferList);
179 const auto numChannels = (int) (isInterleaved ? bufferList.mBuffers[0].mNumberChannels
180 : bufferList.mNumberBuffers);
181
182 for (int ch = 0; ch < numChannels; ++ch)
183 {
184 float* data = channels[(size_t) (inputBusOffsets[(size_t) bus] + ch)];
185
186 const auto mappedChannel = channelMap[ch];
187
188 if (isInterleaved || static_cast<float*> (bufferList.mBuffers[mappedChannel].mData) != data)
189 copyAudioBuffer (bufferList, mappedChannel, n, data);
190 }
191 }

193 void get (int bus, AudioBufferList& buffer, const int* channelMap) noexcept
194 {
195 if (buffer.mNumberBuffers <= 0 || ! isPositiveAndBelow (bus, outputBusOffsets.size() - 1))
196 return;
197
198 const auto n = (UInt32) (buffer.mBuffers[0].mDataByteSize / (buffer.mBuffers[0].mNumberChannels * sizeof (float)));
199 const auto isInterleaved = isAudioBufferInterleaved (buffer);
200 const auto numChannels = (int) (isInterleaved ? buffer.mBuffers[0].mNumberChannels
201 : buffer.mNumberBuffers);
202
203 for (int ch = 0; ch < numChannels; ++ch)
204 {
205 float* data = channels[(size_t) (outputBusOffsets[(size_t) bus] + ch)];
206
207 const auto mappedChannel = channelMap[ch];
208
209 if (data == buffer.mBuffers[mappedChannel].mData && ! isInterleaved)
210 continue; // no copying necessary
211
212 if (buffer.mBuffers[mappedChannel].mData == nullptr && ! isInterleaved)
213 buffer.mBuffers[mappedChannel].mData = data;
214 else
215 copyAudioBuffer (data, mappedChannel, n, buffer);
216 }
217 }

246 float* uniqueBuffer (int idx, float* buffer) noexcept
247 {
248 if (buffer == nullptr)
249 return scratch.getWritePointer (idx);

252 if (buffer == channels[(size_t) ch])
253 return scratch.getWritePointer (idx);
254
255 return buffer;

275 static void copyAudioBuffer (const AudioBufferList& audioBuffer, const int channel, const UInt32 size, float* dst) noexcept
276 {
277 if (! isAudioBufferInterleaved (audioBuffer))
278 {
279 jassert (channel < static_cast<int> (audioBuffer.mNumberBuffers));
280 jassert (audioBuffer.mBuffers[channel].mDataByteSize == (size * sizeof (float)));
281
282 memcpy (dst, audioBuffer.mBuffers[channel].mData, size * sizeof (float));
283 }
284 else
285 {
286 const int numChannels = static_cast<int> (audioBuffer.mBuffers[0].mNumberChannels);
287 const UInt32 n = static_cast<UInt32> (numChannels) * size;
288 const float* src = static_cast<const float*> (audioBuffer.mBuffers[0].mData);
289
290 jassert (channel < numChannels);
291 jassert (audioBuffer.mBuffers[0].mDataByteSize == (n * sizeof (float)));
292
293 for (const float* inData = src; inData < (src + n); inData += numChannels)
294 *dst++ = inData[channel];
295 }
296 }

298 static void copyAudioBuffer (const float *src, const int channel, const UInt32 size, AudioBufferList& audioBuffer) noexcept
299 {
300 if (! isAudioBufferInterleaved (audioBuffer))
301 {
302 jassert (channel < static_cast<int> (audioBuffer.mNumberBuffers));
303 jassert (audioBuffer.mBuffers[channel].mDataByteSize == (size * sizeof (float)));
304
305 memcpy (audioBuffer.mBuffers[channel].mData, src, size * sizeof (float));
306 }
307 else
308 {
309 const int numChannels = static_cast<int> (audioBuffer.mBuffers[0].mNumberChannels);
310 const UInt32 n = static_cast<UInt32> (numChannels) * size;
311 float* dst = static_cast<float*> (audioBuffer.mBuffers[0].mData);
312
313 jassert (channel < numChannels);
314 jassert (audioBuffer.mBuffers[0].mDataByteSize == (n * sizeof (float)));
315
316 for (float* outData = dst; outData < (dst + n); outData += numChannels)
317 outData[channel] = *src++;
318 }

321 template <size_t numLayouts>
322 static bool isLayoutSupported (const AudioProcessor& processor,
323 bool isInput, int busIdx,
324 int numChannels,
325 const short (&channelLayoutList)[numLayouts][2],
326 bool hasLayoutMap = true)
327 {
328 if (const AudioProcessor::Bus* bus = processor.getBus (isInput, busIdx))
329 {
330 if (! bus->isNumberOfChannelsSupported (numChannels))
331 return false;
332
333 if (! hasLayoutMap)
334 return true;
335
336 const int numConfigs = sizeof (channelLayoutList) / sizeof (short[2]);
337
338 for (int i = 0; i < numConfigs; ++i)
339 {
340 if (channelLayoutList[i][isInput ? 0 : 1] == numChannels)
341 return true;
342 }
343 }
344
345 return false;
346 }

348 static Array<AUChannelInfo> getAUChannelInfo (const AudioProcessor& processor)
349 {
350 Array<AUChannelInfo> channelInfo;
351
352 auto hasMainInputBus = (AudioUnitHelpers::getBusCountForWrapper (processor, true) > 0);
353 auto hasMainOutputBus = (AudioUnitHelpers::getBusCountForWrapper (processor, false) > 0);
354
355 if ((! hasMainInputBus) && (! hasMainOutputBus))
356 {
357 // midi effect plug-in: no audio
358 AUChannelInfo info;
359 info.inChannels = 0;
360 info.outChannels = 0;
361
362 return { &info, 1 };
363 }
364
365 auto layout = processor.getBusesLayout();
366 auto maxNumChanToCheckFor = 9;
367
368 auto defaultInputs = processor.getChannelCountOfBus (true, 0);
369 auto defaultOutputs = processor.getChannelCountOfBus (false, 0);
370
371 struct Channels
372 {
373 SInt16 ins, outs;
374
375 std::pair<SInt16, SInt16> makePair() const noexcept { return std::make_pair (ins, outs); }
376
377 bool operator< (const Channels& other) const noexcept { return makePair() < other.makePair(); }
378 bool operator== (const Channels& other) const noexcept { return makePair() == other.makePair(); }
379 };
380
381 SortedSet<Channels> supportedChannels;
382
383 // add the current configuration
384 if (defaultInputs != 0 || defaultOutputs != 0)
385 supportedChannels.add ({ static_cast<SInt16> (defaultInputs),
386 static_cast<SInt16> (defaultOutputs) });
387
388 for (auto inChanNum = hasMainInputBus ? 1 : 0; inChanNum <= (hasMainInputBus ? maxNumChanToCheckFor : 0); ++inChanNum)
389 {
390 auto inLayout = layout;
391
392 if (auto* inBus = processor.getBus (true, 0))
393 if (! isNumberOfChannelsSupported (inBus, inChanNum, inLayout))
394 continue;
395
396 for (auto outChanNum = hasMainOutputBus ? 1 : 0; outChanNum <= (hasMainOutputBus ? maxNumChanToCheckFor : 0); ++outChanNum)
397 {
398 auto outLayout = inLayout;
399
400 if (auto* outBus = processor.getBus (false, 0))
401 if (! isNumberOfChannelsSupported (outBus, outChanNum, outLayout))
402 continue;
403
404 supportedChannels.add ({ static_cast<SInt16> (hasMainInputBus ? outLayout.getMainInputChannels() : 0),
405 static_cast<SInt16> (hasMainOutputBus ? outLayout.getMainOutputChannels() : 0) });
406 }
407 }
408
409 auto hasInOutMismatch = false;
410
411 for (const auto& supported : supportedChannels)
412 {
413 if (supported.ins != supported.outs)
414 {
415 hasInOutMismatch = true;
416 break;
417 }
418 }
419
420 auto hasUnsupportedInput = ! hasMainInputBus, hasUnsupportedOutput = ! hasMainOutputBus;
421
422 for (auto inChanNum = hasMainInputBus ? 1 : 0; inChanNum <= (hasMainInputBus ? maxNumChanToCheckFor : 0); ++inChanNum)
423 {
424 Channels channelConfiguration { static_cast<SInt16> (inChanNum),
425 static_cast<SInt16> (hasInOutMismatch ? defaultOutputs : inChanNum) };
426
427 if (! supportedChannels.contains (channelConfiguration))
428 {
429 hasUnsupportedInput = true;
430 break;
431 }
432 }
433
434 for (auto outChanNum = hasMainOutputBus ? 1 : 0; outChanNum <= (hasMainOutputBus ? maxNumChanToCheckFor : 0); ++outChanNum)
435 {
436 Channels channelConfiguration { static_cast<SInt16> (hasInOutMismatch ? defaultInputs : outChanNum),
437 static_cast<SInt16> (outChanNum) };
438
439 if (! supportedChannels.contains (channelConfiguration))
440 {
441 hasUnsupportedOutput = true;
442 break;
443 }
444 }
445
446 for (const auto& supported : supportedChannels)
447 {
448 AUChannelInfo info;
449
450 // see here: https://developer.apple.com/library/mac/documentation/MusicAudio/Conceptual/AudioUnitProgrammingGuide/TheAudioUnit/TheAudioUnit.html
451 info.inChannels = static_cast<SInt16> (hasMainInputBus ? (hasUnsupportedInput ? supported.ins : (hasInOutMismatch && (! hasUnsupportedOutput) ? -2 : -1)) : 0);
452 info.outChannels = static_cast<SInt16> (hasMainOutputBus ? (hasUnsupportedOutput ? supported.outs : (hasInOutMismatch && (! hasUnsupportedInput) ? -2 : -1)) : 0);
453
454 if (info.inChannels == -2 && info.outChannels == -2)
455 info.inChannels = -1;
456
457 int j;
458 for (j = 0; j < channelInfo.size(); ++j)
459 if (info.inChannels == channelInfo.getReference (j).inChannels
460 && info.outChannels == channelInfo.getReference (j).outChannels)
461 break;
462
463 if (j >= channelInfo.size())
464 channelInfo.add (info);
465 }
466
467 return channelInfo;
468 }

470 static bool isNumberOfChannelsSupported (const AudioProcessor::Bus* b, int numChannels, AudioProcessor::BusesLayout& inOutCurrentLayout)
471 {
472 auto potentialSets = AudioChannelSet::channelSetsWithNumberOfChannels (static_cast<int> (numChannels));
473
474 for (auto set : potentialSets)
475 {
476 auto copy = inOutCurrentLayout;
477
478 if (b->isLayoutSupported (set, &copy))
479 {
480 inOutCurrentLayout = copy;
481 return true;
482 }
483 }
484
485 return false;
486 }

490 {
491 int busCount = juceFilter.getBusCount (isInput);
492
493 #ifdef JucePlugin_PreferredChannelConfigurations
494 short configs[][2] = {JucePlugin_PreferredChannelConfigurations};

498

502

504 #endif
505
