52/* Holds a tresult and a pointer to an object.
53
54 Useful for holding intermediate results of calls to queryInterface.
55*/
56class QueryInterfaceResult
57{
58public:
59 QueryInterfaceResult() = default;
60
61 QueryInterfaceResult (Steinberg::tresult resultIn, void* ptrIn)
62 : result (resultIn), ptr (ptrIn) {}
63
64 bool isOk() const noexcept { return result == Steinberg::kResultOk; }
65
66 Steinberg::tresult extract (void** obj) const
67 {
68 *obj = result == Steinberg::kResultOk ? ptr : nullptr;
69 return result;
70 }
71
72private:
73 Steinberg::tresult result = Steinberg::kResultFalse;
74 void* ptr = nullptr;
75};

77/* Holds a tresult and a pointer to an object.
78
79 Calling InterfaceResultWithDeferredAddRef::extract() will also call addRef
80 on the pointed-to object. It is expected that users will use
81 InterfaceResultWithDeferredAddRef to hold intermediate results of a queryInterface
82 call. When a suitable interface is found, the function can be exited with
83 `return suitableInterface.extract (obj)`, which will set the obj pointer,
84 add a reference to the interface, and return the appropriate result code.
85*/
86class InterfaceResultWithDeferredAddRef
87{
88public:
89 InterfaceResultWithDeferredAddRef() = default;
90
91 template <typename Ptr>
92 InterfaceResultWithDeferredAddRef (Steinberg::tresult resultIn, Ptr* ptrIn)
93 : result (resultIn, ptrIn),
94 addRefFn (doAddRef<Ptr>) {}
95
96 bool isOk() const noexcept { return result.isOk(); }
97
98 Steinberg::tresult extract (void** obj) const
99 {
100 const auto toReturn = result.extract (obj);
101
102 if (result.isOk() && addRefFn != nullptr && *obj != nullptr)
103 addRefFn (*obj);
104
105 return toReturn;
106 }
107
108private:
109 template <typename Ptr>
110 static void doAddRef (void* obj) { static_cast<Ptr*> (obj)->addRef(); }
111
112 QueryInterfaceResult result;
113 void (*addRefFn) (void*) = nullptr;
114};

352static Steinberg::Vst::SpeakerArrangement getVst3SpeakerArrangement (const AudioChannelSet& channels) noexcept
353{
354 using namespace Steinberg::Vst::SpeakerArr;
355
356 if (channels == AudioChannelSet::disabled()) return kEmpty;
357 if (channels == AudioChannelSet::mono()) return kMono;
358 if (channels == AudioChannelSet::stereo()) return kStereo;
359 if (channels == AudioChannelSet::createLCR()) return k30Cine;
360 if (channels == AudioChannelSet::createLRS()) return k30Music;
361 if (channels == AudioChannelSet::createLCRS()) return k40Cine;
362 if (channels == AudioChannelSet::create5point0()) return k50;
363 if (channels == AudioChannelSet::create5point1()) return k51;
364 if (channels == AudioChannelSet::create6point0()) return k60Cine;
365 if (channels == AudioChannelSet::create6point1()) return k61Cine;
366 if (channels == AudioChannelSet::create6point0Music()) return k60Music;
367 if (channels == AudioChannelSet::create6point1Music()) return k61Music;
368 if (channels == AudioChannelSet::create7point0()) return k70Music;
369 if (channels == AudioChannelSet::create7point0SDDS()) return k70Cine;
370 if (channels == AudioChannelSet::create7point1()) return k71CineSideFill;
371 if (channels == AudioChannelSet::create7point1SDDS()) return k71Cine;
372 if (channels == AudioChannelSet::ambisonic()) return kAmbi1stOrderACN;
373 if (channels == AudioChannelSet::quadraphonic()) return k40Music;
374 if (channels == AudioChannelSet::create5point1point4()) return k51_4;
375 if (channels == AudioChannelSet::create7point0point2()) return k71_2 & ~(Steinberg::Vst::kSpeakerLfe);
376 if (channels == AudioChannelSet::create7point1point2()) return k71_2;
377 if (channels == AudioChannelSet::create7point0point4()) return k71_4 & ~(Steinberg::Vst::kSpeakerLfe);
378 if (channels == AudioChannelSet::create7point1point4()) return k71_4;
379 if (channels == AudioChannelSet::create7point1point6()) return k71_6;
380 if (channels == AudioChannelSet::create9point1point6()) return k91_6;
381 if (channels == AudioChannelSet::ambisonic (0)) return (1ull << 20);
382 if (channels == AudioChannelSet::ambisonic (1)) return (1ull << 20) | (1ull << 21) | (1ull << 22) | (1ull << 23);
383 #if VST_VERSION >= 0x030608
384 if (channels == AudioChannelSet::ambisonic (2)) return kAmbi2cdOrderACN;
385 if (channels == AudioChannelSet::ambisonic (3)) return kAmbi3rdOrderACN;
386 #endif
387
388 Steinberg::Vst::SpeakerArrangement result = 0;
389
390 for (const auto& type : channels.getChannelTypes())
391 result |= getSpeakerType (channels, type);
392
393 return result;
394}

396static AudioChannelSet getChannelSetForSpeakerArrangement (Steinberg::Vst::SpeakerArrangement arr) noexcept
397{
398 using namespace Steinberg::Vst::SpeakerArr;
399
400 switch (arr)
401 {
402 case kEmpty: return AudioChannelSet::disabled();
403 case kMono: return AudioChannelSet::mono();
404 case kStereo: return AudioChannelSet::stereo();
405 case k30Cine: return AudioChannelSet::createLCR();
406 case k30Music: return AudioChannelSet::createLRS();
407 case k40Cine: return AudioChannelSet::createLCRS();
408 case k50: return AudioChannelSet::create5point0();
409 case k51: return AudioChannelSet::create5point1();
410 case k60Cine: return AudioChannelSet::create6point0();
411 case k61Cine: return AudioChannelSet::create6point1();
412 case k60Music: return AudioChannelSet::create6point0Music();
413 case k61Music: return AudioChannelSet::create6point1Music();
414 case k70Music: return AudioChannelSet::create7point0();
415 case k70Cine: return AudioChannelSet::create7point0SDDS();
416 case k71CineSideFill: return AudioChannelSet::create7point1();
417 case k71Cine: return AudioChannelSet::create7point1SDDS();
418 case k40Music: return AudioChannelSet::quadraphonic();
419 case k71_2: return AudioChannelSet::create7point1point2();
420 case k71_2 & ~(Steinberg::Vst::kSpeakerLfe): return AudioChannelSet::create7point0point2();
421 case k71_4: return AudioChannelSet::create7point1point4();
422 case k71_4 & ~(Steinberg::Vst::kSpeakerLfe): return AudioChannelSet::create7point0point4();
423 case k71_6: return AudioChannelSet::create7point1point6();
424 case (1 << 20): return AudioChannelSet::ambisonic (0);
425 case kAmbi1stOrderACN: return AudioChannelSet::ambisonic (1);
426 #if VST_VERSION >= 0x030608
427 case kAmbi2cdOrderACN: return AudioChannelSet::ambisonic (2);
428 case kAmbi3rdOrderACN: return AudioChannelSet::ambisonic (3);
429 #endif
430 }
431
432 AudioChannelSet result;
433
434 BigInteger vstChannels (static_cast<int64> (arr));
435 for (auto bit = vstChannels.findNextSetBit (0); bit != -1; bit = vstChannels.findNextSetBit (bit + 1))
436 {
437 AudioChannelSet::ChannelType channelType = getChannelType (arr, 1ull << static_cast<uint64> (bit));
438 if (channelType != AudioChannelSet::unknown)
439 result.addChannel (channelType);
440 }
441
442 // VST3 <-> JUCE layout conversion error: report this bug to the JUCE forum
443 jassert (result.size() == vstChannels.countNumberOfSetBits());
444
445 return result;
446}

861 static BasicOptional<MidiMessage> toMidiMessage (const Steinberg::Vst::LegacyMIDICCOutEvent& e)
862 {
863 if (e.controlNumber <= 127)
864 return MidiMessage::controllerEvent (createSafeChannel (int16 (e.channel)),
865 createSafeNote (int16 (e.controlNumber)),
866 createSafeNote (int16 (e.value)));
867
868 switch (e.controlNumber)
869 {
870 case Steinberg::Vst::kAfterTouch:
871 return MidiMessage::channelPressureChange (createSafeChannel (int16 (e.channel)),
872 createSafeNote (int16 (e.value)));
873
874 case Steinberg::Vst::kPitchBend:
875 return MidiMessage::pitchWheel (createSafeChannel (int16 (e.channel)),
876 (e.value & 0x7f) | ((e.value2 & 0x7f) << 7));
877
878 case Steinberg::Vst::kCtrlProgramChange:
879 return MidiMessage::programChange (createSafeChannel (int16 (e.channel)),
880 createSafeNote (int16 (e.value)));
881
882 case Steinberg::Vst::kCtrlQuarterFrame:
883 return MidiMessage::quarterFrame (createSafeChannel (int16 (e.channel)),
884 createSafeNote (int16 (e.value)));
885
886 case Steinberg::Vst::kCtrlPolyPressure:
887 return MidiMessage::aftertouchChange (createSafeChannel (int16 (e.channel)),
888 createSafeNote (int16 (e.value)),
889 createSafeNote (int16 (e.value2)));
890
891 default:
892 // If this is hit, we're trying to convert a LegacyMIDICCOutEvent with an unknown controlNumber.
893 jassertfalse;
894 return {};
895 }
896 }
897
898 static BasicOptional<MidiMessage> toMidiMessage (const Steinberg::Vst::Event& e)
899 {
900 switch (e.type)
901 {
902 case Steinberg::Vst::Event::kNoteOnEvent:
903 return MidiMessage::noteOn (createSafeChannel (e.noteOn.channel),
904 createSafeNote (e.noteOn.pitch),
905 (Steinberg::uint8) denormaliseToMidiValue (e.noteOn.velocity));
906
907 case Steinberg::Vst::Event::kNoteOffEvent:
908 return MidiMessage::noteOff (createSafeChannel (e.noteOff.channel),
909 createSafeNote (e.noteOff.pitch),
910 (Steinberg::uint8) denormaliseToMidiValue (e.noteOff.velocity));
911
912 case Steinberg::Vst::Event::kPolyPressureEvent:
913 return MidiMessage::aftertouchChange (createSafeChannel (e.polyPressure.channel),
914 createSafeNote (e.polyPressure.pitch),
915 (Steinberg::uint8) denormaliseToMidiValue (e.polyPressure.pressure));
916
917 case Steinberg::Vst::Event::kDataEvent:
918 return MidiMessage::createSysExMessage (e.data.bytes, (int) e.data.size);
919
920 case Steinberg::Vst::Event::kLegacyMIDICCOutEvent:
921 return toMidiMessage (e.midiCCOut);
922
923 case Steinberg::Vst::Event::kNoteExpressionValueEvent:
924 case Steinberg::Vst::Event::kNoteExpressionTextEvent:
925 case Steinberg::Vst::Event::kChordEvent:
926 case Steinberg::Vst::Event::kScaleEvent:
927 return {};
928
929 default:
930 break;
931 }
932
933 // If this is hit, we've been sent an event type that doesn't exist in the VST3 spec.
934 jassertfalse;
935 return {};
936 }

1077
1078public:
1079 FloatCache() = default;
1080
1081 explicit FloatCache (size_t sizeIn)
1082 : values (sizeIn),
1083 flags (divCeil (sizeIn, numFlagBits))
1084 {
1085 std::fill (values.begin(), values.end(), 0.0f);
1086 std::fill (flags.begin(), flags.end(), 0);
1087 }
1088
1089 size_t size() const noexcept { return values.size(); }
1090
1091 void setWithoutNotifying (size_t index, float value)
1092 {
1093 jassert (index < size());

1095 }
1096
1097 void set (size_t index, float value)
1098 {
1099 jassert (index < size());
1100 const auto previous = values[index].exchange (value, std::memory_order_relaxed);
1101 const auto bit = previous == value ? ((FlagType) 0) : ((FlagType) 1 << (index % numFlagBits));
1102 flags[index / numFlagBits].fetch_or (bit, std::memory_order_acq_rel);
1103 }
1104
1105 float get (size_t index) const noexcept
1106 {
1107 jassert (index < size());
1108 return values[index].load (std::memory_order_relaxed);
1109 }

1111 /* Calls the supplied callback for any entries which have been modified
1112 since the last call to this function.
1113 */
1114 template <typename Callback>
1115 void ifSet (Callback&& callback)
1116 {
1117 for (size_t flagIndex = 0; flagIndex < flags.size(); ++flagIndex)
1118 {
1119 const auto prevFlags = flags[flagIndex].exchange (0, std::memory_order_acq_rel);
1120
1121 for (size_t bit = 0; bit < numFlagBits; ++bit)
1122 {
1123 if (prevFlags & ((FlagType) 1 << bit))
1124 {
1125 const auto itemIndex = (flagIndex * numFlagBits) + bit;
1126 callback (itemIndex, values[itemIndex].load (std::memory_order_relaxed));
1127 }
1128 }
1129 }
1130 }
1131
1132private:
1133 static constexpr size_t numFlagBits = 8 * sizeof (FlagType);
1134
1135 static constexpr size_t divCeil (size_t a, size_t b)
1136 {
1137 return (a / b) + ((a % b) != 0);
1138 }
1139
1140 std::vector<std::atomic<float>> values;
1141 std::vector<std::atomic<FlagType>> flags;
