159inline uint32 ByteOrder::swap (uint32 n) noexcept
160{
161 #if JUCE_MAC || JUCE_IOS
162 return OSSwapInt32 (n);
163 #elif (JUCE_GCC || JUCE_CLANG) && JUCE_INTEL && ! JUCE_NO_INLINE_ASM
164 asm("bswap %%eax" : "=a"(n) : "a"(n));
165 return n;
166 #elif JUCE_MSVC
167 return _byteswap_ulong (n);
168 #elif JUCE_ANDROID
169 return bswap_32 (n);
170 #else
171 return (n << 24) | (n >> 24) | ((n & 0xff00) << 8) | ((n & 0xff0000) >> 8);
172 #endif
173}

175inline uint64 ByteOrder::swap (uint64 value) noexcept
176{
177 #if JUCE_MAC || JUCE_IOS
178 return OSSwapInt64 (value);
179 #elif JUCE_MSVC
180 return _byteswap_uint64 (value);
181 #else
182 return (((uint64) swap ((uint32) value)) << 32) | swap ((uint32) (value >> 32));
