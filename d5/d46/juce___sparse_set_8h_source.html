144 void addRange (Range<Type> range)
145 {
146 if (! range.isEmpty())
147 {
148 removeRange (range);
149 ranges.add (range);
150 std::sort (ranges.begin(), ranges.end(),
151 [] (Range<Type> a, Range<Type> b) { return a.getStart() < b.getStart(); });
152 simplify();
153 }
154 }
155
159 void removeRange (Range<Type> rangeToRemove)
160 {
161 if (getTotalRange().intersects (rangeToRemove) && ! rangeToRemove.isEmpty())
162 {
163 for (int i = ranges.size(); --i >= 0;)
164 {
165 auto& r = ranges.getReference(i);
166
167 if (r.getEnd() <= rangeToRemove.getStart())
168 break;
169
170 if (r.getStart() >= rangeToRemove.getEnd())
171 continue;
172
173 if (rangeToRemove.contains (r))
174 {
175 ranges.remove (i);
176 }
177 else if (r.contains (rangeToRemove))
178 {
179 auto r1 = r.withEnd (rangeToRemove.getStart());
180 auto r2 = r.withStart (rangeToRemove.getEnd());
181
182 // this should be covered in if (rangeToRemove.contains (r))
183 jassert (! r1.isEmpty() || ! r2.isEmpty());
184
185 r = r1;
186
187 if (r.isEmpty())
188 r = r2;
189
190 if (! r1.isEmpty() && ! r2.isEmpty())
191 ranges.insert (i + 1, r2);
192 }
193 else if (rangeToRemove.getEnd() > r.getEnd())
194 {
195 r.setEnd (rangeToRemove.getStart());
196 }
197 else
198 {
199 r.setStart (rangeToRemove.getEnd());
200 }
201 }
202 }
203 }

206 void invertRange (Range<Type> range)
207 {
208 SparseSet newItems;
209 newItems.addRange (range);
210
211 for (auto& r : ranges)
212 newItems.removeRange (r);
213
214 removeRange (range);
215

253 void simplify()
254 {
255 for (int i = ranges.size(); --i > 0;)
256 {
257 auto& r1 = ranges.getReference (i - 1);
258 auto& r2 = ranges.getReference (i);
259
260 if (r1.getEnd() == r2.getStart())
261 {
262 r1.setEnd (r2.getEnd());
263 ranges.remove (i);
264 }
265 }
