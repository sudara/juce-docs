38inline uint32 maskPixelComponents (uint32 x) noexcept
39{
40 return (x >> 8) & 0x00ff00ff;
41}

43inline uint32 clampPixelComponents (uint32 x) noexcept
44{
45 return (x | (0x01000100 - maskPixelComponents (x))) & 0x00ff00ff;

186 {
187 auto dEvenBytes = getEvenBytes();
188 dEvenBytes += (((src.getEvenBytes() - dEvenBytes) * amount) >> 8);
189 dEvenBytes &= 0x00ff00ff;
190
191 auto dOddBytes = getOddBytes();
192 dOddBytes += (((src.getOddBytes() - dOddBytes) * amount) >> 8);
193 dOddBytes &= 0x00ff00ff;
194 dOddBytes <<= 8;
195
196 dOddBytes |= dEvenBytes;
197 internal = dOddBytes;

232 forcedinline void premultiply() noexcept
233 {
234 const auto alpha = components.a;
235
236 if (alpha < 0xff)
237 {
238 if (alpha == 0)
239 {
240 components.b = 0;
241 components.g = 0;
242 components.r = 0;
243 }
244 else
245 {
246 components.b = (uint8) ((components.b * alpha + 0x7f) >> 8);
247 components.g = (uint8) ((components.g * alpha + 0x7f) >> 8);
248 components.r = (uint8) ((components.r * alpha + 0x7f) >> 8);
249 }
250 }
251 }
252
254 forcedinline void unpremultiply() noexcept
255 {
256 const auto alpha = components.a;
257
258 if (alpha < 0xff)
259 {
260 if (alpha == 0)
261 {
262 components.b = 0;
263 components.g = 0;
264 components.r = 0;
265 }
266 else
267 {
268 components.b = (uint8) jmin ((uint32) 0xffu, (components.b * 0xffu) / alpha);
269 components.g = (uint8) jmin ((uint32) 0xffu, (components.g * 0xffu) / alpha);
270 components.r = (uint8) jmin ((uint32) 0xffu, (components.r * 0xffu) / alpha);
271 }
272 }
273 }
274
275 forcedinline void desaturate() noexcept
276 {
277 if (components.a < 0xff && components.a > 0)
278 {
279 const auto newUnpremultipliedLevel = (0xff * ((int) components.r + (int) components.g + (int) components.b) / (3 * components.a));
280
281 components.r = components.g = components.b
282 = (uint8) ((newUnpremultipliedLevel * components.a + 0x7f) >> 8);
283 }
284 else
285 {
286 components.r = components.g = components.b
287 = (uint8) (((int) components.r + (int) components.g + (int) components.b) / 3);
288 }
289 }

452 template <class Pixel>
453 forcedinline void blend (const Pixel& src) noexcept
454 {
455 const auto alpha = (uint32) (0x100 - src.getAlpha());
456
457 // getEvenBytes returns 0x00rr00bb on non-android
458 const auto rb = clampPixelComponents (src.getEvenBytes() + maskPixelComponents (getEvenBytes() * alpha));
459 // getOddBytes returns 0x00aa00gg on non-android
460 const auto ag = clampPixelComponents (src.getOddBytes() + ((g * alpha) >> 8));
461
462 g = (uint8) (ag & 0xff);
463
464 #if JUCE_ANDROID
465 b = (uint8) (rb >> 16);
466 r = (uint8) (rb & 0xff);
467 #else
468 r = (uint8) (rb >> 16);
469 b = (uint8) (rb & 0xff);
470 #endif
471 }

483 template <class Pixel>
484 forcedinline void blend (const Pixel& src, uint32 extraAlpha) noexcept
485 {
486 auto ag = maskPixelComponents (extraAlpha * src.getOddBytes());
487 auto rb = maskPixelComponents (extraAlpha * src.getEvenBytes());
488
489 const auto alpha = 0x100 - (ag >> 16);
490
491 ag = clampPixelComponents (ag + (g * alpha >> 8));
492 rb = clampPixelComponents (rb + maskPixelComponents (getEvenBytes() * alpha));
493
494 g = (uint8) (ag & 0xff);
495
496 #if JUCE_ANDROID
497 b = (uint8) (rb >> 16);
498 r = (uint8) (rb & 0xff);
499 #else
500 r = (uint8) (rb >> 16);
501 b = (uint8) (rb & 0xff);
502 #endif
503 }

509 forcedinline void tween (const Pixel& src, uint32 amount) noexcept
510 {
511 auto dEvenBytes = getEvenBytes();
512 dEvenBytes += (((src.getEvenBytes() - dEvenBytes) * amount) >> 8);
513
514 auto dOddBytes = getOddBytes();
515 dOddBytes += (((src.getOddBytes() - dOddBytes) * amount) >> 8);
516
517 g = (uint8) (dOddBytes & 0xff); // dOddBytes = 0x00aa00gg
518
519 #if JUCE_ANDROID
520 r = (uint8) (dEvenBytes & 0xff); // dEvenBytes = 0x00bb00rr
521 b = (uint8) (dEvenBytes >> 16);
522 #else
523 b = (uint8) (dEvenBytes & 0xff); // dEvenBytes = 0x00rr00bb
524 r = (uint8) (dEvenBytes >> 16);
525 #endif
