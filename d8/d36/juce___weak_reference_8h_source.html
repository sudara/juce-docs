150 class Master
151 {
152 public:
153 Master() = default;
154
155 ~Master() noexcept
156 {
157 // You must remember to call clear() in your source object's destructor! See the notes
158 // for the WeakReference class for an example of how to do this.
159 jassert (sharedPointer == nullptr || sharedPointer->get() == nullptr);
160 }
161
165 SharedRef getSharedPointer (ObjectType* object)
166 {
167 if (sharedPointer == nullptr)
168 {
169 sharedPointer = *new SharedPointer (object);
170 }
171 else
172 {
173 // You're trying to create a weak reference to an object that has already been deleted!!
174 jassert (sharedPointer->get() != nullptr);
175 }
176
177 return sharedPointer;
178 }
179
184 void clear() noexcept
185 {
186 if (sharedPointer != nullptr)
187 sharedPointer->clearPointer();
188 }
189
191 int getNumActiveWeakReferences() const noexcept
192 {
193 return sharedPointer == nullptr ? 0 : (sharedPointer->getReferenceCount() - 1);
194 }
195
196 private:
197 SharedRef sharedPointer;
198
199 JUCE_DECLARE_NON_COPYABLE (Master)
200 };
201
202private:
203 SharedRef holder;
204
205 static SharedRef getRef (ObjectType* o)
206 {
207 if (o != nullptr)
208 return o->masterReference.getSharedPointer (o);
209
210 return {};
211 }
212};
