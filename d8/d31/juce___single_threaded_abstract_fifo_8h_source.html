64class SingleThreadedAbstractFifo
65{
66public:
68 SingleThreadedAbstractFifo() = default;
69
71 explicit SingleThreadedAbstractFifo (int sizeIn)
72 : size (sizeIn)
73 {
74 // This class only works properly when the size is a power of two.
75 // Use nextPowerOfTwo() to find a good size, and ensure that your
76 // backing storage is the same size.
77 jassert (isPowerOfTwo (sizeIn));
78 }
79
81 int getRemainingSpace() const { return size - numReadable; }
82
84 int getNumReadable() const { return numReadable; }
85
87 int getSize() const { return size; }
88
94 std::array<Range<int>, 2> write (int num)
95 {
96 const auto startPos = (readPos + numReadable) & (size - 1);
97 const auto maxToWrite = jmin (getRemainingSpace(), num);
98 const auto firstBlockSize = jmin (maxToWrite, size - startPos);
99
100 numReadable += maxToWrite;
101
102 return { { { startPos, startPos + firstBlockSize }, { 0, maxToWrite - firstBlockSize } } };
103 }
104
110 std::array<Range<int>, 2> read (int num)
111 {
112 const auto startPos = readPos;
113 const auto maxToRead = jmin (numReadable, num);
114 const auto firstBlockSize = jmin (maxToRead, size - startPos);
115
116 readPos = (startPos + maxToRead) & (size - 1);
117 numReadable -= maxToRead;
118
119 return { { { startPos, startPos + firstBlockSize }, { 0, maxToRead - firstBlockSize } } };
120 }
121
122private:
123 int size = 0, readPos = 0, numReadable = 0;
124};
