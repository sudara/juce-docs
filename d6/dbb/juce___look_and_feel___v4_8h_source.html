62 template <typename... ItemColours>
63 ColourScheme (ItemColours... coloursToUse)
64 {
65 static_assert (sizeof... (coloursToUse) == numColours, "Must supply one colour for each UIColour item");
66 const Colour c[] = { Colour (coloursToUse)... };
67
68 for (int i = 0; i < numColours; ++i)
69 palette[i] = c[i];
70 }
71

73 ColourScheme& operator= (const ColourScheme&) = default;

88 };
