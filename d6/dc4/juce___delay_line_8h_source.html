172 template <typename ProcessContext>
173 void process (const ProcessContext& context) noexcept
174 {
175 const auto& inputBlock = context.getInputBlock();
176 auto& outputBlock = context.getOutputBlock();
177 const auto numChannels = outputBlock.getNumChannels();
178 const auto numSamples = outputBlock.getNumSamples();
179
180 jassert (inputBlock.getNumChannels() == numChannels);
181 jassert (inputBlock.getNumChannels() == writePos.size());
182 jassert (inputBlock.getNumSamples() == numSamples);
183
184 if (context.isBypassed)
185 {
186 outputBlock.copyFrom (inputBlock);
187 return;
188 }
189
190 for (size_t channel = 0; channel < numChannels; ++channel)
191 {
192 auto* inputSamples = inputBlock.getChannelPointer (channel);
193 auto* outputSamples = outputBlock.getChannelPointer (channel);
194
195 for (size_t i = 0; i < numSamples; ++i)
196 {
197 pushSample ((int) channel, inputSamples[i]);
198 outputSamples[i] = popSample ((int) channel);
199 }
200 }
201 }

215 interpolateSample (int channel) const
216 {
217 auto index1 = readPos[(size_t) channel] + delayInt;
218 auto index2 = index1 + 1;
219
220 if (index2 >= totalSize)

232 template <typename T = InterpolationType>
233 typename std::enable_if <std::is_same <T, DelayLineInterpolationTypes::Lagrange3rd>::value, SampleType>::type
234 interpolateSample (int channel) const
235 {
236 auto index1 = readPos[(size_t) channel] + delayInt;
237 auto index2 = index1 + 1;
238 auto index3 = index2 + 1;
239 auto index4 = index3 + 1;
240
241 if (index4 >= totalSize)
242 {
243 index1 %= totalSize;
244 index2 %= totalSize;
245 index3 %= totalSize;
246 index4 %= totalSize;
247 }
248
249 auto* samples = bufferData.getReadPointer (channel);
250
251 auto value1 = samples[index1];
252 auto value2 = samples[index2];
253 auto value3 = samples[index3];
254 auto value4 = samples[index4];
255
256 auto d1 = delayFrac - 1.f;
257 auto d2 = delayFrac - 2.f;
258 auto d3 = delayFrac - 3.f;
259
260 auto c1 = -d1 * d2 * d3 / 6.f;
261 auto c2 = d2 * d3 * 0.5f;
262 auto c3 = -d1 * d3 * 0.5f;
263 auto c4 = d1 * d2 / 6.f;
264
265 return value1 * c1 + delayFrac * (value2 * c2 + value3 * c3 + value4 * c4);
266 }

268 template <typename T = InterpolationType>
269 typename std::enable_if <std::is_same <T, DelayLineInterpolationTypes::Thiran>::value, SampleType>::type
270 interpolateSample (int channel)
271 {
272 auto index1 = readPos[(size_t) channel] + delayInt;
273 auto index2 = index1 + 1;
274
275 if (index2 >= totalSize)
276 {
277 index1 %= totalSize;
278 index2 %= totalSize;

280
281 auto value1 = bufferData.getSample (channel, index1);
282 auto value2 = bufferData.getSample (channel, index2);
283
284 auto output = delayFrac == 0 ? value1 : value2 + alpha * (value1 - v[(size_t) channel]);
285 v[(size_t) channel] = output;
