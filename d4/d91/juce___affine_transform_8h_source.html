75 template <typename ValueType>
76 void transformPoint (ValueType& x, ValueType& y) const noexcept
77 {
78 auto oldX = x;
79 x = static_cast<ValueType> (mat00 * oldX + mat01 * y + mat02);
80 y = static_cast<ValueType> (mat10 * oldX + mat11 * y + mat12);
81 }
82
88 template <typename ValueType>
89 void transformPoints (ValueType& x1, ValueType& y1,
90 ValueType& x2, ValueType& y2) const noexcept
91 {
92 auto oldX1 = x1, oldX2 = x2;
93 x1 = static_cast<ValueType> (mat00 * oldX1 + mat01 * y1 + mat02);
94 y1 = static_cast<ValueType> (mat10 * oldX1 + mat11 * y1 + mat12);
95 x2 = static_cast<ValueType> (mat00 * oldX2 + mat01 * y2 + mat02);
96 y2 = static_cast<ValueType> (mat10 * oldX2 + mat11 * y2 + mat12);
97 }

104 template <typename ValueType>

106 ValueType& x2, ValueType& y2,
107 ValueType& x3, ValueType& y3) const noexcept
108 {
109 auto oldX1 = x1, oldX2 = x2, oldX3 = x3;
110 x1 = static_cast<ValueType> (mat00 * oldX1 + mat01 * y1 + mat02);
111 y1 = static_cast<ValueType> (mat10 * oldX1 + mat11 * y1 + mat12);

113 y2 = static_cast<ValueType> (mat10 * oldX2 + mat11 * y2 + mat12);
114 x3 = static_cast<ValueType> (mat00 * oldX3 + mat01 * y3 + mat02);
115 y3 = static_cast<ValueType> (mat10 * oldX3 + mat11 * y3 + mat12);
116 }
