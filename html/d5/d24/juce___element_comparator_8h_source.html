79template <class ElementType, class ElementComparator>
80static void sortArray (ElementComparator& comparator,
81 ElementType* const array,
82 int firstElement,
83 int lastElement,
84 const bool retainOrderOfEquivalentItems)
85{
86 jassert (firstElement >= 0);
87
88 if (lastElement > firstElement)
89 {
90 SortFunctionConverter<ElementComparator> converter (comparator);
91
92 if (retainOrderOfEquivalentItems)
93 std::stable_sort (array + firstElement, array + lastElement + 1, converter);
94 else
95 std::sort (array + firstElement, array + lastElement + 1, converter);
96 }
97}

124template <class ElementType, class ElementComparator>
125static int findInsertIndexInSortedArray (ElementComparator& comparator,
126 ElementType* const array,
127 const ElementType newElement,
128 int firstElement,
129 int lastElement)
130{
131 jassert (firstElement <= lastElement);
132
133 ignoreUnused (comparator); // if you pass in an object with a static compareElements() method, this
134 // avoids getting warning messages about the parameter being unused
135
136 while (firstElement < lastElement)
137 {
138 if (comparator.compareElements (newElement, array [firstElement]) == 0)
139 {
140 ++firstElement;
141 break;
142 }
143 else
144 {
145 const int halfway = (firstElement + lastElement) >> 1;
146
147 if (halfway == firstElement)
148 {
149 if (comparator.compareElements (newElement, array [halfway]) >= 0)
150 ++firstElement;
151
152 break;
153 }
154 else if (comparator.compareElements (newElement, array [halfway]) >= 0)
155 {
156 firstElement = halfway;
157 }
158 else
159 {
160 lastElement = halfway;
161 }
162 }
163 }
164
165 return firstElement;
166}
