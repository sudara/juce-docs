277 JUCE_COMRESULT FindText (BSTR text, BOOL backward, BOOL ignoreCase,
278 ComTypes::ITextRangeProvider** pRetVal) override
279 {
280 return owner->withTextInterface (pRetVal, [&] (const AccessibilityTextInterface& textInterface)
281 {
282 auto selectionText = textInterface.getText (selectionRange);
283 String textToSearchFor (text);
284
285 auto offset = (backward ? (ignoreCase ? selectionText.lastIndexOfIgnoreCase (textToSearchFor) : selectionText.lastIndexOf (textToSearchFor))
286 : (ignoreCase ? selectionText.indexOfIgnoreCase (textToSearchFor) : selectionText.indexOf (textToSearchFor)));
287
288 if (offset != -1)
289 *pRetVal = new UIATextRangeProvider (*owner, { offset, offset + textToSearchFor.length() });

295 JUCE_COMRESULT GetAttributeValue (TEXTATTRIBUTEID attributeId, VARIANT* pRetVal) override
296 {
297 return owner->withTextInterface (pRetVal, [&] (const AccessibilityTextInterface& textInterface)
298 {
299 VariantHelpers::clear (pRetVal);
300
301 switch (attributeId)
302 {
303 case ComTypes::UIA_IsReadOnlyAttributeId:
304 {
305 VariantHelpers::setBool (textInterface.isReadOnly(), pRetVal);
306 break;
307 }
308
309 case ComTypes::UIA_CaretPositionAttributeId:
310 {
311 auto cursorPos = textInterface.getTextInsertionOffset();
312
313 auto caretPos = [&]
314 {
315 if (cursorPos == 0)
316 return ComTypes::CaretPosition_BeginningOfLine;
317
318 if (cursorPos == textInterface.getTotalNumCharacters())
319 return ComTypes::CaretPosition_EndOfLine;
320
321 return ComTypes::CaretPosition_Unknown;
322 }();

333 JUCE_COMRESULT GetBoundingRectangles (SAFEARRAY** pRetVal) override
334 {
335 return owner->withTextInterface (pRetVal, [&] (const AccessibilityTextInterface& textInterface)
336 {
337 auto rectangleList = textInterface.getTextBounds (selectionRange);
338 auto numRectangles = rectangleList.getNumRectangles();
339
340 *pRetVal = SafeArrayCreateVector (VT_R8, 0, 4 * (ULONG) numRectangles);
341
342 if (*pRetVal == nullptr)
343 return E_FAIL;
344
345 if (numRectangles > 0)
346 {
347 double* doubleArr = nullptr;
348
349 if (FAILED (SafeArrayAccessData (*pRetVal, reinterpret_cast<void**> (&doubleArr))))
350 {
351 SafeArrayDestroy (*pRetVal);
352 return E_FAIL;
353 }
354
355 for (int i = 0; i < numRectangles; ++i)
356 {
357 auto r = Desktop::getInstance().getDisplays().logicalToPhysical (rectangleList.getRectangle (i));
358
359 doubleArr[i * 4] = r.getX();
360 doubleArr[i * 4 + 1] = r.getY();
361 doubleArr[i * 4 + 2] = r.getWidth();
362 doubleArr[i * 4 + 3] = r.getHeight();
363 }
364
365 if (FAILED (SafeArrayUnaccessData (*pRetVal)))
366 {
367 SafeArrayDestroy (*pRetVal);
368 return E_FAIL;
369 }
370 }
371
372 return S_OK;
373 });
374 }

431 JUCE_COMRESULT MoveEndpointByRange (ComTypes::TextPatternRangeEndpoint endpoint,
432 ComTypes::ITextRangeProvider* targetRange,
433 ComTypes::TextPatternRangeEndpoint targetEndpoint) override
434 {
435 if (targetRange == nullptr)
436 return E_INVALIDARG;
437
438 if (! isElementValid())
439 return (HRESULT) UIA_E_ELEMENTNOTAVAILABLE;
440

444 auto targetPoint = (targetEndpoint == ComTypes::TextPatternRangeEndpoint_Start ? otherRange.getStart()

454 JUCE_COMRESULT MoveEndpointByUnit (ComTypes::TextPatternRangeEndpoint endpoint,
455 ComTypes::TextUnit unit,
456 int count,
457 int* pRetVal) override
458 {
459 return owner->withTextInterface (pRetVal, [&] (const AccessibilityTextInterface& textInterface)
460 {
461 if (count == 0 || textInterface.getTotalNumCharacters() == 0)
462 return S_OK;
463
464 auto endpointToMove = (endpoint == ComTypes::TextPatternRangeEndpoint_Start ? selectionRange.getStart()
465 : selectionRange.getEnd());
466
467 const auto direction = (count > 0 ? AccessibilityTextHelpers::Direction::forwards
468 : AccessibilityTextHelpers::Direction::backwards);
469
470 const auto boundaryType = getBoundaryType (unit);
471
472 // handle case where endpoint is on a boundary
473 if (AccessibilityTextHelpers::findTextBoundary (textInterface, endpointToMove, boundaryType, direction) == endpointToMove)
474 endpointToMove += (direction == AccessibilityTextHelpers::Direction::forwards ? 1 : -1);
475
476 int numMoved;
477 for (numMoved = 0; numMoved < std::abs (count); ++numMoved)
478 {
479 auto nextEndpoint = AccessibilityTextHelpers::findTextBoundary (textInterface,
480 endpointToMove,
481 boundaryType,
482 direction);
483
484 if (nextEndpoint == endpointToMove)
485 break;
486
487 endpointToMove = nextEndpoint;
488 }
489
490 *pRetVal = numMoved;
491 setEndpointChecked (endpoint, endpointToMove);
492
493 return S_OK;
494 });
495 }

536 static AccessibilityTextHelpers::BoundaryType getBoundaryType (ComTypes::TextUnit unit)
537 {
538 switch (unit)
539 {
540 case ComTypes::TextUnit_Character:
541 return AccessibilityTextHelpers::BoundaryType::character;
542
543 case ComTypes::TextUnit_Format:
544 case ComTypes::TextUnit_Word:

546
547 case ComTypes::TextUnit_Line:
548 return AccessibilityTextHelpers::BoundaryType::line;
549
550 case ComTypes::TextUnit_Paragraph:
551 case ComTypes::TextUnit_Page:
552 case ComTypes::TextUnit_Document:

560 void setEndpointChecked (ComTypes::TextPatternRangeEndpoint endpoint, int newEndpoint)
561 {
562 if (endpoint == ComTypes::TextPatternRangeEndpoint_Start)
563 {
564 if (selectionRange.getEnd() < newEndpoint)
565 selectionRange.setEnd (newEndpoint);
566
567 selectionRange.setStart (newEndpoint);
568 }
569 else
570 {
571 if (selectionRange.getStart() > newEndpoint)
572 selectionRange.setStart (newEndpoint);
573
574 selectionRange.setEnd (newEndpoint);
575 }
576 }
