730 template <typename ElementComparator, typename TargetValueType>
731 int indexOfSorted (ElementComparator& comparator, TargetValueType elementToLookFor) const
732 {
733 ignoreUnused (comparator); // if you pass in an object with a static compareElements() method, this
734 // avoids getting warning messages about the parameter being unused
735
736 const ScopedLockType lock (getLock());
737
738 for (int s = 0, e = values.size();;)
739 {
740 if (s >= e)
741 return -1;
742
743 if (comparator.compareElements (elementToLookFor, values[s]) == 0)
744 return s;
745
746 auto halfway = (s + e) / 2;
747
748 if (halfway == s)
749 return -1;
750
751 if (comparator.compareElements (elementToLookFor, values[halfway]) >= 0)
752 s = halfway;
753 else
754 e = halfway;
755 }
756 }
