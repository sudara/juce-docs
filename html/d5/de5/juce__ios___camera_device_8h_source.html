33 using InternalOpenCameraResultCallback = std::function<void (const String& /*cameraId*/, const String& /*error*/)>;
34
35 Pimpl (CameraDevice& ownerToUse, const String& cameraIdToUse, int /*index*/,
36 int /*minWidth*/, int /*minHeight*/, int /*maxWidth*/, int /*maxHeight*/,
37 bool useHighQuality)
38 : owner (ownerToUse),
39 cameraId (cameraIdToUse),
40 captureSession (*this, useHighQuality)
41 {
42 }
43
44 String getCameraId() const noexcept { return cameraId; }
45
46 void open (InternalOpenCameraResultCallback cameraOpenCallbackToUse)
47 {
48 cameraOpenCallback = std::move (cameraOpenCallbackToUse);
49
50 if (cameraOpenCallback == nullptr)
51 {
52 // A valid camera open callback must be passed.
53 jassertfalse;
54 return;
55 }
56
57 [AVCaptureDevice requestAccessForMediaType: AVMediaTypeVideo
58 completionHandler: ^(BOOL granted)
59 {
60 // Access to video is required for camera to work,
61 // black images will be produced otherwise!
62 jassertquiet (granted);
63 }];
64
65 [AVCaptureDevice requestAccessForMediaType: AVMediaTypeAudio
66 completionHandler: ^(BOOL granted)
67 {
68 // Access to audio is required for camera to work,
69 // silence will be produced otherwise!
70 jassertquiet (granted);
71 }];
72
73 captureSession.startSessionForDeviceWithId (cameraId);
74 }
75
76 bool openedOk() const noexcept { return captureSession.openedOk(); }
77
78 void takeStillPicture (std::function<void (const Image&)> pictureTakenCallbackToUse)
79 {
80 if (pictureTakenCallbackToUse == nullptr)
81 {
82 jassertfalse;
83 return;
84 }
85
86 pictureTakenCallback = std::move (pictureTakenCallbackToUse);
87
88 triggerStillPictureCapture();
89 }
90
91 void startRecordingToFile (const File& file, int /*quality*/)
92 {
93 file.deleteFile();
94
95 captureSession.startRecording (file);
96 }
97
98 void stopRecording()
99 {
100 captureSession.stopRecording();
101 }

324 class CaptureSession
325 {
326 public:
327 CaptureSession (Pimpl& ownerToUse, bool useHighQuality)
328 : owner (ownerToUse),
329 captureSessionQueue (dispatch_queue_create ("JuceCameraDeviceBackgroundDispatchQueue", DISPATCH_QUEUE_SERIAL)),
330 captureSession ([[AVCaptureSession alloc] init]),
331 delegate (nullptr),
332 stillPictureTaker (*this),
333 videoRecorder (*this)
334 {
335 static SessionDelegateClass cls;
336 delegate.reset ([cls.createInstance() init]);
337 SessionDelegateClass::setOwner (delegate.get(), this);
338
339 JUCE_BEGIN_IGNORE_WARNINGS_GCC_LIKE ("-Wundeclared-selector")
340 [[NSNotificationCenter defaultCenter] addObserver: delegate.get()
341 selector: @selector (sessionDidStartRunning:)
342 name: AVCaptureSessionDidStartRunningNotification
343 object: captureSession.get()];
344
345 [[NSNotificationCenter defaultCenter] addObserver: delegate.get()
346 selector: @selector (sessionDidStopRunning:)
347 name: AVCaptureSessionDidStopRunningNotification
348 object: captureSession.get()];
349
350 [[NSNotificationCenter defaultCenter] addObserver: delegate.get()
351 selector: @selector (runtimeError:)
352 name: AVCaptureSessionRuntimeErrorNotification
353 object: captureSession.get()];
354
355 [[NSNotificationCenter defaultCenter] addObserver: delegate.get()
356 selector: @selector (sessionWasInterrupted:)
357 name: AVCaptureSessionWasInterruptedNotification
358 object: captureSession.get()];
359
360 [[NSNotificationCenter defaultCenter] addObserver: delegate.get()
361 selector: @selector (sessionInterruptionEnded:)
362 name: AVCaptureSessionInterruptionEndedNotification
363 object: captureSession.get()];
364 JUCE_END_IGNORE_WARNINGS_GCC_LIKE
365
366 dispatch_async (captureSessionQueue,^
367 {
368 [captureSession.get() setSessionPreset: useHighQuality ? AVCaptureSessionPresetHigh
369 : AVCaptureSessionPresetMedium];
370 });
371
372 ++numCaptureSessions;
373 }
374
375 ~CaptureSession()
376 {
377 [[NSNotificationCenter defaultCenter] removeObserver: delegate.get()];
378
379 stopRecording();
380
381 if (--numCaptureSessions == 0)
382 {
383 dispatch_async (captureSessionQueue, ^
384 {
385 if (captureSession.get().running)
386 [captureSession.get() stopRunning];
387
388 sessionClosedEvent.signal();
389 });
390
391 sessionClosedEvent.wait (-1);
392 }
393 }
394
395 bool openedOk() const noexcept { return sessionStarted; }
396
397 void startSessionForDeviceWithId (const String& cameraIdToUse)
398 {
399 dispatch_async (captureSessionQueue,^
400 {
401 cameraDevice = [AVCaptureDevice deviceWithUniqueID: juceStringToNS (cameraIdToUse)];
402 auto audioDevice = [AVCaptureDevice defaultDeviceWithMediaType: AVMediaTypeAudio];
403
404 [captureSession.get() beginConfiguration];
405
406 // This will add just video...
407 auto error = addInputToDevice (cameraDevice);
408
409 if (error.isNotEmpty())
410 {
411 MessageManager::callAsync ([weakRef = WeakReference<CaptureSession> { this }, error]() mutable
412 {
413 if (weakRef != nullptr)
414 weakRef->owner.cameraOpenCallback ({}, error);
415 });
416
417 return;
418 }
419
420 // ... so add audio explicitly here
421 error = addInputToDevice (audioDevice);
422
423 if (error.isNotEmpty())
424 {
425 MessageManager::callAsync ([weakRef = WeakReference<CaptureSession> { this }, error]() mutable
426 {
427 if (weakRef != nullptr)
428 weakRef->owner.cameraOpenCallback ({}, error);
429 });
430
431 return;
432 }
433
434 [captureSession.get() commitConfiguration];
435
436 if (! captureSession.get().running)
437 [captureSession.get() startRunning];
438 });
439 }

499 String addInputToDevice (AVCaptureDevice* device)
500 {
501 NSError* error = nil;
502
503 auto input = [AVCaptureDeviceInput deviceInputWithDevice: device
504 error: &error];
505
506 if (error != nil)
507 return nsStringToJuce (error.localizedDescription);
508
509 if (! [captureSession.get() canAddInput: input])
510 return "Could not add input to camera session.";
511
512 [captureSession.get() addInput: input];
513 return {};
514 }

590 StillPictureTaker (CaptureSession& cs)

606
607 void takePicture (AVCaptureVideoOrientation orientationToUse)
608 {
609 if (takingPicture)
610 {
611 // Picture taking already in progress!
612 jassertfalse;
613 return;
614 }
615
616 takingPicture = true;
617
618 printImageOutputDebugInfo (captureOutput);
619
620 if (auto* connection = findVideoConnection (captureOutput))
621 {
622 #if JUCE_USE_NEW_CAMERA_API
623 if (@available (iOS 10.0, *))
624 {
625 if ([captureOutput isKindOfClass: [AVCapturePhotoOutput class]])
626 {
627 auto* photoOutput = (AVCapturePhotoOutput*) captureOutput;
628 auto outputConnection = [photoOutput connectionWithMediaType: AVMediaTypeVideo];
629 outputConnection.videoOrientation = orientationToUse;
630
631 [photoOutput capturePhotoWithSettings: [AVCapturePhotoSettings photoSettings]
632 delegate: id<AVCapturePhotoCaptureDelegate> (photoOutputDelegate.get())];
633
634 return;
635 }
636 }
637 #endif
638
639 auto* stillImageOutput = (AVCaptureStillImageOutput*) captureOutput;
640 auto outputConnection = [stillImageOutput connectionWithMediaType: AVMediaTypeVideo];
641 outputConnection.videoOrientation = orientationToUse;
642
643 [stillImageOutput captureStillImageAsynchronouslyFromConnection: connection completionHandler:
644 ^(CMSampleBufferRef imageSampleBuffer, NSError* error)
645 {
646 takingPicture = false;
647
648 if (error != nil)
649 {
650 JUCE_CAMERA_LOG ("Still picture capture failed, error: " + nsStringToJuce (error.localizedDescription));
651 jassertfalse;
652 return;
653 }
654
655 NSData* imageData = [AVCaptureStillImageOutput jpegStillImageNSDataRepresentation: imageSampleBuffer];
656
657 auto image = ImageFileFormat::loadFrom (imageData.bytes, (size_t) imageData.length);
658
659 callListeners (image);
660
661 MessageManager::callAsync ([this, image] { notifyPictureTaken (image); });
662 }];
663 }
664 else
665 {
666 // Could not find a connection of video type
667 jassertfalse;
668 }
669 }

853 static UIImage* getImageWithCorrectOrientation (CGImagePropertyOrientation imageOrientation,
854 CGImageRef imageData)
855 {
856 auto origWidth = CGImageGetWidth (imageData);
857 auto origHeight = CGImageGetHeight (imageData);
858
859 auto targetSize = getTargetImageDimensionFor (imageOrientation, imageData);
860
861 UIGraphicsBeginImageContext (targetSize);
862 CGContextRef context = UIGraphicsGetCurrentContext();
863
864 switch (imageOrientation)
865 {
866 case kCGImagePropertyOrientationUp:
867 CGContextScaleCTM (context, 1.0, -1.0);
868 CGContextTranslateCTM (context, 0.0, -targetSize.height);
869 break;
870 case kCGImagePropertyOrientationRight:
871 CGContextRotateCTM (context, 90 * MathConstants<CGFloat>::pi / 180);
872 CGContextScaleCTM (context, targetSize.height / origHeight, -targetSize.width / origWidth);
873 break;
874 case kCGImagePropertyOrientationDown:
875 CGContextTranslateCTM (context, targetSize.width, 0.0);
876 CGContextScaleCTM (context, -1.0, 1.0);
877 break;
878 case kCGImagePropertyOrientationLeft:
879 CGContextRotateCTM (context, -90 * MathConstants<CGFloat>::pi / 180);
880 CGContextScaleCTM (context, targetSize.height / origHeight, -targetSize.width / origWidth);
881 CGContextTranslateCTM (context, -targetSize.width, -targetSize.height);
882 break;
883 case kCGImagePropertyOrientationUpMirrored:
884 case kCGImagePropertyOrientationDownMirrored:
885 case kCGImagePropertyOrientationLeftMirrored:
886 case kCGImagePropertyOrientationRightMirrored:
887 default:
888 // Not implemented.
889 jassertfalse;
890 break;
891 }
892
893 CGContextDrawImage (context, CGRectMake (0, 0, targetSize.width, targetSize.height), imageData);
894
895 UIImage* correctedImage = UIGraphicsGetImageFromCurrentImageContext();
896 UIGraphicsEndImageContext();
897
898 return correctedImage;
899 }

901 static CGSize getTargetImageDimensionFor (CGImagePropertyOrientation imageOrientation,
902 CGImageRef imageData)
903 {
904 auto width = CGImageGetWidth (imageData);
905 auto height = CGImageGetHeight (imageData);
906
907 switch (imageOrientation)
908 {
909 case kCGImagePropertyOrientationUp:
910 case kCGImagePropertyOrientationUpMirrored:
911 case kCGImagePropertyOrientationDown:
912 case kCGImagePropertyOrientationDownMirrored:
913 return CGSizeMake ((CGFloat) width, (CGFloat) height);
914 case kCGImagePropertyOrientationRight:
915 case kCGImagePropertyOrientationRightMirrored:
916 case kCGImagePropertyOrientationLeft:
917 case kCGImagePropertyOrientationLeftMirrored:
918 return CGSizeMake ((CGFloat) height, (CGFloat) width);
919 }

921 jassertfalse;
