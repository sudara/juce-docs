80 juce_wchar operator*() const noexcept
81 {
82 auto byte = (signed char) *data;
83
84 if (byte >= 0)
85 return (juce_wchar) (uint8) byte;
86
87 uint32 n = (uint32) (uint8) byte;
88 uint32 mask = 0x7f;
89 uint32 bit = 0x40;
90 int numExtraValues = 0;
91
92 while ((n & bit) != 0 && bit > 0x8)
93 {
94 mask >>= 1;
95 ++numExtraValues;
96 bit >>= 1;
97 }
98
99 n &= mask;
100
101 for (int i = 1; i <= numExtraValues; ++i)
102 {
103 auto nextByte = (uint32) (uint8) data[i];
104
105 if ((nextByte & 0xc0) != 0x80)
106 break;
107
108 n <<= 6;
109 n |= (nextByte & 0x3f);
110 }
111
112 return (juce_wchar) n;
113 }

128 bit = static_cast<uint8> (bit >> 1);

328 void write (const juce_wchar charToWrite) noexcept
329 {
330 auto c = (uint32) charToWrite;
331
332 if (c >= 0x80)
333 {
334 int numExtraBytes = 1;
335 if (c >= 0x800)
336 {
337 ++numExtraBytes;
338 if (c >= 0x10000)
339 ++numExtraBytes;
340 }
341
342 *data++ = (CharType) ((uint32) (0xff << (7 - numExtraBytes)) | (c >> (numExtraBytes * 6)));
343
344 while (--numExtraBytes >= 0)
345 *data++ = (CharType) (0x80 | (0x3f & (c >> (numExtraBytes * 6))));
346 }
347 else
348 {
349 *data++ = (CharType) c;

499 static bool isValidString (const CharType* dataToTest, int maxBytesToRead)
500 {
501 while (--maxBytesToRead >= 0 && *dataToTest != 0)
502 {
503 auto byte = (signed char) *dataToTest++;
504
505 if (byte < 0)
506 {
507 int bit = 0x40;
508 int numExtraValues = 0;
509
510 while ((byte & bit) != 0)
511 {
512 if (bit < 8)
513 return false;
514
515 ++numExtraValues;
516 bit >>= 1;
517
518 if (bit == 8 && (numExtraValues > maxBytesToRead
519 || *CharPointer_UTF8 (dataToTest - 1) > 0x10ffff))
520 return false;
521 }
522
523 if (numExtraValues == 0)
524 return false;
525
526 maxBytesToRead -= numExtraValues;
527 if (maxBytesToRead < 0)
528 return false;
529
530 while (--numExtraValues >= 0)
531 if ((*dataToTest++ & 0xc0) != 0x80)
532 return false;
533 }
534 }
535
536 return true;
537 }
