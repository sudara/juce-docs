68template <typename... Processors>
69class ProcessorChain
70{
71public:

87 void prepare (const ProcessSpec& spec)

90 }

99 template <typename ProcessContext>
100 void process (const ProcessContext& context) noexcept
101 {
102 detail::forEachInTuple ([this, &context] (auto& proc, auto index) noexcept { this->processOne (context, proc, index); },
103 processors);
104 }
105
106private:
107 template <typename Context, typename Proc, size_t Ix, std::enable_if_t<! detail::UseContextDirectly<Context, Ix>::value, int> = 0>
108 void processOne (const Context& context, Proc& proc, std::integral_constant<size_t, Ix>) noexcept
109 {
110 jassert (context.getOutputBlock().getNumChannels() == context.getInputBlock().getNumChannels());
111 ProcessContextReplacing<typename Context::SampleType> replacingContext (context.getOutputBlock());
112 replacingContext.isBypassed = (bypassed[Ix] || context.isBypassed);
113
114 proc.process (replacingContext);
115 }
116
117 template <typename Context, typename Proc, size_t Ix, std::enable_if_t<detail::UseContextDirectly<Context, Ix>::value, int> = 0>
118 void processOne (const Context& context, Proc& proc, std::integral_constant<size_t, Ix>) noexcept
119 {
120 auto contextCopy = context;
121 contextCopy.isBypassed = (bypassed[Ix] || context.isBypassed);
122
123 proc.process (contextCopy);
124 }
125
126 std::tuple<Processors...> processors;
127 std::array<bool, sizeof...(Processors)> bypassed { {} };
128};
