28template <typename T, juce::int32 N>
29class b2GrowableStack
30{
31public:
32 b2GrowableStack()
33 {
34 m_stack = m_array;
35 m_count = 0;
36 m_capacity = N;
37 }
38
39 ~b2GrowableStack()
40 {
41 if (m_stack != m_array)
42 {
43 b2Free(m_stack);
44 m_stack = NULL;
45 }
46 }
47
48 void Push(const T& element)
49 {
50 if (m_count == m_capacity)
51 {
52 T* old = m_stack;
53 m_capacity *= 2;
54 m_stack = (T*)b2Alloc(m_capacity * sizeof(T));
55 std::memcpy(m_stack, old, m_count * sizeof(T));
56 if (old != m_array)
57 {
58 b2Free(old);
59 }
60 }
61
62 m_stack[m_count] = element;
63 ++m_count;
64 }
65
66 T Pop()
67 {
68 b2Assert(m_count > 0);
69 --m_count;
70 return m_stack[m_count];
71 }
72
73 juce::int32 GetCount()
74 {
75 return m_count;
76 }
77
78private:
79 T* m_stack;
80 T m_array[N];
81 juce::int32 m_count;
82 juce::int32 m_capacity;
83};
