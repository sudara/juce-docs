33/* When you wrap a WindowRef as an NSWindow, it seems to bugger up the HideWindow
34 function, so when the host tries (and fails) to hide the window, this stuff catches
35 the event and forces it to update.
36*/
37static pascal OSStatus windowVisibilityBodge (EventHandlerCallRef, EventRef e, void* user)
38{
39 NSWindow* hostWindow = (NSWindow*) user;
40
41 switch (GetEventKind (e))
42 {
43 case kEventWindowInit: [hostWindow display]; break;
44 case kEventWindowShown: [hostWindow orderFront: nil]; break;
45 case kEventWindowHidden: [hostWindow orderOut: nil]; break;
46 }
47
48 return eventNotHandledErr;
49}
50
51inline void attachWindowHidingHooks (Component* comp, void* hostWindowRef, NSWindow* nsWindow)
52{
53 const EventTypeSpec eventsToCatch[] =
54 {
55 { kEventClassWindow, kEventWindowInit },
56 { kEventClassWindow, kEventWindowShown },
57 { kEventClassWindow, kEventWindowHidden }
58 };
59
60 EventHandlerRef ref;
61 InstallWindowEventHandler ((WindowRef) hostWindowRef,
62 NewEventHandlerUPP (windowVisibilityBodge),
63 GetEventTypeCount (eventsToCatch), eventsToCatch,
64 (void*) nsWindow, &ref);
65
66 comp->getProperties().set ("carbonEventRef", String::toHexString ((pointer_sized_int) (void*) ref));
67}
68
69inline void removeWindowHidingHooks (Component* comp)
70{
71 if (comp != nullptr)
72 RemoveEventHandler ((EventHandlerRef) (void*) (pointer_sized_int)
73 comp->getProperties() ["carbonEventRef"].toString().getHexValue64());
74}
