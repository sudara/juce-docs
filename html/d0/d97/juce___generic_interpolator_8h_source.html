183 forcedinline void pushInterpolationSamples (const float* input,
184 int numOutputSamplesToProduce) noexcept
185 {
186 if (numOutputSamplesToProduce >= memorySize)
187 {
188 const auto* const offsetInput = input + (numOutputSamplesToProduce - memorySize);
189
190 for (int i = 0; i < memorySize; ++i)
191 pushInterpolationSample (offsetInput[i]);
192 }
193 else
194 {
195 for (int i = 0; i < numOutputSamplesToProduce; ++i)
196 pushInterpolationSample (input[i]);
197 }
198 }

200 forcedinline void pushInterpolationSamples (const float* input,
201 int numOutputSamplesToProduce,
202 int numInputSamplesAvailable,
203 int wrapAround) noexcept
204 {
205 if (numOutputSamplesToProduce >= memorySize)
206 {
207 if (numInputSamplesAvailable >= memorySize)
208 {
209 pushInterpolationSamples (input,
210 numOutputSamplesToProduce);
211 }
212 else
213 {
214 pushInterpolationSamples (input + ((numOutputSamplesToProduce - numInputSamplesAvailable) - 1),
215 numInputSamplesAvailable);
216
217 if (wrapAround > 0)
218 {
219 numOutputSamplesToProduce -= wrapAround;
220
221 pushInterpolationSamples (input + ((numOutputSamplesToProduce - (memorySize - numInputSamplesAvailable)) - 1),
222 memorySize - numInputSamplesAvailable);
223 }
224 else
225 {
226 for (int i = numInputSamplesAvailable; i < memorySize; ++i)
227 pushInterpolationSample (0.0f);
228 }
229 }
230 }
231 else
232 {
233 if (numOutputSamplesToProduce > numInputSamplesAvailable)
234 {
235 for (int i = 0; i < numInputSamplesAvailable; ++i)
236 pushInterpolationSample (input[i]);
237
238 const auto extraSamples = numOutputSamplesToProduce - numInputSamplesAvailable;
239
240 if (wrapAround > 0)
241 {
242 const auto* const offsetInput = input + (numInputSamplesAvailable - wrapAround);
243
244 for (int i = 0; i < extraSamples; ++i)
245 pushInterpolationSample (offsetInput[i]);
246 }
247 else
248 {
249 for (int i = 0; i < extraSamples; ++i)
250 pushInterpolationSample (0.0f);
251 }
252 }
253 else
254 {
255 for (int i = 0; i < numOutputSamplesToProduce; ++i)
256 pushInterpolationSample (input[i]);
257 }

262 int interpolate (double speedRatio,
263 const float* input,
264 float* output,
265 int numOutputSamplesToProduce) noexcept
266 {
267 auto pos = subSamplePos;
268 int numUsed = 0;
269
270 while (numOutputSamplesToProduce > 0)
271 {
272 while (pos >= 1.0)
273 {
274 pushInterpolationSample (input[numUsed++]);
275 pos -= 1.0;
276 }
277
278 *output++ = InterpolatorTraits::valueAtOffset (lastInputSamples, (float) pos, indexBuffer);
279 pos += speedRatio;
280 --numOutputSamplesToProduce;
281 }
282
283 subSamplePos = pos;
284 return numUsed;
285 }

287 int interpolate (double speedRatio,
288 const float* input, float* output,
289 int numOutputSamplesToProduce,
290 int numInputSamplesAvailable,
291 int wrap) noexcept
292 {
293 auto originalIn = input;
294 auto pos = subSamplePos;
295 bool exceeded = false;
296
297 if (speedRatio < 1.0)
298 {
299 for (int i = numOutputSamplesToProduce; --i >= 0;)
300 {
301 if (pos >= 1.0)
302 {
303 if (exceeded)
304 {
305 pushInterpolationSample (0.0f);
306 }
307 else
308 {
309 pushInterpolationSample (*input++);
310
311 if (--numInputSamplesAvailable <= 0)
312 {
313 if (wrap > 0)
314 {
315 input -= wrap;
316 numInputSamplesAvailable += wrap;
317 }
318 else
319 {
320 exceeded = true;
321 }
322 }
323 }
324
325 pos -= 1.0;
326 }
327
328 *output++ = InterpolatorTraits::valueAtOffset (lastInputSamples, (float) pos, indexBuffer);
329 pos += speedRatio;
330 }
331 }
332 else
333 {
334 for (int i = numOutputSamplesToProduce; --i >= 0;)
335 {
336 while (pos < speedRatio)
337 {
338 if (exceeded)
339 {
340 pushInterpolationSample (0);
341 }
342 else
343 {
344 pushInterpolationSample (*input++);
345
346 if (--numInputSamplesAvailable <= 0)
347 {
348 if (wrap > 0)
349 {
350 input -= wrap;
351 numInputSamplesAvailable += wrap;
352 }
353 else
354 {
355 exceeded = true;
356 }
357 }
358 }
359
360 pos += 1.0;
361 }
362
363 pos -= speedRatio;
364 *output++ = InterpolatorTraits::valueAtOffset (lastInputSamples, jmax (0.0f, 1.0f - (float) pos), indexBuffer);
365 }
366 }
367
368 subSamplePos = pos;
369
370 if (wrap == 0)
371 return (int) (input - originalIn);
372
373 return ((int) (input - originalIn) + wrap) % wrap;
374 }
