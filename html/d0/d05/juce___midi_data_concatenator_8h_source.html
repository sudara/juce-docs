49 template <typename UserDataType, typename CallbackType>
50 void pushMidiData (const void* inputData, int numBytes, double time,
51 UserDataType* input, CallbackType& callback)
52 {
53 auto d = static_cast<const uint8*> (inputData);
54
55 while (numBytes > 0)
56 {
57 auto nextByte = *d;
58
59 if (pendingSysexSize != 0 || nextByte == 0xf0)
60 {
61 processSysex (d, numBytes, time, input, callback);
62 currentMessageLen = 0;
63 continue;
64 }
65
66 ++d;
67 --numBytes;
68
69 if (isRealtimeMessage (nextByte))
70 {
71 callback.handleIncomingMidiMessage (input, MidiMessage (nextByte, time));
72 // These can be embedded in the middle of a normal message, so we won't
73 // reset the currentMessageLen here.
74 continue;
75 }
76
77 if (isInitialByte (nextByte))
78 {
79 currentMessage[0] = nextByte;
80 currentMessageLen = 1;
81 }
82 else if (currentMessageLen > 0 && currentMessageLen < 3)
83 {
84 currentMessage[currentMessageLen++] = nextByte;
85 }
86 else
87 {
88 // message is too long or invalid MIDI - abandon it and start again with the next byte
89 currentMessageLen = 0;
90 continue;
91 }
92
93 auto expectedLength = MidiMessage::getMessageLengthFromFirstByte (currentMessage[0]);
94
95 if (expectedLength == currentMessageLen)
96 {
97 callback.handleIncomingMidiMessage (input, MidiMessage (currentMessage, expectedLength, time));
98 currentMessageLen = 1; // reset, but leave the first byte to use as the running status byte
99 }
100 }
101 }

104 template <typename UserDataType, typename CallbackType>
105 void processSysex (const uint8*& d, int& numBytes, double time,
106 UserDataType* input, CallbackType& callback)
107 {
108 if (*d == 0xf0)
109 {
110 pendingSysexSize = 0;
111 pendingSysexTime = time;
112 }
113
114 pendingSysexData.ensureSize ((size_t) (pendingSysexSize + numBytes), false);
115 auto totalMessage = static_cast<uint8*> (pendingSysexData.getData());
116 auto dest = totalMessage + pendingSysexSize;
117
118 do
119 {
120 if (pendingSysexSize > 0 && isStatusByte (*d))
121 {
122 if (*d == 0xf7)
123 {
124 *dest++ = *d++;
125 ++pendingSysexSize;
126 --numBytes;
127 break;
128 }
129
130 if (*d >= 0xfa || *d == 0xf8)
131 {
132 callback.handleIncomingMidiMessage (input, MidiMessage (*d, time));
133 ++d;
134 --numBytes;
135 }
136 else
137 {
138 pendingSysexSize = 0;
139 int used = 0;
140 const MidiMessage m (d, numBytes, used, 0, time);
141
142 if (used > 0)
143 {
144 callback.handleIncomingMidiMessage (input, m);
145 numBytes -= used;
146 d += used;
147 }
148
149 break;
150 }
151 }
152 else
153 {
154 *dest++ = *d++;
155 ++pendingSysexSize;
156 --numBytes;
157 }
158 }
159 while (numBytes > 0);
160
161 if (pendingSysexSize > 0)
162 {
163 if (totalMessage [pendingSysexSize - 1] == 0xf7)
164 {
165 callback.handleIncomingMidiMessage (input, MidiMessage (totalMessage, pendingSysexSize, pendingSysexTime));
166 pendingSysexSize = 0;
167 }
168 else
169 {
170 callback.handlePartialSysexMessage (input, totalMessage, pendingSysexSize, pendingSysexTime);
171 }
172 }
173 }

177 static bool isInitialByte (uint8 byte) { return isStatusByte (byte) && byte != 0xf7; }
