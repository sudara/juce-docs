70class AUBufferList {
71 enum EPtrState {
72 kPtrsInvalid,
73 kPtrsToMyMemory,
74 kPtrsToExternalMemory
75 };
76public:
78 AUBufferList() : mPtrState(kPtrsInvalid), mExternalMemory(false), mPtrs(NULL), mMemory(NULL),
79 mAllocatedStreams(0), mAllocatedFrames(0), mAllocatedBytes(0) { }
81 ~AUBufferList();

84 AudioBufferList & PrepareBuffer(const CAStreamBasicDescription &format, UInt32 nFrames);
86 AudioBufferList & PrepareNullBuffer(const CAStreamBasicDescription &format, UInt32 nFrames);
87
89 AudioBufferList & SetBufferList(const AudioBufferList &abl) {
90 if (mAllocatedStreams < abl.mNumberBuffers)
91 COMPONENT_THROW(-1);
92 mPtrState = kPtrsToExternalMemory;
93 memcpy(mPtrs, &abl, (char *)&abl.mBuffers[abl.mNumberBuffers] - (char *)&abl);
94 return *mPtrs;
95 }
96
98 void SetBuffer(UInt32 index, const AudioBuffer &ab) {
99 if (mPtrState == kPtrsInvalid || index >= mPtrs->mNumberBuffers)
100 COMPONENT_THROW(-1);
101 mPtrState = kPtrsToExternalMemory;
102 mPtrs->mBuffers[index] = ab;
103 }
104
106 void InvalidateBufferList() { mPtrState = kPtrsInvalid; }
107
109 AudioBufferList & GetBufferList() const {
110 if (mPtrState == kPtrsInvalid)
111 COMPONENT_THROW(-1);
112 return *mPtrs;
113 }
114
116 void CopyBufferListTo(AudioBufferList &abl) const {
117 if (mPtrState == kPtrsInvalid)
118 COMPONENT_THROW(-1);
119 memcpy(&abl, mPtrs, (char *)&abl.mBuffers[abl.mNumberBuffers] - (char *)&abl);
120 }
121
123 void CopyBufferContentsTo(AudioBufferList &abl) const {
124 if (mPtrState == kPtrsInvalid)
125 COMPONENT_THROW(-1);
126 const AudioBuffer *srcbuf = mPtrs->mBuffers;
127 AudioBuffer *destbuf = abl.mBuffers;
128
129 for (UInt32 i = 0; i < abl.mNumberBuffers; ++i, ++srcbuf, ++destbuf) {
130 if (i >= mPtrs->mNumberBuffers) // duplicate last source to additional outputs [4341137]
131 --srcbuf;
132 if (destbuf->mData != srcbuf->mData)
133 memmove(destbuf->mData, srcbuf->mData, srcbuf->mDataByteSize);
134 destbuf->mDataByteSize = srcbuf->mDataByteSize;
135 }
136 }

182// Allocates an array of samples (type T), to be optimally aligned for the processor
184template <class T>
185class TAUBuffer {
186public:
187 enum {
188 kAlignInterval = 0x10,
189 kAlignMask = kAlignInterval - 1
190 };
191
193 TAUBuffer() : mMemObject(NULL), mAlignedBuffer(NULL), mBufferSizeBytes(0)
194 {
195 }
196
198 TAUBuffer(UInt32 numElems, UInt32 numChannels) : mMemObject(NULL), mAlignedBuffer(NULL),
199 mBufferSizeBytes(0)
200 {
201 Allocate(numElems, numChannels);
202 }
203
205 ~TAUBuffer()
206 {
207 Deallocate();
208 }
209
211 void Allocate(UInt32 numElems) // can also re-allocate
212 {
213 UInt32 reqSize = numElems * sizeof(T);
214
215 if (mMemObject != NULL && reqSize == mBufferSizeBytes)
216 return; // already allocated
217
218 mBufferSizeBytes = reqSize;
219 mMemObject = CA_realloc(mMemObject, reqSize);
220 UInt32 misalign = (uintptr_t)mMemObject & kAlignMask;
221 if (misalign) {
222 mMemObject = CA_realloc(mMemObject, reqSize + kAlignMask);
223 mAlignedBuffer = (T *)((char *)mMemObject + kAlignInterval - misalign);
224 } else
225 mAlignedBuffer = (T *)mMemObject;
226 }
227
229 void Deallocate()
230 {
231 if (mMemObject == NULL) return; // so this method has no effect if we're using
232 // an external buffer
233
234 free(mMemObject);
235 mMemObject = NULL;
236 mAlignedBuffer = NULL;
237 mBufferSizeBytes = 0;
238 }
239
241 void AllocateClear(UInt32 numElems) // can also re-allocate
242 {
243 Allocate(numElems);
244 Clear();
245 }
246
248 void Clear()
249 {
250 memset(mAlignedBuffer, 0, mBufferSizeBytes);
251 }
252
253 // accessors
254
256 operator T *() { return mAlignedBuffer; }
257
258private:
260 void * mMemObject; // null when using an external buffer
262 T * mAlignedBuffer; // always valid once allocated
264 UInt32 mBufferSizeBytes;
265};
