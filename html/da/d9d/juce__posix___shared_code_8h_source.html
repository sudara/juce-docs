63bool Process::setMaxNumberOfFileHandles (int newMaxNumber) noexcept
64{
65 rlimit lim;
66
67 if (getrlimit (RLIMIT_NOFILE, &lim) == 0)
68 {
69 if (newMaxNumber <= 0 && lim.rlim_cur == RLIM_INFINITY && lim.rlim_max == RLIM_INFINITY)
70 return true;
71
72 if (newMaxNumber > 0 && lim.rlim_cur >= (rlim_t) newMaxNumber)
73 return true;
74 }
75
76 lim.rlim_cur = lim.rlim_max = newMaxNumber <= 0 ? RLIM_INFINITY : (rlim_t) newMaxNumber;
77 return setrlimit (RLIMIT_NOFILE, &lim) == 0;
78}

118File File::getCurrentWorkingDirectory()
119{
120 HeapBlock<char> heapBuffer;
121
122 char localBuffer[1024];
123 auto cwd = getcwd (localBuffer, sizeof (localBuffer) - 1);
124 size_t bufferSize = 4096;
125
126 while (cwd == nullptr && errno == ERANGE)
127 {
128 heapBuffer.malloc (bufferSize);
129 cwd = getcwd (heapBuffer, bufferSize - 1);
130 bufferSize += 1024;
131 }
132
133 return File (CharPointer_UTF8 (cwd));
134}

143int juce_siginterrupt (int sig, int flag)
144{
145 #if JUCE_WASM
146 ignoreUnused (sig, flag);
147 return 0;
148 #else
149 #if JUCE_ANDROID
150 using juce_sigactionflags_type = unsigned long;
151 #else
152 using juce_sigactionflags_type = int;
153 #endif
154
155 struct ::sigaction act;
156 (void) ::sigaction (sig, nullptr, &act);
157
158 if (flag != 0)
159 act.sa_flags &= static_cast<juce_sigactionflags_type> (~SA_RESTART);
160 else
161 act.sa_flags |= static_cast<juce_sigactionflags_type> (SA_RESTART);
162
163 return ::sigaction (sig, &act, nullptr);
164 #endif
165}

335 accessTime = (int64) info.st_atime * 1000;

346{
347 #if ! JUCE_WASM
348 juce_statStruct info;
349
350 if ((modificationTime != 0 || accessTime != 0) && juce_stat (fullPath, info))

352 #if JUCE_MAC || (JUCE_IOS && __DARWIN_ONLY_64_BIT_INO_T)

363

365 : info.st_mtimespec.tv_sec;
366
367 times[1].tv_usec = setModificationTime ? static_cast<__darwin_suseconds_t> ((modificationTime % 1000) * 1000)
368 : static_cast<__darwin_suseconds_t> (info.st_mtimespec.tv_nsec / 1000);
369
370 return utimes (fullPath.toUTF8(), times) == 0;
371 #else
372 struct utimbuf times;
373 times.actime = accessTime != 0 ? static_cast<time_t> (accessTime / 1000) : static_cast<time_t> (info.st_atime);

554void MemoryMappedFile::openInternal (const File& file, AccessMode mode, bool exclusive)
555{
556 jassert (mode == readOnly || mode == readWrite);
557
558 if (range.getStart() > 0)
559 {
560 auto pageSize = sysconf (_SC_PAGE_SIZE);
561 range.setStart (range.getStart() - (range.getStart() % pageSize));
562 }
563
564 auto filename = file.getFullPathName().toUTF8();
565
566 if (mode == readWrite)
567 fileHandle = open (filename, O_CREAT | O_RDWR, 00644);
568 else
569 fileHandle = open (filename, O_RDONLY);
570
571 if (fileHandle != -1)
572 {
573 auto m = mmap (nullptr, (size_t) range.getLength(),
574 mode == readWrite ? (PROT_READ | PROT_WRITE) : PROT_READ,
575 exclusive ? MAP_PRIVATE : MAP_SHARED, fileHandle,
576 (off_t) range.getStart());
577
578 if (m != MAP_FAILED)
579 {
580 address = m;
581 madvise (m, (size_t) range.getLength(), MADV_SEQUENTIAL);
582 }
583 else
584 {
585 range = Range<int64>();
586 }
587
588 close (fileHandle);
589 fileHandle = 0;
590 }
591}

598 if (fileHandle != 0)

720
721class InterProcessLock::Pimpl
722{
723public:
724 Pimpl (const String& lockName, int timeOutMillisecs)
725 {
726 #if JUCE_MAC
727 if (! createLockFile (File ("~/Library/Caches/com.juce.locks").getChildFile (lockName), timeOutMillisecs))
728 // Fallback if the user's home folder is on a network drive with no ability to lock..
729 createLockFile (File ("/tmp/com.juce.locks").getChildFile (lockName), timeOutMillisecs);
730
731 #else
732 File tempFolder ("/var/tmp");
733
734 if (! tempFolder.isDirectory())
735 tempFolder = "/tmp";
736
737 createLockFile (tempFolder.getChildFile (lockName), timeOutMillisecs);
738 #endif
739 }
740
741 ~Pimpl()
742 {
743 closeFile();
744 }
745
746 bool createLockFile (const File& file, int timeOutMillisecs)
747 {
748 file.create();
749 handle = open (file.getFullPathName().toUTF8(), O_RDWR);
750
751 if (handle != 0)
752 {
753 struct flock fl;
754 zerostruct (fl);
755
756 fl.l_whence = SEEK_SET;
757 fl.l_type = F_WRLCK;
758
759 auto endTime = Time::currentTimeMillis() + timeOutMillisecs;
760
761 for (;;)
762 {
763 auto result = fcntl (handle, F_SETLK, &fl);
764
765 if (result >= 0)
766 return true;
767
768 auto error = errno;
769
770 if (error != EINTR)
771 {
772 if (error == EBADF || error == ENOTSUP)
773 return false;
774
775 if (timeOutMillisecs == 0
776 || (timeOutMillisecs > 0 && Time::currentTimeMillis() >= endTime))
777 break;
778
779 Thread::sleep (10);
780 }
781 }
782 }
783
784 closeFile();
785 return true; // only false if there's a file system error. Failure to lock still returns true.
786 }
787
788 void closeFile()
789 {
790 if (handle != 0)
791 {
792 struct flock fl;
793 zerostruct (fl);
794
795 fl.l_whence = SEEK_SET;
796 fl.l_type = F_UNLCK;
797
798 while (! (fcntl (handle, F_SETLKW, &fl) >= 0 || errno != EINTR))
799 {}
800
801 close (handle);
802 handle = 0;
803 }
804 }
805
806 int handle = 0, refCount = 1;
807};

960bool Thread::setThreadPriority (void* handle, int priority)
961{
962 constexpr auto maxInputPriority = 10;
963
964 #if JUCE_LINUX || JUCE_BSD
965 constexpr auto lowestRrPriority = 8;
966 #else

972
973 if (handle == nullptr)
974 handle = (void*) pthread_self();

976 if (pthread_getschedparam ((pthread_t) handle, &policy, &param) != 0)
977 return false;

979 policy = priority < lowestRrPriority ? SCHED_OTHER : SCHED_RR;
980
981 const auto minPriority = sched_get_priority_min (policy);
982 const auto maxPriority = sched_get_priority_max (policy);
983
984 param.sched_priority = [&]
985 {
986 if (policy == SCHED_OTHER)
987 return 0;
988
989 return jmap (priority, lowestRrPriority, maxInputPriority, minPriority, maxPriority);
990 }();
991
992 return pthread_setschedparam ((pthread_t) handle, policy, &param) == 0;
993}

1088 ActiveProcess (const StringArray& arguments, int streamFlags)
1089 {
1090 auto exe = arguments[0].unquoted();
1091
1092 // Looks like you're trying to launch a non-existent exe or a folder (perhaps on OSX
1093 // you're trying to launch the .app folder rather than the actual binary inside it?)
1094 jassert (File::getCurrentWorkingDirectory().getChildFile (exe).existsAsFile()
1095 || ! exe.containsChar (File::getSeparatorChar()));
1096
1097 int pipeHandles[2] = {};
1098
1099 if (pipe (pipeHandles) == 0)
1100 {
1101 auto result = fork();
1102
1103 if (result < 0)
1104 {
1105 close (pipeHandles[0]);
1106 close (pipeHandles[1]);
1107 }
1108 else if (result == 0)
1109 {
1110 // we're the child process..
1111 close (pipeHandles[0]); // close the read handle
1112
1113 if ((streamFlags & wantStdOut) != 0)
1114 dup2 (pipeHandles[1], STDOUT_FILENO); // turns the pipe into stdout
1115 else
1116 dup2 (open ("/dev/null", O_WRONLY), STDOUT_FILENO);
1117
1118 if ((streamFlags & wantStdErr) != 0)
1119 dup2 (pipeHandles[1], STDERR_FILENO);

1122
1123 close (pipeHandles[1]);
1124

1126
1127 for (auto& arg : arguments)
1128 if (arg.isNotEmpty())
1129 argv.add (const_cast<char*> (arg.toRawUTF8()));
1130
1131 argv.add (nullptr);
1132
1133 execvp (exe.toRawUTF8(), argv.getRawDataPointer());

1135 }
1136 else
1137 {
1138 // we're the parent process..
1139 childPID = result;
1140 pipeHandle = pipeHandles[0];
1141 close (pipeHandles[1]); // close the write handle
1142 }
1143 }

1265
1266 ~Pimpl()

1271
1272 void start (int newPeriod)
1273 {
1274 if (periodMs == newPeriod)
1275 return;
1276
1277 if (thread.get_id() == std::this_thread::get_id())
1278 {
1279 periodMs = newPeriod;
1280 return;
1281 }
1282
1283 stop();
1284
1285 periodMs = newPeriod;
1286
1287 thread = std::thread ([this, newPeriod]
1288 {
1289 setThisThreadToRealtime ((uint64) newPeriod);
1290
1291 auto lastPeriod = periodMs.load();
1292 Clock clock (lastPeriod);
1293
1294 std::unique_lock<std::mutex> unique_lock (timerMutex);
1295
1296 while (periodMs != 0)
1297 {
1298 clock.next();
1299 while (periodMs != 0 && clock.wait (stopCond, unique_lock));
1300
1301 if (periodMs == 0)
1302 break;
1303
1304 owner.hiResTimerCallback();
1305
1306 auto nextPeriod = periodMs.load();
1307
1308 if (lastPeriod != nextPeriod)
1309 {
1310 lastPeriod = nextPeriod;
1311 clock = Clock (lastPeriod);
1312 }
1313 }
1314
1315 periodMs = 0;
1316 });
1317 }
1318
1319 void stop()
1320 {
1321 periodMs = 0;
1322
1323 const auto thread_id = thread.get_id();
1324
1325 if (thread_id == std::thread::id() || thread_id == std::this_thread::get_id())
1326 return;
1327
1328 {
1329 std::unique_lock<std::mutex> unique_lock (timerMutex);
1330 stopCond.notify_one();
1331 }
1332
1333 thread.join();
1334 }

1336 HighResolutionTimer& owner;
1337 std::atomic<int> periodMs { 0 };

1353 {

1355 }
