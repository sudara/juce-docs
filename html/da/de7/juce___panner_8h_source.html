32enum class PannerRule
33{
34 linear, // regular 6 dB or linear panning rule, allows the panned sound to be
35 // perceived as having a constant level when summed to mono
36 balanced, // both left and right are 1 when pan value is 0, with left decreasing
37 // to 0 above this value and right decreasing to 0 below it
38 sin3dB, // alternate version of the regular 3 dB panning rule with a sine curve
39 sin4p5dB, // alternate version of the regular 4.5 dB panning rule with a sine curve
40 sin6dB, // alternate version of the regular 6 dB panning rule with a sine curve
41 squareRoot3dB, // regular 3 dB or constant power panning rule, allows the panned sound
42 // to be perceived as having a constant level regardless of the pan position
43 squareRoot4p5dB // regular 4.5 dB panning rule, a compromise option between 3 dB and 6 dB panning rules
44};

78 template <typename ProcessContext>
79 void process (const ProcessContext& context) noexcept
80 {
81 const auto& inputBlock = context.getInputBlock();
82 auto& outputBlock = context.getOutputBlock();
83
84 const auto numInputChannels = inputBlock.getNumChannels();
85 const auto numOutputChannels = outputBlock.getNumChannels();
86 const auto numSamples = outputBlock.getNumSamples();
87
88 jassertquiet (inputBlock.getNumSamples() == numSamples);
89
90 if (numOutputChannels != 2 || numInputChannels == 0 || numInputChannels > 2)
91 return;
92
93 if (numInputChannels == 2)
94 {
95 outputBlock.copyFrom (inputBlock);
96 }
97 else
98 {
99 outputBlock.getSingleChannelBlock (0).copyFrom (inputBlock);
100 outputBlock.getSingleChannelBlock (1).copyFrom (inputBlock);
101 }
102
103 if (context.isBypassed)
104 return;
105
106 outputBlock.getSingleChannelBlock (0).multiplyBy (leftVolume);
107 outputBlock.getSingleChannelBlock (1).multiplyBy (rightVolume);
108 }
109
110private:
111 //==============================================================================
112 void update();
113
114 //==============================================================================
115 Rule currentRule = Rule::balanced;
116 SampleType pan = 0.0;
117 SmoothedValue<SampleType> leftVolume, rightVolume;
118 double sampleRate = 44100.0;
119};
