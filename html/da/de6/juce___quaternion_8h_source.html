46 static Quaternion fromAngle (Type angle, Vector3D<Type> axis) noexcept
47 {
48 return Quaternion (axis.normalised() * std::sin (angle / (Type) 2), std::cos (angle / (Type) 2));
49 }
50

52 {

55 return *this;
56 }
57
58 Quaternion& operator*= (Quaternion other) noexcept
59 {
60 const Type oldScalar (scalar);
61 scalar = (scalar * other.scalar) - (vector * other.vector);
62 vector = (other.vector * oldScalar) + (vector * other.scalar) + (vector ^ other.vector);
63 return *this;
64 }
65
66 Type length() const noexcept { return std::sqrt (normal()); }
67 Type normal() const noexcept { return scalar * scalar + vector.lengthSquared(); }
68
69 Quaternion normalised() const noexcept
70 {
71 const Type len (length());
72 jassert (len > 0);
73 return Quaternion (vector / len, scalar / len);
74 }
75
77 Matrix3D<Type> getRotationMatrix() const noexcept
78 {
79 const Type norm (normal());
80 const Type s (norm > 0 ? ((Type) 2) / norm : 0);
81 const Type xs (s * vector.x), ys (s * vector.y), zs (s * vector.z);
82 const Type wx (xs * scalar), wy (ys * scalar), wz (zs * scalar);
83 const Type xx (xs * vector.x), xy (ys * vector.x), xz (zs * vector.x);
84 const Type yy (ys * vector.y), yz (zs * vector.y), zz (zs * vector.z);
85
86 return Matrix3D<Type> (((Type) 1) - (yy + zz), xy - wz, xz + wy, 0,
87 xy + wz, ((Type) 1) - (xx+ zz), yz - wx, 0,
88 xz - wy, yz + wx, ((Type) 1) - (xx + yy), 0,
89 0, 0, 0, (Type) 1);
90 }
