32class PointerState

39public:

43 bool operator!= (const PointerState& other) const noexcept { return tie() != other.tie(); }

45 JUCE_NODISCARD PointerState withPositionOffset (Point<float> x) const noexcept { return with (&PointerState::position, position + x); }

53 Point<float> position;
54 float pressure = MouseInputSource::defaultPressure;
55 float orientation = MouseInputSource::defaultOrientation;
56 float rotation = MouseInputSource::defaultRotation;
57 float tiltX = MouseInputSource::defaultTiltX;
58 float tiltY = MouseInputSource::defaultTiltY;
59
60 bool isPressureValid() const noexcept { return 0.0f <= pressure && pressure <= 1.0f; }
61 bool isOrientationValid() const noexcept { return 0.0f <= orientation && orientation <= MathConstants<float>::twoPi; }
62 bool isRotationValid() const noexcept { return 0.0f <= rotation && rotation <= MathConstants<float>::twoPi; }
63 bool isTiltValid (bool isX) const noexcept
64 {
65 return isX ? (-1.0f <= tiltX && tiltX <= 1.0f)
66 : (-1.0f <= tiltY && tiltY <= 1.0f);
67 }
