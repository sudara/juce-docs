56public:
57 NativeContext (Component& comp,
58 const OpenGLPixelFormat& cPixelFormat,
59 void* shareContext,
60 bool useMultisamplingIn,
61 OpenGLVersion)
62 : component (comp), contextToShareWith (shareContext), dummy (*this)
63 {
64 display = XWindowSystem::getInstance()->getDisplay();
65
66 XWindowSystemUtilities::ScopedXLock xLock;
67
68 X11Symbols::getInstance()->xSync (display, False);
69
70 const std::vector<GLint> optionalAttribs
71 {
72 GLX_SAMPLE_BUFFERS, useMultisamplingIn ? 1 : 0,
73 GLX_SAMPLES, cPixelFormat.multisamplingLevel
74 };
75
76 if (! tryChooseVisual (cPixelFormat, optionalAttribs) && ! tryChooseVisual (cPixelFormat, {}))
77 return;
78
79 auto* peer = component.getPeer();
80 jassert (peer != nullptr);
81
82 auto windowH = (Window) peer->getNativeHandle();
83 auto colourMap = X11Symbols::getInstance()->xCreateColormap (display, windowH, bestVisual->visual, AllocNone);
84
85 XSetWindowAttributes swa;
86 swa.colormap = colourMap;
87 swa.border_pixel = 0;
88 swa.event_mask = embeddedWindowEventMask;
89
90 auto glBounds = component.getTopLevelComponent()
91 ->getLocalArea (&component, component.getLocalBounds());
92
93 glBounds = Desktop::getInstance().getDisplays().logicalToPhysical (glBounds);
94
95 embeddedWindow = X11Symbols::getInstance()->xCreateWindow (display, windowH,
96 glBounds.getX(), glBounds.getY(),
97 (unsigned int) jmax (1, glBounds.getWidth()),
98 (unsigned int) jmax (1, glBounds.getHeight()),
99 0, bestVisual->depth,
100 InputOutput,
101 bestVisual->visual,
102 CWBorderPixel | CWColormap | CWEventMask,
103 &swa);
104
105 X11Symbols::getInstance()->xSaveContext (display, (XID) embeddedWindow, windowHandleXContext, (XPointer) peer);
106
107 X11Symbols::getInstance()->xMapWindow (display, embeddedWindow);
108 X11Symbols::getInstance()->xFreeColormap (display, colourMap);
109
110 X11Symbols::getInstance()->xSync (display, False);
111
112 juce_LinuxAddRepaintListener (peer, &dummy);
113 }
114
115 ~NativeContext()
116 {
117 if (auto* peer = component.getPeer())
118 {
119 juce_LinuxRemoveRepaintListener (peer, &dummy);
120
121 if (embeddedWindow != 0)
122 {
123 XWindowSystemUtilities::ScopedXLock xLock;
124
125 X11Symbols::getInstance()->xUnmapWindow (display, embeddedWindow);
126 X11Symbols::getInstance()->xDestroyWindow (display, embeddedWindow);
127 X11Symbols::getInstance()->xSync (display, False);
128
129 XEvent event;
130 while (X11Symbols::getInstance()->xCheckWindowEvent (display,
131 embeddedWindow,
132 embeddedWindowEventMask,
133 &event) == True)
134 {
135 }
136 }
137 }
138
139 if (bestVisual != nullptr)
140 X11Symbols::getInstance()->xFree (bestVisual);
141 }

143 bool initialiseOnRenderThread (OpenGLContext& c)
144 {
145 XWindowSystemUtilities::ScopedXLock xLock;
146 renderContext = glXCreateContext (display, bestVisual, (GLXContext) contextToShareWith, GL_TRUE);
147 c.makeActive();
148 context = &c;
149
150 return true;
151 }
152
153 void shutdownOnRenderThread()
154 {
155 XWindowSystemUtilities::ScopedXLock xLock;
156 context = nullptr;
157 deactivateCurrentContext();
158 glXDestroyContext (display, renderContext);
159 renderContext = nullptr;
160 }
161
162 bool makeActive() const noexcept

167 }
168

192 bounds = newBounds;

194
195 XWindowSystemUtilities::ScopedXLock xLock;

197 physicalBounds.getX(), physicalBounds.getY(),
198 (unsigned int) jmax (1, physicalBounds.getWidth()),
199 (unsigned int) jmax (1, physicalBounds.getHeight()));
200 }

202 bool setSwapInterval (int numFramesPerSwap)
203 {
204 if (numFramesPerSwap == swapFrames)
205 return true;
206
207 if (auto GLXSwapIntervalSGI
208 = (PFNGLXSWAPINTERVALSGIPROC) OpenGLHelpers::getExtensionFunction ("glXSwapIntervalSGI"))
209 {
210 XWindowSystemUtilities::ScopedXLock xLock;
211 swapFrames = numFramesPerSwap;
212 GLXSwapIntervalSGI (numFramesPerSwap);
213 return true;
214 }
215
216 return false;
217 }

232private:
233 bool tryChooseVisual (const OpenGLPixelFormat& format, const std::vector<GLint>& optionalAttribs)
234 {
235 std::vector<GLint> allAttribs
236 {
237 GLX_RGBA,
238 GLX_DOUBLEBUFFER,
239 GLX_RED_SIZE, format.redBits,
240 GLX_GREEN_SIZE, format.greenBits,
241 GLX_BLUE_SIZE, format.blueBits,
242 GLX_ALPHA_SIZE, format.alphaBits,
243 GLX_DEPTH_SIZE, format.depthBufferBits,
244 GLX_STENCIL_SIZE, format.stencilBufferBits,
245 GLX_ACCUM_RED_SIZE, format.accumulationBufferRedBits,
246 GLX_ACCUM_GREEN_SIZE, format.accumulationBufferGreenBits,
247 GLX_ACCUM_BLUE_SIZE, format.accumulationBufferBlueBits,
248 GLX_ACCUM_ALPHA_SIZE, format.accumulationBufferAlphaBits
249 };
250
251 allAttribs.insert (allAttribs.end(), optionalAttribs.begin(), optionalAttribs.end());
252
253 allAttribs.push_back (None);
254
255 bestVisual = glXChooseVisual (display, X11Symbols::getInstance()->xDefaultScreen (display), allAttribs.data());
256
257 return bestVisual != nullptr;
258 }
