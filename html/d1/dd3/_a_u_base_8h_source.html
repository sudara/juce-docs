316 // Override this method if your AU processes multiple output busses completely independently --
317 // you'll want to just call Render without the NeedsToRender check.
318 // Otherwise, override Render().
319 //
320 // N.B. Implementations of this method can assume that the output's buffer list has already been
321 // prepared and access it with GetOutput(inBusNumber)->GetBufferList() instead of
322 // GetOutput(inBusNumber)->PrepareBuffer(nFrames) -- if PrepareBuffer is called, a
323 // copy may occur after rendering.
325 virtual OSStatus RenderBus( AudioUnitRenderActionFlags & ioActionFlags,
326 const AudioTimeStamp & inTimeStamp,
327 UInt32 inBusNumber,
328 UInt32 inNumberFrames)
329 {
330 if (NeedsToRender(inTimeStamp))
331 return Render(ioActionFlags, inTimeStamp, inNumberFrames);
332 return noErr; // was presumably already rendered via another bus
333 }
334
335 // N.B. For a unit with only one output bus, it can assume in its implementation of this
336 // method that the output's buffer list has already been prepared and access it with
337 // GetOutput(0)->GetBufferList() instead of GetOutput(0)->PrepareBuffer(nFrames)
338 // -- if PrepareBuffer is called, a copy may occur after rendering.
340 virtual OSStatus Render( AudioUnitRenderActionFlags & ioActionFlags,
341 const AudioTimeStamp & inTimeStamp,
342 UInt32 inNumberFrames)
343 {
344 return noErr;
345 }

795private:
797 // shared between Render and RenderSlice, inlined to minimize function call overhead
798 OSStatus DoRenderBus( AudioUnitRenderActionFlags & ioActionFlags,
799 const AudioTimeStamp & inTimeStamp,
800 UInt32 inBusNumber,
801 AUOutputElement * theOutput,
802 UInt32 inNumberFrames,
803 AudioBufferList & ioData)
804 {
805 if (theOutput != NULL)
806 {
807 if (ioData.mBuffers[0].mData == NULL || (theOutput->WillAllocateBuffer() && Outputs().GetNumberOfElements() > 1))
808 // will render into cache buffer
809 theOutput->PrepareBuffer(inNumberFrames);
810 else
811 // will render into caller's buffer
812 theOutput->SetBufferList(ioData);
813 }
814
815 OSStatus result = RenderBus(ioActionFlags, inTimeStamp, inBusNumber, inNumberFrames);
816 if (result == noErr && theOutput != NULL) {
817 if (ioData.mBuffers[0].mData == NULL) {
818 theOutput->CopyBufferListTo(ioData);
819 AUTRACE(kCATrace_AUBaseDoRenderBus, mComponentInstance, inNumberFrames, (intptr_t)theOutput->GetBufferList().mBuffers[0].mData, 0, *(UInt32 *)ioData.mBuffers[0].mData);
820 } else {
821 theOutput->CopyBufferContentsTo(ioData);
822 AUTRACE(kCATrace_AUBaseDoRenderBus, mComponentInstance, inNumberFrames, (intptr_t)theOutput->GetBufferList().mBuffers[0].mData, (intptr_t)ioData.mBuffers[0].mData, *(UInt32 *)ioData.mBuffers[0].mData);
823 theOutput->InvalidateBufferList();
824 }
825 }
826 return result;
827 }
