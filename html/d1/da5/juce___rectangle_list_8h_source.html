111 void add (RectangleType rect)
112 {
113 jassert (rect.isFinite()); // You must provide a valid rectangle to this method!
114
115 if (! rect.isEmpty())
116 {
117 if (isEmpty())
118 {
119 rects.add (rect);
120 }
121 else
122 {
123 bool anyOverlaps = false;
124
125 for (int j = rects.size(); --j >= 0;)
126 {
127 auto& ourRect = rects.getReference (j);
128
129 if (rect.intersects (ourRect))
130 {
131 if (rect.contains (ourRect))
132 rects.remove (j);
133 else if (! ourRect.reduceIfPartlyContainedIn (rect))
134 anyOverlaps = true;
135 }
136 }
137
138 if (anyOverlaps && ! isEmpty())
139 {
140 RectangleList r (rect);
141
142 for (auto& ourRect : rects)
143 {
144 if (rect.intersects (ourRect))
145 {
146 r.subtract (ourRect);
147
148 if (r.isEmpty())
149 return;
150 }
151 }
152
153 rects.addArray (r.rects);
154 }
155 else
156 {
157 rects.add (rect);
158 }
159 }
160 }
161 }

205 void subtract (RectangleType rect)
206 {
207 if (auto numRects = rects.size())
208 {
209 auto x1 = rect.getX();
210 auto y1 = rect.getY();
211 auto x2 = x1 + rect.getWidth();
212 auto y2 = y1 + rect.getHeight();
213
214 for (int i = numRects; --i >= 0;)
215 {
216 auto& r = rects.getReference (i);
217
218 auto rx1 = r.getX();
219 auto ry1 = r.getY();
220 auto rx2 = rx1 + r.getWidth();
221 auto ry2 = ry1 + r.getHeight();
222
223 if (! (x2 <= rx1 || x1 >= rx2 || y2 <= ry1 || y1 >= ry2))
224 {
225 if (x1 > rx1 && x1 < rx2)
226 {
227 if (y1 <= ry1 && y2 >= ry2 && x2 >= rx2)
228 {
229 r.setWidth (x1 - rx1);
230 }
231 else
232 {
233 r.setX (x1);
234 r.setWidth (rx2 - x1);
235
236 rects.insert (++i, RectangleType (rx1, ry1, x1 - rx1, ry2 - ry1));
237 ++i;
238 }
239 }
240 else if (x2 > rx1 && x2 < rx2)
241 {
242 r.setX (x2);
243 r.setWidth (rx2 - x2);
244
245 if (y1 > ry1 || y2 < ry2 || x1 > rx1)
246 {
247 rects.insert (++i, RectangleType (rx1, ry1, x2 - rx1, ry2 - ry1));
248 ++i;
249 }
250 }
251 else if (y1 > ry1 && y1 < ry2)
252 {
253 if (x1 <= rx1 && x2 >= rx2 && y2 >= ry2)
254 {
255 r.setHeight (y1 - ry1);
256 }
257 else
258 {
259 r.setY (y1);
260 r.setHeight (ry2 - y1);
261
262 rects.insert (++i, RectangleType (rx1, ry1, rx2 - rx1, y1 - ry1));
263 ++i;
264 }
265 }
266 else if (y2 > ry1 && y2 < ry2)
267 {
268 r.setY (y2);
269 r.setHeight (ry2 - y2);
270
271 if (x1 > rx1 || x2 < rx2 || y1 > ry1)
272 {
273 rects.insert (++i, RectangleType (rx1, ry1, rx2 - rx1, y2 - ry1));
274 ++i;
275 }
276 }
277 else
278 {
279 rects.remove (i);
280 }
281 }
282 }

315 bool clipTo (RectangleType rect)
316 {
317 jassert (rect.isFinite()); // You must provide a valid rectangle to this method!
318
319 bool notEmpty = false;
320
321 if (rect.isEmpty())
322 {
323 clear();
324 }
325 else
326 {
327 for (int i = rects.size(); --i >= 0;)
328 {
329 auto& r = rects.getReference (i);
330
331 if (! rect.intersectRectangle (r))
332 rects.remove (i);
333 else
334 notEmpty = true;
335 }
336 }
337
338 return notEmpty;
339 }

350 template <typename OtherValueType>
351 bool clipTo (const RectangleList<OtherValueType>& other)
352 {
353 if (isEmpty())
354 return false;
355
356 RectangleList result;
357
358 for (auto& rect : rects)
359 {
360 for (auto& r : other)
361 {
362 auto clipped = r.template toType<ValueType>();
363
364 if (rect.intersectRectangle (clipped))
365 result.rects.add (clipped);
366 }
367 }
368
369 swapWith (result);
370 return ! isEmpty();
371 }

484 RectangleType getBounds() const noexcept
485 {
486 if (isEmpty())
487 return {};
488
489 auto& r = rects.getReference (0);
490
491 if (rects.size() == 1)
492 return r;
493
494 auto minX = r.getX();
495 auto minY = r.getY();
496 auto maxX = minX + r.getWidth();
497 auto maxY = minY + r.getHeight();
498
499 for (int i = rects.size(); --i > 0;)
500 {
501 auto& r2 = rects.getReference (i);
502
503 minX = jmin (minX, r2.getX());
504 minY = jmin (minY, r2.getY());
505 maxX = jmax (maxX, r2.getRight());
506 maxY = jmax (maxY, r2.getBottom());
507 }
508
509 return { minX, minY, maxX - minX, maxY - minY };
510 }

518 void consolidate()
519 {
520 for (int i = 0; i < rects.size() - 1; ++i)
521 {
522 auto& r = rects.getReference (i);
523 auto rx1 = r.getX();
524 auto ry1 = r.getY();
525 auto rx2 = rx1 + r.getWidth();
526 auto ry2 = ry1 + r.getHeight();
527
528 for (int j = rects.size(); --j > i;)
529 {
530 auto& r2 = rects.getReference (j);
531 auto jrx1 = r2.getX();
532 auto jry1 = r2.getY();
533 auto jrx2 = jrx1 + r2.getWidth();
534 auto jry2 = jry1 + r2.getHeight();
535
536 // if the vertical edges of any blocks are touching and their horizontals don't
537 // line up, split them horizontally..
538 if (jrx1 == rx2 || jrx2 == rx1)
539 {
540 if (jry1 > ry1 && jry1 < ry2)
541 {
542 r.setHeight (jry1 - ry1);
543 rects.add (RectangleType (rx1, jry1, rx2 - rx1, ry2 - jry1));
544 i = -1;
545 break;
546 }
547
548 if (jry2 > ry1 && jry2 < ry2)
549 {
550 r.setHeight (jry2 - ry1);
551 rects.add (RectangleType (rx1, jry2, rx2 - rx1, ry2 - jry2));
552 i = -1;
553 break;
554 }
555 else if (ry1 > jry1 && ry1 < jry2)
556 {
557 r2.setHeight (ry1 - jry1);
558 rects.add (RectangleType (jrx1, ry1, jrx2 - jrx1, jry2 - ry1));
559 i = -1;
560 break;
561 }
562 else if (ry2 > jry1 && ry2 < jry2)
563 {
564 r2.setHeight (ry2 - jry1);
565 rects.add (RectangleType (jrx1, ry2, jrx2 - jrx1, jry2 - ry2));
566 i = -1;
567 break;
568 }
569 }
570 }
571 }
572
573 for (int i = 0; i < rects.size() - 1; ++i)
574 {
575 auto& r = rects.getReference (i);
576
577 for (int j = rects.size(); --j > i;)
578 {
579 if (r.enlargeIfAdjacent (rects.getReference (j)))
580 {
581 rects.remove (j);
582 i = -1;
583 break;
584 }
585 }
586 }
