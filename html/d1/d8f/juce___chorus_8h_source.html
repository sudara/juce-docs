89 template <typename ProcessContext>
90 void process (const ProcessContext& context) noexcept
91 {
92 const auto& inputBlock = context.getInputBlock();
93 auto& outputBlock = context.getOutputBlock();
94 const auto numChannels = outputBlock.getNumChannels();
95 const auto numSamples = outputBlock.getNumSamples();
96
97 jassert (inputBlock.getNumChannels() == numChannels);
98 jassert (inputBlock.getNumChannels() == lastOutput.size());
99 jassert (inputBlock.getNumSamples() == numSamples);
100
101 if (context.isBypassed)
102 {
103 outputBlock.copyFrom (inputBlock);
104 return;
105 }
106
107 auto delayValuesBlock = AudioBlock<SampleType>(bufferDelayTimes).getSubBlock (0, numSamples);
108 auto contextDelay = ProcessContextReplacing<SampleType> (delayValuesBlock);
109 delayValuesBlock.clear();
110
111 osc.process (contextDelay);
112 delayValuesBlock.multiplyBy (oscVolume);
113
114 auto* delaySamples = bufferDelayTimes.getWritePointer (0);
115
116 for (size_t i = 0; i < numSamples; ++i)
117 {
118 auto lfo = jmax (static_cast<SampleType> (1.0), maximumDelayModulation * delaySamples[i] + centreDelay);
119 delaySamples[i] = static_cast<SampleType> (lfo * sampleRate / 1000.0);
120 }
121
122 dryWet.pushDrySamples (inputBlock);
123
124 for (size_t channel = 0; channel < numChannels; ++channel)
125 {
126 auto* inputSamples = inputBlock .getChannelPointer (channel);
127 auto* outputSamples = outputBlock.getChannelPointer (channel);
128
129 for (size_t i = 0; i < numSamples; ++i)
130 {
131 auto input = inputSamples[i];
132 auto output = input - lastOutput[channel];
133
134 delay.pushSample ((int) channel, output);
135 delay.setDelay (delaySamples[i]);
136 output = delay.popSample ((int) channel);
137
138 outputSamples[i] = output;
139 lastOutput[channel] = output * feedbackVolume[channel].getNextValue();
140 }
141 }
142
143 dryWet.mixWetSamples (outputBlock);
144 }
145
146private:
147 //==============================================================================
148 void update();
149
150 //==============================================================================
151 Oscillator<SampleType> osc;
152 DelayLine<SampleType, DelayLineInterpolationTypes::Linear> delay;
153 SmoothedValue<SampleType, ValueSmoothingTypes::Linear> oscVolume;
154 std::vector<SmoothedValue<SampleType, ValueSmoothingTypes::Linear>> feedbackVolume { 2 };
155 DryWetMixer<SampleType> dryWet;
156 std::vector<SampleType> lastOutput { 2 };
157 AudioBuffer<SampleType> bufferDelayTimes;
158
159 double sampleRate = 44100.0;
160 SampleType rate = 1.0, depth = 0.25, feedback = 0.0, mix = 0.5,
161 centreDelay = 7.0;
162
163 static constexpr SampleType maxDepth = 1.0,
164 maxCentreDelayMs = 100.0,
165 oscVolumeMultiplier = 0.5,
166 maximumDelayModulation = 20.0;
