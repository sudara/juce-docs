44class OpenGLContext::NativeContext
45{
46public:
47 NativeContext (Component& c,
48 const OpenGLPixelFormat& pixFormat,
49 void* contextToShare,
50 bool multisampling,
51 OpenGLVersion version)
52 : component (c), openGLversion (version),
53 useDepthBuffer (pixFormat.depthBufferBits > 0),
54 useMSAA (multisampling)
55 {
56 JUCE_AUTORELEASEPOOL
57 {
58 if (auto* peer = component.getPeer())
59 {
60 auto bounds = peer->getAreaCoveredBy (component);
61
62 view = [[JuceGLView alloc] initWithFrame: convertToCGRect (bounds)];
63 view.opaque = YES;
64 view.hidden = NO;
65 view.backgroundColor = [UIColor blackColor];
66 view.userInteractionEnabled = NO;
67
68 glLayer = (CAEAGLLayer*) [view layer];
69 glLayer.opaque = true;
70
71 updateWindowPosition (bounds);
72
73 [((UIView*) peer->getNativeHandle()) addSubview: view];
74
75 if (version == openGL3_2 && [[UIDevice currentDevice].systemVersion floatValue] >= 7.0)
76 {
77 if (! createContext (kEAGLRenderingAPIOpenGLES3, contextToShare))
78 {
79 releaseContext();
80 createContext (kEAGLRenderingAPIOpenGLES2, contextToShare);
81 }
82 }
83 else
84 {
85 createContext (kEAGLRenderingAPIOpenGLES2, contextToShare);
86 }
87
88 if (context != nil)
89 {
90 // I'd prefer to put this stuff in the initialiseOnRenderThread() call, but doing
91 // so causes mysterious timing-related failures.
92 [EAGLContext setCurrentContext: context];
93 gl::loadFunctions();
94 createGLBuffers();
95 deactivateCurrentContext();
96 }
97 else
98 {
99 jassertfalse;
100 }
101 }
102 else
103 {
104 jassertfalse;
105 }
106 }
107 }
108
109 ~NativeContext()
110 {
111 releaseContext();
112 [view removeFromSuperview];
113 [view release];
114 }

116 bool initialiseOnRenderThread (OpenGLContext&) { return true; }

149 void swapBuffers()
150 {
151 if (useMSAA)
152 {
153 glBindFramebuffer (GL_DRAW_FRAMEBUFFER, frameBufferHandle);
154 glBindFramebuffer (GL_READ_FRAMEBUFFER, msaaBufferHandle);
155
156 if (openGLversion >= openGL3_2)
157 {
158 auto w = roundToInt (lastBounds.getWidth() * glLayer.contentsScale);
159 auto h = roundToInt (lastBounds.getHeight() * glLayer.contentsScale);
160
161 glBlitFramebuffer (0, 0, w, h,
162 0, 0, w, h,
163 GL_COLOR_BUFFER_BIT,
164 GL_NEAREST);
165 }
166 else
167 {
168 ::glResolveMultisampleFramebufferAPPLE();
169 }
170 }
171
172 glBindRenderbuffer (GL_RENDERBUFFER, colorBufferHandle);
173 [context presentRenderbuffer: GL_RENDERBUFFER];
174
175 if (needToRebuildBuffers)
176 {
177 needToRebuildBuffers = false;
178
179 freeGLBuffers();
180 createGLBuffers();
181 makeActive();
182 }
183 }
184
185 void updateWindowPosition (Rectangle<int> bounds)
186 {
187 view.frame = convertToCGRect (bounds);
188 glLayer.contentsScale = (CGFloat) (Desktop::getInstance().getDisplays().getPrimaryDisplay()->scale
189 / component.getDesktopScaleFactor());
190
191 if (lastBounds != bounds)
192 {
193 lastBounds = bounds;
194 needToRebuildBuffers = true;
195 }
196 }

201 return false;
202 }

223 bool createContext (EAGLRenderingAPI type, void* contextToShare)
224 {
225 jassert (context == nil);
226 context = [EAGLContext alloc];

228 context = contextToShare != nullptr

242 void createGLBuffers()
243 {
244 glGenFramebuffers (1, &frameBufferHandle);
245 glGenRenderbuffers (1, &colorBufferHandle);
246
247 glBindFramebuffer (GL_FRAMEBUFFER, frameBufferHandle);
248 glBindRenderbuffer (GL_RENDERBUFFER, colorBufferHandle);
249
250 glFramebufferRenderbuffer (GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER, colorBufferHandle);
251
252 bool ok = [context renderbufferStorage: GL_RENDERBUFFER fromDrawable: glLayer];
253 jassert (ok); ignoreUnused (ok);
254
255 GLint width, height;
256 glGetRenderbufferParameteriv (GL_RENDERBUFFER, GL_RENDERBUFFER_WIDTH, &width);
257 glGetRenderbufferParameteriv (GL_RENDERBUFFER, GL_RENDERBUFFER_HEIGHT, &height);
258
259 if (useMSAA)
260 {
261 glGenFramebuffers (1, &msaaBufferHandle);
262 glGenRenderbuffers (1, &msaaColorHandle);
263
264 glBindFramebuffer (GL_FRAMEBUFFER, msaaBufferHandle);
265 glBindRenderbuffer (GL_RENDERBUFFER, msaaColorHandle);
266
267 glRenderbufferStorageMultisample (GL_RENDERBUFFER, 4, GL_RGBA8, width, height);
268
269 glFramebufferRenderbuffer (GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER, msaaColorHandle);
270 }
271
272 if (useDepthBuffer)
273 {
274 glGenRenderbuffers (1, &depthBufferHandle);
275 glBindRenderbuffer (GL_RENDERBUFFER, depthBufferHandle);
276
277 if (useMSAA)
278 glRenderbufferStorageMultisample (GL_RENDERBUFFER, 4, GL_DEPTH_COMPONENT16, width, height);
279 else
280 glRenderbufferStorage (GL_RENDERBUFFER, GL_DEPTH_COMPONENT16, width, height);
281
282 glFramebufferRenderbuffer (GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_RENDERBUFFER, depthBufferHandle);
283 }
284
285 jassert (glCheckFramebufferStatus (GL_FRAMEBUFFER) == GL_FRAMEBUFFER_COMPLETE);
286 JUCE_CHECK_OPENGL_ERROR
287 }
288
289 void freeGLBuffers()
290 {
291 JUCE_CHECK_OPENGL_ERROR
292 [context renderbufferStorage: GL_RENDERBUFFER fromDrawable: nil];
293
294 deleteFrameBuffer (frameBufferHandle);
295 deleteFrameBuffer (msaaBufferHandle);
296 deleteRenderBuffer (colorBufferHandle);
297 deleteRenderBuffer (depthBufferHandle);
298 deleteRenderBuffer (msaaColorHandle);
299
300 JUCE_CHECK_OPENGL_ERROR
301 }
302
303 static void deleteFrameBuffer (GLuint& i) { if (i != 0) glDeleteFramebuffers (1, &i); i = 0; }
304 static void deleteRenderBuffer (GLuint& i) { if (i != 0) glDeleteRenderbuffers (1, &i); i = 0; }

306 JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR (NativeContext)
