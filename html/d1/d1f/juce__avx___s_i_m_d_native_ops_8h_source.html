237 static forcedinline int8_t JUCE_VECTOR_CALLTYPE sum (__m256i a) noexcept
238 {
239 __m256i lo = _mm256_unpacklo_epi8 (a, _mm256_setzero_si256());
240 __m256i hi = _mm256_unpackhi_epi8 (a, _mm256_setzero_si256());
241
242 for (int i = 0; i < 3; ++i)
243 {
244 lo = _mm256_hadd_epi16 (lo, lo);
245 hi = _mm256_hadd_epi16 (hi, hi);
246 }
247
248 #if JUCE_GCC
249 return (int8_t) ((lo[0] & 0xff) +
250 (hi[0] & 0xff) +
251 (lo[2] & 0xff) +
252 (hi[2] & 0xff));
253 #else
254 constexpr int mask = (2 << 0) | (3 << 2) | (0 << 4) | (1 << 6);
255
256 return (int8_t) ((_mm256_cvtsi256_si32 (lo) & 0xff) +
257 (_mm256_cvtsi256_si32 (hi) & 0xff) +
258 (_mm256_cvtsi256_si32 (_mm256_permute4x64_epi64 (lo, mask)) & 0xff) +
259 (_mm256_cvtsi256_si32 (_mm256_permute4x64_epi64 (hi, mask)) & 0xff));
260 #endif
261 }
262
263 static forcedinline __m256i JUCE_VECTOR_CALLTYPE mul (__m256i a, __m256i b)
264 {
265 // unpack and multiply
266 __m256i even = _mm256_mullo_epi16 (a, b);
267 __m256i odd = _mm256_mullo_epi16 (_mm256_srli_epi16 (a, 8), _mm256_srli_epi16 (b, 8));
268
269 return _mm256_or_si256 (_mm256_slli_epi16 (odd, 8),
270 _mm256_srli_epi16 (_mm256_slli_epi16 (even, 8), 8));
271 }
