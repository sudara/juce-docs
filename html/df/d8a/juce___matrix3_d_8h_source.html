110 static Matrix3D rotation (Vector3D<Type> eulerAngleRadians) noexcept
111 {
112 auto cx = std::cos (eulerAngleRadians.x), sx = std::sin (eulerAngleRadians.x),
113 cy = std::cos (eulerAngleRadians.y), sy = std::sin (eulerAngleRadians.y),
114 cz = std::cos (eulerAngleRadians.z), sz = std::sin (eulerAngleRadians.z);
115
116 return { (cy * cz) + (sx * sy * sz), cx * sz, (cy * sx * sz) - (cz * sy), 0,
117 (cz * sx * sy) - (cy * sz), cx * cz, (cy * cz * sx) + (sy * sz), 0,
118 cx * sy, -sx, cx * cy, 0,
119 0, 0, 0, Type (1) };
120 }
