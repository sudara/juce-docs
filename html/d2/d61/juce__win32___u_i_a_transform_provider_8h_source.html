37 JUCE_COMRESULT Move (double x, double y) override
38 {
39 if (! isElementValid())
40 return (HRESULT) UIA_E_ELEMENTNOTAVAILABLE;
41
42 if (auto* peer = getPeer())
43 {
44 RECT rect;
45 GetWindowRect ((HWND) peer->getNativeHandle(), &rect);
46
47 rect.left = roundToInt (x);
48 rect.top = roundToInt (y);
49
50 auto bounds = Rectangle<int>::leftTopRightBottom (rect.left, rect.top, rect.right, rect.bottom);
51
52 peer->setBounds (Desktop::getInstance().getDisplays().physicalToLogical (bounds),
53 peer->isFullScreen());
54 }
55
56 return S_OK;
57 }
58
59 JUCE_COMRESULT Resize (double width, double height) override
60 {
61 if (! isElementValid())
62 return (HRESULT) UIA_E_ELEMENTNOTAVAILABLE;
63
64 if (auto* peer = getPeer())
65 {
66 auto scale = peer->getPlatformScaleFactor();
67
68 peer->getComponent().setSize (roundToInt (width / scale),
69 roundToInt (height / scale));
70 }
71
72 return S_OK;
73 }
74
75 JUCE_COMRESULT Rotate (double) override
76 {
77 if (! isElementValid())

79
80 return (HRESULT) UIA_E_NOTSUPPORTED;
81 }

109 });
110 }
