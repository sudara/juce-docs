127template <typename Type>
128Type jmap (Type sourceValue, Type sourceRangeMin, Type sourceRangeMax, Type targetRangeMin, Type targetRangeMax)
129{
130 jassert (sourceRangeMax != sourceRangeMin); // mapping from a range of zero will produce NaN!
131 return targetRangeMin + ((targetRangeMax - targetRangeMin) * (sourceValue - sourceRangeMin)) / (sourceRangeMax - sourceRangeMin);
132}

144template <typename Type>
145Type mapToLog10 (Type value0To1, Type logRangeMin, Type logRangeMax)
146{
147 jassert (logRangeMin > 0);
148 jassert (logRangeMax > 0);
149
150 auto logMin = std::log10 (logRangeMin);
151 auto logMax = std::log10 (logRangeMax);
152
153 return std::pow ((Type) 10.0, value0To1 * (logMax - logMin) + logMin);
154}

166template <typename Type>
167Type mapFromLog10 (Type valueInLogRange, Type logRangeMin, Type logRangeMax)
168{
169 jassert (logRangeMin > 0);
170 jassert (logRangeMax > 0);
171
172 auto logMin = std::log10 (logRangeMin);
173 auto logMax = std::log10 (logRangeMax);
174
175 return (std::log10 (valueInLogRange) - logMin) / (logMax - logMin);
176}

179template <typename Type, typename Size>
180Type findMinimum (const Type* data, Size numValues)
181{
182 if (numValues <= 0)
183 return Type (0);
184
185 auto result = *data++;
186
187 while (--numValues > 0) // (> 0 rather than >= 0 because we've already taken the first sample)
188 {
189 auto v = *data++;
190
191 if (v < result)
192 result = v;
193 }
194
195 return result;
196}

219template <typename Type>
220void findMinAndMax (const Type* values, int numValues, Type& lowest, Type& highest)
221{
222 if (numValues <= 0)
223 {
224 lowest = Type (0);
225 highest = Type (0);
226 }
227 else
228 {
229 auto mn = *values++;
230 auto mx = mn;
231
232 while (--numValues > 0) // (> 0 rather than >= 0 because we've already taken the first sample)
233 {
234 auto v = *values++;
235
236 if (mx < v) mx = v;
237 if (v < mn) mn = v;
238 }
239
240 lowest = mn;
241 highest = mx;
242 }
243}

534inline int nextPowerOfTwo (int n) noexcept
535{
536 --n;
537 n |= (n >> 1);
538 n |= (n >> 2);
539 n |= (n >> 4);

552inline int countNumberOfBits (uint32 n) noexcept
553{
554 n -= ((n >> 1) & 0x55555555);
555 n = (((n >> 2) & 0x33333333) + (n & 0x33333333));
556 n = (((n >> 4) + n) & 0x0f0f0f0f);
557 n += (n >> 8);
558 n += (n >> 16);
559 return (int) (n & 0x3f);
560}

563inline int countNumberOfBits (uint64 n) noexcept
564{
565 return countNumberOfBits ((uint32) n) + countNumberOfBits ((uint32) (n >> 32));
566}
