274 template <typename Integral, std::enable_if_t<std::is_integral<Integral>::value, int> = 0>
275 static Range findMinAndMax (const ValueType* values, Integral numValues) noexcept
276 {
277 if (numValues <= 0)
278 return Range();
279
280 const ValueType first (*values++);
281 Range r (first, first);
282
283 while (--numValues > 0) // (> 0 rather than >= 0 because we've already taken the first sample)
284 {
285 const ValueType v (*values++);
286
287 if (r.end < v) r.end = v;
288 if (v < r.start) r.start = v;
289 }
290
291 return r;
292 }
