46 static PacketX2 makeSysEx (uint8_t group,
47 uint8_t status,
48 uint8_t numBytes,
49 const uint8_t* data)
50 {
51 jassert (numBytes <= 6);
52
53 std::array<uint8_t, 8> bytes{{}};
54 bytes[0] = (0x3 << 0x4) | group;
55 bytes[1] = (uint8_t) (status << 0x4) | numBytes;
56
57 std::memcpy (bytes.data() + 2, data, numBytes);
58
59 std::array<uint32_t, 2> words;
60
61 size_t index = 0;
62
63 for (auto& word : words)
64 word = ByteOrder::bigEndianInt (bytes.data() + 4 * index++);
65
66 return PacketX2 { words };
67 }
68
69 static PacketX4 makeSysEx8 (uint8_t group,
70 uint8_t status,
71 uint8_t numBytes,
72 uint8_t dataStart,
73 const uint8_t* data)
74 {
75 jassert (numBytes <= 16 - dataStart);
76
77 std::array<uint8_t, 16> bytes{{}};
78 bytes[0] = (0x5 << 0x4) | group;
79 bytes[1] = (uint8_t) (status << 0x4) | numBytes;
80
81 std::memcpy (bytes.data() + dataStart, data, numBytes);
82
83 std::array<uint32_t, 4> words;
84
85 size_t index = 0;
86
87 for (auto& word : words)
88 word = ByteOrder::bigEndianInt (bytes.data() + 4 * index++);
89
90 return PacketX4 { words };
91 }
92 };
