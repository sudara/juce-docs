52 void start()
53 {
54 if (isRunning())
55 stop();
56
57 shouldExit = false;
58
59 thread = std::thread { [this]
60 {
61 Thread::setCurrentThreadPriority (7);
62 Thread::setCurrentThreadName ("JUCE Plugin Message Thread");
63
64 MessageManager::getInstance()->setCurrentThreadAsMessageThread();
65 XWindowSystem::getInstance();
66
67 threadInitialised.signal();
68
69 for (;;)
70 {
71 if (! dispatchNextMessageOnSystemQueue (true))
72 Thread::sleep (1);
73
74 if (shouldExit)
75 break;
76 }
77 } };
78
79 threadInitialised.wait();
80 }
81
82 void stop()
83 {
84 if (! isRunning())
85 return;

87 shouldExit = true;

89 }
