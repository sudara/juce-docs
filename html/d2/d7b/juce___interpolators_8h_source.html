41 struct WindowedSincTraits
42 {
43 static constexpr float algorithmicLatency = 100.0f;
44
45 static forcedinline float windowedSinc (float firstFrac, int index) noexcept
46 {
47 auto index2 = index + 1;
48 auto frac = firstFrac;
49
50 auto value1 = lookupTable[index];
51 auto value2 = lookupTable[index2];
52
53 return value1 + (frac * (value2 - value1));
54 }
55
56 static forcedinline float valueAtOffset (const float* const inputs, const float offset, int indexBuffer) noexcept
57 {
58 const int numCrossings = 100;
59 const float floatCrossings = (float) numCrossings;
60 float result = 0.0f;
61
62 auto samplePosition = indexBuffer;
63 float firstFrac = 0.0f;
64 float lastSincPosition = -1.0f;
65 int index = 0, sign = -1;
66
67 for (int i = -numCrossings; i <= numCrossings; ++i)
68 {
69 auto sincPosition = (1.0f - offset) + (float) i;
70
71 if (i == -numCrossings || (sincPosition >= 0 && lastSincPosition < 0))
72 {
73 auto indexFloat = (sincPosition >= 0.f ? sincPosition : -sincPosition) * 100.0f;
74 auto indexFloored = std::floor (indexFloat);
75 index = (int) indexFloored;
76 firstFrac = indexFloat - indexFloored;
77 sign = (sincPosition < 0 ? -1 : 1);
78 }
79
80 if (sincPosition == 0.0f)
81 result += inputs[samplePosition];
82 else if (sincPosition < floatCrossings && sincPosition > -floatCrossings)
83 result += inputs[samplePosition] * windowedSinc (firstFrac, index);
84
85 if (++samplePosition == numCrossings * 2)
86 samplePosition = 0;
87
88 lastSincPosition = sincPosition;
89 index += 100 * sign;
90 }
91
92 return result;
93 }
94
95 static const float lookupTable[10001];
96 };

105 struct CatmullRomTraits
106 {
107 //==============================================================================
108 static constexpr float algorithmicLatency = 2.0f;
109
110 static forcedinline float valueAtOffset (const float* const inputs, const float offset, int index) noexcept
111 {
112 auto y0 = inputs[index]; if (++index == 4) index = 0;
113 auto y1 = inputs[index]; if (++index == 4) index = 0;
114 auto y2 = inputs[index]; if (++index == 4) index = 0;
115 auto y3 = inputs[index];
116
117 auto halfY0 = 0.5f * y0;
118 auto halfY3 = 0.5f * y3;
119
120 return y1 + offset * ((0.5f * y2 - halfY0)
121 + (offset * (((y0 + 2.0f * y2) - (halfY3 + 2.5f * y1))
122 + (offset * ((halfY3 + 1.5f * y1) - (halfY0 + 1.5f * y2))))));
123 }
124 };

126 struct LinearTraits
127 {
128 static constexpr float algorithmicLatency = 1.0f;
129
130 static forcedinline float valueAtOffset (const float* const inputs, const float offset, int index) noexcept
131 {
132 auto y0 = inputs[index];
133 auto y1 = inputs[index == 0 ? 1 : 0];
134
135 return y1 * offset + y0 * (1.0f - offset);
136 }
137 };

139 struct ZeroOrderHoldTraits
140 {
141 static constexpr float algorithmicLatency = 0.0f;
142
143 static forcedinline float valueAtOffset (const float* const inputs, const float, int) noexcept
144 {
145 return inputs[0];
146 }
147 };
