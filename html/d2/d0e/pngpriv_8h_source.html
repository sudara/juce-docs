91/* Compile time options.
92 * =====================
93 * In a multi-arch build the compiler may compile the code several times for the
94 * same object module, producing different binaries for different architectures.
95 * When this happens configure-time setting of the target host options cannot be
96 * done and this interferes with the handling of the ARM NEON optimizations, and
97 * possibly other similar optimizations. Put additional tests here; in general
98 * this is needed when the same option can be changed at both compile time and
99 * run time depending on the target OS (i.e. iOS vs Android.)
100 *
101 * NOTE: symbol prefixing does not pass $(CFLAGS) to the preprocessor, because
102 * this is not possible with certain compilers (Oracle SUN OS CC), as a result
103 * it is necessary to ensure that all extern functions that *might* be used
104 * regardless of $(CFLAGS) get declared in this file. The test on __ARM_NEON__
105 * below is one example of this behavior because it is controlled by the
106 * presence or not of -mfpu=neon on the GCC command line, it is possible to do
107 * this in $(CC), e.g. "CC=gcc -mfpu=neon", but people who build libpng rarely
108 * do this.
109 */
110#ifndef PNG_ARM_NEON_OPT
111 /* ARM NEON optimizations are being controlled by the compiler settings,
112 * typically the target FPU. If the FPU has been set to NEON (-mfpu=neon
113 * with GCC) then the compiler will define __ARM_NEON__ and we can rely
114 * unconditionally on NEON instructions not crashing, otherwise we must
115 * disable use of NEON instructions.
116 *
117 * NOTE: at present these optimizations depend on 'ALIGNED_MEMORY', so they
118 * can only be turned on automatically if that is supported too. If
119 * PNG_ARM_NEON_OPT is set in CPPFLAGS (to >0) then arm/arm_init.c will fail
120 * to compile with an appropriate #error if ALIGNED_MEMORY has been turned
121 * off.
122 *
123 * Note that gcc-4.9 defines __ARM_NEON instead of the deprecated
124 * __ARM_NEON__, so we check both variants.
125 *
126 * To disable ARM_NEON optimizations entirely, and skip compiling the
127 * associated assembler code, pass --enable-arm-neon=no to configure
128 * or put -DPNG_ARM_NEON_OPT=0 in CPPFLAGS.
129 */
130# if (defined(__ARM_NEON__) || defined(__ARM_NEON)) && \
131 defined(PNG_ALIGNED_MEMORY_SUPPORTED)
132# define PNG_ARM_NEON_OPT 2
133# else
134# define PNG_ARM_NEON_OPT 0
135# endif
136#endif

138#if PNG_ARM_NEON_OPT > 0
139 /* NEON optimizations are to be at least considered by libpng, so enable the
140 * callbacks to do this.
141 */
142# define PNG_FILTER_OPTIMIZATIONS png_init_filter_functions_neon

144 /* By default the 'intrinsics' code in arm/filter_neon_intrinsics.c is used
145 * if possible - if __ARM_NEON__ is set and the compiler version is not known
146 * to be broken. This is controlled by PNG_ARM_NEON_IMPLEMENTATION which can
147 * be:
148 *
149 * 1 The intrinsics code (the default with __ARM_NEON__)
150 * 2 The hand coded assembler (the default without __ARM_NEON__)
151 *
152 * It is possible to set PNG_ARM_NEON_IMPLEMENTATION in CPPFLAGS, however
153 * this is *NOT* supported and may cease to work even after a minor revision
154 * to libpng. It *is* valid to do this for testing purposes, e.g. speed
155 * testing or a new compiler, but the results should be communicated to the
156 * libpng implementation list for incorporation in the next minor release.
157 */
158# ifndef PNG_ARM_NEON_IMPLEMENTATION
159# if defined(__ARM_NEON__) || defined(__ARM_NEON)
160# if defined(__clang__)

173# endif /* __GNUC__ */

179# define PNG_ARM_NEON_IMPLEMENTATION 2
180# endif /* __aarch64__ */
181# endif /* __ARM_NEON__ */
182# endif /* !PNG_ARM_NEON_IMPLEMENTATION */

743/* This returns the number of trailing bits in the last byte of a row, 0 if the
744 * last byte is completely full of pixels. It is, in principle, (pixel_bits x
745 * width) % 8, but that would overflow for large 'width'. The second macro is
746 * the same except that it returns the number of unused bits in the last byte;
747 * (8-TRAILBITS), but 0 when TRAILBITS is 0.

753 * (png_uint_32), however the compiler always knows it is in the range 0..7.

756 (((pixel_bits) * ((width) % (png_uint_32)8)) % 8)

1249/* Combine a row of data, dealing with alpha, etc. if requested. 'row' is an
1250 * array of png_ptr->width pixels. If the image is not interlaced or this
1251 * is the final pass this just does a memcpy, otherwise the "display" flag
1252 * is used to determine whether to copy pixels that are not in the current pass.
1253 *
1254 * Because 'png_do_read_interlace' (below) replicates pixels this allows this
1255 * function to achieve the documented 'blocky' appearance during interlaced read
1256 * if display is 1 and the 'sparkle' appearance, where existing pixels in 'row'
1257 * are not changed if they are not in the current pass, when display is 0.
1258 *
1259 * 'display' must be 0 or 1, otherwise the memcpy will be done regardless.
1260 *
1261 * The API always reads from the png_struct row buffer and always assumes that
1262 * it is full width (png_do_read_interlace has already been called.)
1263 *
1264 * This function is only ever used to write to row buffers provided by the
1265 * caller of the relevant libpng API and the row must have already been
1266 * transformed by the read transformations.
1267 *
1268 * The PNG_USE_COMPILE_TIME_MASKS option causes generation of pre-computed
1269 * bitmasks for use within the code, otherwise runtime generated masks are used.
1270 * The default is compile time masks.
1271 */
1272#ifndef PNG_USE_COMPILE_TIME_MASKS
1273# define PNG_USE_COMPILE_TIME_MASKS 1
1274#endif
1275PNG_INTERNAL_FUNCTION(void,png_combine_row,(png_const_structrp png_ptr,
1276 png_bytep row, int display),PNG_EMPTY);

1872/* An internal API to validate the format of a floating point number.
1873 * The result is the index of the next character. If the number is
1874 * not valid it will be the index of a character in the supposed number.
1875 *
1876 * The format of a number is defined in the PNG extensions specification
1877 * and this API is strictly conformant to that spec, not anyone elses!
1878 *
1879 * The format as a regular expression is:
1880 *
1881 * [+-]?[0-9]+.?([Ee][+-]?[0-9]+)?
1882 *
1883 * or:
1884 *
1885 * [+-]?.[0-9]+(.[0-9]+)?([Ee][+-]?[0-9]+)?
1886 *
1887 * The complexity is that either integer or fraction must be present and the
1888 * fraction is permitted to have no digits only if the integer is present.
1889 *
1890 * NOTE: The dangling E problem.
1891 * There is a PNG valid floating point number in the following:
1892 *
1893 * PNG floating point numbers are not greedy.
1894 *
1895 * Working this out requires *TWO* character lookahead (because of the
1896 * sign), the parser does not do this - it will fail at the 'r' - this
1897 * doesn't matter for PNG sCAL chunk values, but it requires more care
1898 * if the value were ever to be embedded in something more complex. Use
1899 * ANSI-C strtod if you need the lookahead.
1900 */
1901/* State table for the parser. */
1902#define PNG_FP_INTEGER 0 /* before or in integer */
1903#define PNG_FP_FRACTION 1 /* before or in fraction */
1904#define PNG_FP_EXPONENT 2 /* before or in exponent */
1905#define PNG_FP_STATE 3 /* mask for the above */
1906#define PNG_FP_SAW_SIGN 4 /* Saw +/- in current state */
1907#define PNG_FP_SAW_DIGIT 8 /* Saw a digit in current state */
1908#define PNG_FP_SAW_DOT 16 /* Saw a dot in current state */
1909#define PNG_FP_SAW_E 32 /* Saw an E (or e) in current state */
1910#define PNG_FP_SAW_ANY 60 /* Saw any of the above 4 */
1911
1912/* These three values don't affect the parser. They are set but not used.
1913 */
1914#define PNG_FP_WAS_VALID 64 /* Preceding substring is a valid fp number */
1915#define PNG_FP_NEGATIVE 128 /* A negative number, including "-0" */
1916#define PNG_FP_NONZERO 256 /* A non-zero value */
1917#define PNG_FP_STICKY 448 /* The above three flags */
1918
1919/* This is available for the caller to store in 'state' if required. Do not
1920 * call the parser after setting it (the parser sometimes clears it.)
1921 */
1922#define PNG_FP_INVALID 512 /* Available for callers as a distinct value */
1923
1924/* Result codes for the parser (boolean - true meants ok, false means
1925 * not ok yet.)
1926 */
1927#define PNG_FP_MAYBE 0 /* The number may be valid in the future */
1928#define PNG_FP_OK 1 /* The number is valid */
1929
1930/* Tests on the sticky non-zero and negative flags. To pass these checks
1931 * the state must also indicate that the whole number is valid - this is
1932 * achieved by testing PNG_FP_SAW_DIGIT (see the implementation for why this
1933 * is equivalent to PNG_FP_OK above.)
1934 */
1935#define PNG_FP_NZ_MASK (PNG_FP_SAW_DIGIT | PNG_FP_NEGATIVE | PNG_FP_NONZERO)
1936 /* NZ_MASK: the string is valid and a non-zero negative value */
1937#define PNG_FP_Z_MASK (PNG_FP_SAW_DIGIT | PNG_FP_NONZERO)
1938 /* Z MASK: the string is valid and a non-zero value. */
1939 /* PNG_FP_SAW_DIGIT: the string is valid. */
1940#define PNG_FP_IS_ZERO(state) (((state) & PNG_FP_Z_MASK) == PNG_FP_SAW_DIGIT)
1941#define PNG_FP_IS_POSITIVE(state) (((state) & PNG_FP_NZ_MASK) == PNG_FP_Z_MASK)
1942#define PNG_FP_IS_NEGATIVE(state) (((state) & PNG_FP_NZ_MASK) == PNG_FP_NZ_MASK)

1949 * terminated by '\0').

1952 * a 'maybe' state even though a preceding integer.fraction is valid.

1955 * the parser again (from the start, with state 0) but with a string
