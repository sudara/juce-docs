247 template <class CharPointer>
248 void appendCharPointer (CharPointer startOfTextToAppend,
249 CharPointer endOfTextToAppend)
250 {
251 jassert (startOfTextToAppend.getAddress() != nullptr && endOfTextToAppend.getAddress() != nullptr);
252
253 size_t extraBytesNeeded = 0, numChars = 1;
254
255 for (auto t = startOfTextToAppend; t != endOfTextToAppend && ! t.isEmpty(); ++numChars)
256 extraBytesNeeded += CharPointerType::getBytesRequiredFor (t.getAndAdvance());
257
258 if (extraBytesNeeded > 0)
259 {
260 auto byteOffsetOfNull = getByteOffsetOfEnd();
261
262 preallocateBytes (byteOffsetOfNull + extraBytesNeeded);
263 CharPointerType (addBytesToPointer (text.getAddress(), (int) byteOffsetOfNull))
264 .writeWithCharLimit (startOfTextToAppend, (int) numChars);
265 }
266 }
267
269 void appendCharPointer (CharPointerType textToAppend);

276 template <class CharPointer>
277 void appendCharPointer (CharPointer textToAppend, size_t maxCharsToTake)
278 {
279 if (textToAppend.getAddress() != nullptr)
280 {
281 size_t extraBytesNeeded = 0, numChars = 1;
282
283 for (auto t = textToAppend; numChars <= maxCharsToTake && ! t.isEmpty(); ++numChars)
284 extraBytesNeeded += CharPointerType::getBytesRequiredFor (t.getAndAdvance());
285
286 if (extraBytesNeeded > 0)
287 {
288 auto byteOffsetOfNull = getByteOffsetOfEnd();
289
290 preallocateBytes (byteOffsetOfNull + extraBytesNeeded);
291 CharPointerType (addBytesToPointer (text.getAddress(), (int) byteOffsetOfNull))
292 .writeWithCharLimit (textToAppend, (int) numChars);
293 }
294 }

1117 static String toDecimalStringWithSignificantFigures (DecimalType number, int numberOfSignificantFigures)
1118 {
1119 jassert (numberOfSignificantFigures > 0);
1120
1121 if (number == 0)
1122 {
1123 if (numberOfSignificantFigures > 1)
1124 {
1125 String result ("0.0");
1126
1127 for (int i = 2; i < numberOfSignificantFigures; ++i)
1128 result += "0";
1129
1130 return result;
1131 }
1132
1133 return "0";
1134 }
1135
1136 auto numDigitsBeforePoint = (int) std::ceil (std::log10 (number < 0 ? -number : number));
1137
1138 auto shift = numberOfSignificantFigures - numDigitsBeforePoint;
1139 auto factor = std::pow (10.0, shift);
1140 auto rounded = std::round (number * factor) / factor;
1141
1142 std::stringstream ss;
1143 ss << std::fixed << std::setprecision (std::max (shift, 0)) << rounded;
1144 return ss.str();
1145 }
