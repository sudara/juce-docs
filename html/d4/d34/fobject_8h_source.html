136inline FObject* FObject::unknownToObject (FUnknown* unknown)
137{
138 FObject* object = nullptr;
139 if (unknown)
140 {
141 unknown->queryInterface (FObject::iid, (void**)&object);
142 if (object)
143 object->release (); // queryInterface has added ref
144 }
145 return object;
146}

242//-----------------------------------------------------------------------
243template <class T>
244inline void AssignShared (T*& dest, T* newPtr)
245{
246 if (dest == newPtr)
247 return;
248
249 if (dest)
250 dest->release ();
251 dest = newPtr;
252 if (dest)
253 dest->addRef ();
254}

256//-----------------------------------------------------------------------

259{
260 if (dest == newPtr)
261 return;
262
263 if (dest)

268}
