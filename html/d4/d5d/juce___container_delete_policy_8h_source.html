40template <typename ObjectType>
41struct ContainerDeletePolicy
42{
43 static void destroy (ObjectType* object)
44 {
45 // If the line below triggers a compiler error, it means that you are using
46 // an incomplete type for ObjectType (for example, a type that is declared
47 // but not defined). This is a problem because then the following delete is
48 // undefined behaviour. The purpose of the sizeof is to capture this situation.
49 // If this was caused by a OwnedArray of a forward-declared type, move the
50 // implementation of all methods trying to use the OwnedArray (e.g. the destructor
51 // of the class owning it) into cpp files where they can see to the definition
52 // of ObjectType. This should fix the error.
53 ignoreUnused (sizeof (ObjectType));
54
55 delete object;
56 }
57};
