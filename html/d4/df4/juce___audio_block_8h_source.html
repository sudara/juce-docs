121 AudioBlock (HeapBlock<char>& heapBlockToUseForAllocation,
122 size_t numberOfChannels, size_t numberOfSamples,
123 size_t alignmentInBytes = defaultAlignment) noexcept
124 : numChannels (static_cast<ChannelCountType> (numberOfChannels)),
125 numSamples (numberOfSamples)
126 {
127 auto roundedUpNumSamples = (numberOfSamples + elementMask) & ~elementMask;
128 auto channelSize = sizeof (SampleType) * roundedUpNumSamples;
129 auto channelListBytes = sizeof (SampleType*) * numberOfChannels;
130 auto extraBytes = alignmentInBytes - 1;
131
132 heapBlockToUseForAllocation.malloc (channelListBytes + extraBytes + channelSize * numberOfChannels);
133
134 auto* chanArray = unalignedPointerCast<SampleType**> (heapBlockToUseForAllocation.getData());
135 channels = chanArray;
136
137 auto* data = unalignedPointerCast<SampleType*> (addBytesToPointer (chanArray, channelListBytes));
138 data = snapPointerToAlignment (data, alignmentInBytes);
139
140 for (ChannelCountType i = 0; i < numChannels; ++i)
141 {
142 chanArray[i] = data;
143 data += roundedUpNumSamples;
144 }
145 }

522 Range<typename std::remove_const<NumericType>::type> findMinAndMax() const noexcept
523 {
524 if (numChannels == 0)
525 return {};
526
527 auto n = numSamples * sizeFactor;
528 auto minmax = FloatVectorOperations::findMinAndMax (getDataPointer (0), n);
529
530 for (size_t ch = 1; ch < numChannels; ++ch)
531 minmax = minmax.getUnionWith (FloatVectorOperations::findMinAndMax (getDataPointer (ch), n));
532
533 return minmax;
534 }

576 template <typename Src1SampleType, typename Src2SampleType, typename FunctionType>
577 static void process (AudioBlock<Src1SampleType> inBlock, AudioBlock<Src2SampleType> outBlock, FunctionType&& function)
578 {
579 auto len = inBlock.getNumSamples();
580 auto numChans = inBlock.getNumChannels();
581
582 jassert (len == outBlock.getNumSamples());
583 jassert (numChans == outBlock.getNumChannels());
584
585 for (ChannelCountType c = 0; c < numChans; ++c)
586 {
587 auto* src = inBlock.getChannelPointer (c);
588 auto* dst = outBlock.getChannelPointer (c);
589
590 for (size_t i = 0; i < len; ++i)
591 dst[i] = function (src[i]);
592 }
593 }

641 void moveInternal (size_t srcPos, size_t dstPos,
642 size_t numElements = std::numeric_limits<size_t>::max()) const noexcept
643 {
644 jassert (srcPos <= numSamples && dstPos <= numSamples);
645 auto len = jmin (numSamples - srcPos, numSamples - dstPos, numElements) * sizeof (SampleType);
646
647 if (len != 0)
648 for (size_t ch = 0; ch < numChannels; ++ch)
649 ::memmove (getChannelPointer (ch) + dstPos,
650 getChannelPointer (ch) + srcPos, len);
651 }
