304 void remapTable (int newNumberOfSlots)
305 {
306 const ScopedLockType sl (getLock());
307
308 Array<HashEntry*> newSlots;
309 newSlots.insertMultiple (0, nullptr, newNumberOfSlots);
310
311 for (auto i = getNumSlots(); --i >= 0;)
312 {
313 HashEntry* nextEntry = nullptr;
314
315 for (auto* entry = hashSlots.getUnchecked(i); entry != nullptr; entry = nextEntry)
316 {
317 auto hashIndex = generateHashFor (entry->key, newNumberOfSlots);
318
319 nextEntry = entry->nextEntry;
320 entry->nextEntry = newSlots.getUnchecked (hashIndex);
321
322 newSlots.set (hashIndex, entry);
323 }
324 }
325
326 hashSlots.swapWith (newSlots);
327 }
