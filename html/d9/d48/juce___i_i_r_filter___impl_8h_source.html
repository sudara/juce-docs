36template <typename NumericType>
37template <size_t Num>
38Coefficients<NumericType>& Coefficients<NumericType>::assignImpl (const NumericType* values)
39{
40 static_assert (Num % 2 == 0, "Must supply an even number of coefficients");
41 const auto a0Index = Num / 2;
42 const auto a0 = values[a0Index];
43 const auto a0Inv = a0 != NumericType() ? static_cast<NumericType> (1) / values[a0Index]
44 : NumericType();
45
46 coefficients.clearQuick();
47 coefficients.ensureStorageAllocated ((int) jmax ((size_t) 8, Num));
48
49 for (size_t i = 0; i < Num; ++i)
50 if (i != a0Index)
51 coefficients.add (values[i] * a0Inv);
52
53 return *this;
54}

70template <typename SampleType>
71void Filter<SampleType>::reset (SampleType resetToValue)
72{
73 auto newOrder = coefficients->getFilterOrder();
74
75 if (newOrder != order)
76 {
77 memory.malloc (jmax (order, newOrder, static_cast<size_t> (3)) + 1);
78 state = snapPointerToAlignment (memory.getData(), sizeof (SampleType));
79 order = newOrder;
80 }
81
82 for (size_t i = 0; i < order; ++i)
83 state[i] = resetToValue;
84}

89template <typename SampleType>
90template <typename ProcessContext, bool bypassed>
91void Filter<SampleType>::processInternal (const ProcessContext& context) noexcept
92{
93 static_assert (std::is_same<typename ProcessContext::SampleType, SampleType>::value,
94 "The sample-type of the IIR filter must match the sample-type supplied to this process callback");
95 check();
96
97 auto&& inputBlock = context.getInputBlock();
98 auto&& outputBlock = context.getOutputBlock();
99
100 // This class can only process mono signals. Use the ProcessorDuplicator class
101 // to apply this filter on a multi-channel audio stream.
102 jassert (inputBlock.getNumChannels() == 1);
103 jassert (outputBlock.getNumChannels() == 1);
104
105 auto numSamples = inputBlock.getNumSamples();
106 auto* src = inputBlock .getChannelPointer (0);
107 auto* dst = outputBlock.getChannelPointer (0);
108 auto* coeffs = coefficients->getRawCoefficients();
109
110 switch (order)
111 {
112 case 1:
113 {
114 auto b0 = coeffs[0];
115 auto b1 = coeffs[1];
116 auto a1 = coeffs[2];
117
118 auto lv1 = state[0];
119
120 for (size_t i = 0; i < numSamples; ++i)
121 {
122 auto input = src[i];
123 auto output = input * b0 + lv1;
124
125 dst[i] = bypassed ? input : output;
126
127 lv1 = (input * b1) - (output * a1);
128 }
129
130 util::snapToZero (lv1); state[0] = lv1;
131 }
132 break;
133
134 case 2:
135 {
136 auto b0 = coeffs[0];
137 auto b1 = coeffs[1];
138 auto b2 = coeffs[2];
139 auto a1 = coeffs[3];
140 auto a2 = coeffs[4];
141
142 auto lv1 = state[0];
143 auto lv2 = state[1];
144
145 for (size_t i = 0; i < numSamples; ++i)
146 {
147 auto input = src[i];
148 auto output = (input * b0) + lv1;
149 dst[i] = bypassed ? input : output;
150
151 lv1 = (input * b1) - (output* a1) + lv2;
152 lv2 = (input * b2) - (output* a2);
153 }
154
155 util::snapToZero (lv1); state[0] = lv1;
156 util::snapToZero (lv2); state[1] = lv2;
157 }
158 break;
159
160 case 3:
161 {
162 auto b0 = coeffs[0];
163 auto b1 = coeffs[1];
164 auto b2 = coeffs[2];
165 auto b3 = coeffs[3];
166 auto a1 = coeffs[4];
167 auto a2 = coeffs[5];
168 auto a3 = coeffs[6];
169
170 auto lv1 = state[0];
171 auto lv2 = state[1];
172 auto lv3 = state[2];
173
174 for (size_t i = 0; i < numSamples; ++i)
175 {
176 auto input = src[i];
177 auto output = (input * b0) + lv1;
178 dst[i] = bypassed ? input : output;
179
180 lv1 = (input * b1) - (output* a1) + lv2;
181 lv2 = (input * b2) - (output* a2) + lv3;
182 lv3 = (input * b3) - (output* a3);
183 }
184
185 util::snapToZero (lv1); state[0] = lv1;
186 util::snapToZero (lv2); state[1] = lv2;
187 util::snapToZero (lv3); state[2] = lv3;
188 }
189 break;
190
191 default:
192 {
193 for (size_t i = 0; i < numSamples; ++i)
194 {
195 auto input = src[i];
196 auto output= (input * coeffs[0]) + state[0];
197 dst[i] = bypassed ? input : output;
198
199 for (size_t j = 0; j < order - 1; ++j)
200 state[j] = (input * coeffs[j + 1]) - (output* coeffs[order + j + 1]) + state[j + 1];
201
202 state[order - 1] = (input * coeffs[order]) - (output* coeffs[order * 2]);
203 }
204
205 snapToZero();
206 }
207 }
208}

210template <typename SampleType>
211SampleType JUCE_VECTOR_CALLTYPE Filter<SampleType>::processSample (SampleType sample) noexcept
212{
213 check();
214 auto* c = coefficients->getRawCoefficients();
215
216 auto output = (c[0] * sample) + state[0];
217
218 for (size_t j = 0; j < order - 1; ++j)
219 state[j] = (c[j + 1] * sample) - (c[order + j + 1] * output) + state[j + 1];
220
221 state[order - 1] = (c[order] * sample) - (c[order * 2] * output);
222
223 return output;
224}
