224inline void b2Contact::GetWorldManifold(b2WorldManifold* worldManifold) const
225{
226 const b2Body* bodyA = m_fixtureA->GetBody();
227 const b2Body* bodyB = m_fixtureB->GetBody();
228 const b2Shape* shapeA = m_fixtureA->GetShape();
229 const b2Shape* shapeB = m_fixtureB->GetShape();
230
231 worldManifold->Initialize(&m_manifold, bodyA->GetTransform(), shapeA->m_radius, bodyB->GetTransform(), shapeB->m_radius);
232}

234inline void b2Contact::SetEnabled(bool flag)
235{
