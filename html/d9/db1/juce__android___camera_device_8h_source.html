702 auto* env = getEnv();

709
710 const int size = env->CallIntMethod (keysList, JavaList.size);
711
712 JUCE_CAMERA_LOG ("Camera id: " + cameraId + ", characteristics keys num: " + String (size));

716 auto key = LocalRef<jobject> (env->CallObjectMethod (keysList, JavaList.get, i));

725

727 {

731 {
732 printRangeArrayElements (keyValue, keyName);

735 {

737

739 {
740 JUCE_CAMERA_LOG ("Key: " + keyName);

742 for (int j = 0, k = 1; j < keyValueString.length(); j += chunkSize, ++k)
743 JUCE_CAMERA_LOG ("value part " + String (k) + ": " + keyValueString.substring (j, k + chunkSize));
744 }
745 else
746 {
747 JUCE_CAMERA_LOG ("Key: " + keyName + ", value: " + keyValueString);
748 }
749 }

1292 static int getRotationAngle (Desktop::DisplayOrientation deviceOrientationFromAccelerometerSensor,
1293 Desktop::DisplayOrientation targetOrientation,
1294 bool cameraLensFrontFacing,
1295 int cameraSensorOrientation)
1296 {
1297 auto isSensorOrientationHorizontal = deviceOrientationFromAccelerometerSensor == Desktop::rotatedAntiClockwise
1298 || deviceOrientationFromAccelerometerSensor == Desktop::rotatedClockwise;
1299
1300 if (cameraLensFrontFacing && isSensorOrientationHorizontal)
1301 {
1302 // flip angles for front camera
1303 return getRotationAngle (deviceOrientationFromAccelerometerSensor, targetOrientation, false, (cameraSensorOrientation + 180) % 360);
1304 }
1305
1306 switch (targetOrientation)
1307 {
1308 case Desktop::rotatedAntiClockwise:
1309 return cameraSensorOrientation == 90 ? 0 : 180;
1310 case Desktop::rotatedClockwise:
1311 return cameraSensorOrientation == 90 ? 180 : 0;
1312 case Desktop::upright:
1313 case Desktop::upsideDown:
1314 if ((targetOrientation == Desktop::upright && ! cameraLensFrontFacing)
1315 || (targetOrientation == Desktop::upsideDown && cameraLensFrontFacing))
1316 {
1317 return cameraSensorOrientation;
1318 }
1319 else
1320 {
1321 if (deviceOrientationFromAccelerometerSensor == Desktop::upright || deviceOrientationFromAccelerometerSensor == Desktop::upsideDown)
1322 return cameraSensorOrientation;
1323 else
1324 return (cameraSensorOrientation + 180) % 360;
1325 }
1326 break;
1327 case Desktop::allOrientations:
1328 default:
1329 return 0;
1330 }

1781 void updateState (jobject captureResult)
1782 {
1783 // IllegalStateException can be thrown when accessing CaptureSession,
1784 // claiming that capture session was already closed but we may not
1785 // get relevant callback yet, so check for this and bailout when needed.
1786 if (jniCheckHasExceptionOccurredAndClear())
1787 return;
1788
1789 switch (currentState)
1790 {
1791 case State::pendingFocusLock:
1792 {
1793 JUCE_CAMERA_LOG ("Still picture capture, updateState(), State::pendingFocusLock...");
1794
1795 auto controlAfStateValue = getCaptureResultIntegerKeyValue (CaptureResult.CONTROL_AF_STATE, captureResult);
1796
1797 if (controlAfStateValue.get() == nullptr)
1798 {
1799 captureStillPictureDelayed();
1800 return;
1801 }
1802
1803 auto autoToFocusNotAvailable = autoFocusMode == 0;
1804
1805 if (autoToFocusNotAvailable || autoFocusHasFinished (controlAfStateValue))
1806 {
1807 auto controlAeStateIntValue = getControlAEState (captureResult);
1808 static constexpr int controlAeStateConverged = 2;
1809
1810 if (controlAeStateIntValue == -1 || controlAeStateIntValue == controlAeStateConverged)
1811 {
1812 currentState = State::pictureTaken;
1813 captureStillPictureDelayed();
1814 }
1815 else
1816 {
1817 runPrecaptureSequence();
1818 }
1819 }
1820
1821 break;
1822 }
1823
1824 case State::pendingExposurePrecapture:
1825 {
1826 JUCE_CAMERA_LOG ("Still picture capture, updateState(), State::pendingExposurePrecapture...");
1827
1828 auto controlAeStateIntValue = getControlAEState (captureResult);
1829 static constexpr int controlAeStateFlashRequired = 4;
1830 static constexpr int controlAeStatePrecapture = 5;
1831
1832 if (controlAeStateIntValue == -1 || controlAeStateIntValue == controlAeStateFlashRequired
1833 || controlAeStateIntValue == controlAeStatePrecapture)
1834 {
1835 currentState = State::pendingExposurePostPrecapture;
1836 }
1837
1838 break;
1839 }
1840
1841 case State::pendingExposurePostPrecapture:
1842 {
1843 JUCE_CAMERA_LOG ("Still picture capture, updateState(), State::pendingExposurePostPrecapture...");
1844
1845 auto controlAeStateIntValue = getControlAEState (captureResult);
1846 static constexpr int controlAeStatePrecapture = 5;
1847
1848 if (controlAeStateIntValue == -1 || controlAeStateIntValue != controlAeStatePrecapture)
1849 {
1850 currentState = State::pictureTaken;
1851 captureStillPictureDelayed();
1852 }
1853
1854 break;
1855 }
1856 case State::idle:
1857 case State::pictureTaken:
1858 { /* do nothing */ break; }
1859 };
