38class CarbonViewWrapperComponent : public Component,
39 public ComponentMovementWatcher,

42public:

44 : ComponentMovementWatcher (this),

54 ~CarbonViewWrapperComponent()
55 {
56 jassert (embeddedView == 0); // must call deleteWindow() in the subclass's destructor!

59 virtual HIViewRef attachView (WindowRef windowRef, HIViewRef rootView) = 0;
60 virtual void removeView (HIViewRef embeddedView) = 0;
61 virtual void handleMouseDown (int, int) {}
62 virtual void handlePaint() {}
63
64 virtual bool getEmbeddedViewSize (int& w, int& h)
65 {
66 if (embeddedView == 0)
67 return false;
68
69 HIRect bounds;
70 HIViewGetBounds (embeddedView, &bounds);
71 w = jmax (1, roundToInt (bounds.size.width));
72 h = jmax (1, roundToInt (bounds.size.height));
73 return true;
74 }
75
76 void createWindow()
77 {
78 if (wrapperWindow == nil)
79 {
80 Rect r;
81 r.left = (short) getScreenX();
82 r.top = (short) getScreenY();
83 r.right = (short) (r.left + getWidth());
84 r.bottom = (short) (r.top + getHeight());
85
86 CreateNewWindow (kDocumentWindowClass,
87 (WindowAttributes) (kWindowStandardHandlerAttribute | kWindowCompositingAttribute
88 | kWindowNoShadowAttribute | kWindowNoTitleBarAttribute),
89 &r, &wrapperWindow);
90
91 jassert (wrapperWindow != 0);
92 if (wrapperWindow == 0)
93 return;
94
95 carbonWindow = [[NSWindow alloc] initWithWindowRef: wrapperWindow];
96
97 [getOwnerWindow() addChildWindow: carbonWindow
98 ordered: NSWindowAbove];
99
100 embeddedView = attachView (wrapperWindow, HIViewGetRoot (wrapperWindow));
101
102 // Check for the plugin creating its own floating window, and if there is one,
103 // we need to reparent it to make it visible..
104 if (carbonWindow.childWindows.count > 0)
105 if (NSWindow* floatingChildWindow = [[carbonWindow childWindows] objectAtIndex: 0])
106 [getOwnerWindow() addChildWindow: floatingChildWindow
107 ordered: NSWindowAbove];
108
109 EventTypeSpec windowEventTypes[] =
110 {
111 { kEventClassWindow, kEventWindowGetClickActivation },
112 { kEventClassWindow, kEventWindowHandleDeactivate },
113 { kEventClassWindow, kEventWindowBoundsChanging },
114 { kEventClassMouse, kEventMouseDown },
115 { kEventClassMouse, kEventMouseMoved },
116 { kEventClassMouse, kEventMouseDragged },
117 { kEventClassMouse, kEventMouseUp },
118 { kEventClassWindow, kEventWindowDrawContent },
119 { kEventClassWindow, kEventWindowShown },
120 { kEventClassWindow, kEventWindowHidden }
121 };
122
123 EventHandlerUPP upp = NewEventHandlerUPP (carbonEventCallback);
124 InstallWindowEventHandler (wrapperWindow, upp,
125 sizeof (windowEventTypes) / sizeof (EventTypeSpec),
126 windowEventTypes, this, &eventHandlerRef);
127
128 setOurSizeToEmbeddedViewSize();
129 setEmbeddedWindowToOurSize();
130
131 creationTime = Time::getCurrentTime();
132 }
133 }
134
135 void deleteWindow()
136 {
137 removeView (embeddedView);
138 embeddedView = 0;
139
140 if (wrapperWindow != nil)
141 {
142 NSWindow* ownerWindow = getOwnerWindow();
143
144 if ([[ownerWindow childWindows] count] > 0)
145 {
146 [ownerWindow removeChildWindow: carbonWindow];
147 [carbonWindow close];
148 }
149
150 RemoveEventHandler (eventHandlerRef);
151 DisposeWindow (wrapperWindow);
152 wrapperWindow = nil;
153 }
154 }

156 //==============================================================================
157 void setOurSizeToEmbeddedViewSize()
158 {
159 int w, h;
160 if (getEmbeddedViewSize (w, h))
161 {
162 if (w != getWidth() || h != getHeight())
163 {
164 startTimer (50);
165 setSize (w, h);
166
167 if (Component* p = getParentComponent())
168 p->setSize (w, h);
169 }
170 else
171 {
172 startTimer (jlimit (50, 500, getTimerInterval() + 20));
173 }
174 }
175 else
176 {
177 stopTimer();
178 }
179 }
180
181 void setEmbeddedWindowToOurSize()
182 {
183 if (! recursiveResize)
184 {
185 recursiveResize = true;
186
187 if (embeddedView != 0)
188 {
189 HIRect r;
190 r.origin.x = 0;
191 r.origin.y = 0;
192 r.size.width = (float) getWidth();
193 r.size.height = (float) getHeight();
194 HIViewSetFrame (embeddedView, &r);
195 }
196
197 if (wrapperWindow != nil)
198 {
199 jassert (getTopLevelComponent()->getDesktopScaleFactor() == 1.0f);
200 Rectangle<int> screenBounds (getScreenBounds() * Desktop::getInstance().getGlobalScaleFactor());
201
202 Rect wr;
203 wr.left = (short) screenBounds.getX();
204 wr.top = (short) screenBounds.getY();
205 wr.right = (short) screenBounds.getRight();
206 wr.bottom = (short) screenBounds.getBottom();
207
208 SetWindowBounds (wrapperWindow, kWindowContentRgn, &wr);
209
210 // This group stuff is mainly a workaround for Mackie plugins like FinalMix..
211 WindowGroupRef group = GetWindowGroup (wrapperWindow);
212 WindowRef attachedWindow;
213
214 if (GetIndexedWindow (group, 2, kWindowGroupContentsReturnWindows, &attachedWindow) == noErr)
215 {
216 SelectWindow (attachedWindow);
217 ActivateWindow (attachedWindow, TRUE);
218 HideWindow (wrapperWindow);
219 }
220
221 ShowWindow (wrapperWindow);
222 }
223
224 recursiveResize = false;
225 }
226 }

258 static void recursiveHIViewRepaint (HIViewRef view)
259 {
260 HIViewSetNeedsDisplay (view, true);
261 HIViewRef child = HIViewGetFirstSubview (view);
262
263 while (child != 0)
264 {
265 recursiveHIViewRepaint (child);
266 child = HIViewGetNextView (child);
267 }
268 }
269
270 void timerCallback() override
271 {
272 if (isShowing())
273 {
274 setOurSizeToEmbeddedViewSize();
275
276 // To avoid strange overpainting problems when the UI is first opened, we'll
277 // repaint it a few times during the first second that it's on-screen..
278 if (repaintChildOnCreation && (Time::getCurrentTime() - creationTime).inMilliseconds() < 1000)
279 recursiveHIViewRepaint (HIViewGetRoot (wrapperWindow));
280 }
281 }

283 void setRepaintsChildHIViewWhenCreated (bool b) noexcept

286 }

288 OSStatus carbonEventHandler (EventHandlerCallRef /*nextHandlerRef*/, EventRef event)
289 {
290 switch (GetEventKind (event))
291 {
292 case kEventWindowHandleDeactivate:
293 ActivateWindow (wrapperWindow, TRUE);
294 return noErr;
295
296 case kEventWindowGetClickActivation:
297 {
298 getTopLevelComponent()->toFront (false);
299 [carbonWindow makeKeyAndOrderFront: nil];
300
301 ClickActivationResult howToHandleClick = kActivateAndHandleClick;
302
303 SetEventParameter (event, kEventParamClickActivation, typeClickActivationResult,
304 sizeof (ClickActivationResult), &howToHandleClick);
305
306 if (embeddedView != 0)
307 HIViewSetNeedsDisplay (embeddedView, true);
308
309 return noErr;
310 }
311 }
312
313 return eventNotHandledErr;
314 }
315
316 static pascal OSStatus carbonEventCallback (EventHandlerCallRef nextHandlerRef, EventRef event, void* userData)
317 {
318 return ((CarbonViewWrapperComponent*) userData)->carbonEventHandler (nextHandlerRef, event);
319 }
