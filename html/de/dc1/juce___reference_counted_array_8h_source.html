566 template <class ElementComparator>
567 int indexOfSorted (ElementComparator& comparator,
568 const ObjectClass* objectToLookFor) const noexcept
569 {
570 ignoreUnused (comparator);
571 const ScopedLockType lock (getLock());
572 int s = 0, e = values.size();
573
574 while (s < e)
575 {
576 if (comparator.compareElements (objectToLookFor, values[s]) == 0)
577 return s;
578
579 auto halfway = (s + e) / 2;
580
581 if (halfway == s)
582 break;
583
584 if (comparator.compareElements (objectToLookFor, values[halfway]) >= 0)
585 s = halfway;
586 else
587 e = halfway;
588 }
589
590 return -1;
591 }

689 void removeRange (int startIndex,
690 int numberToRemove)
691 {
692 const ScopedLockType lock (getLock());
693 startIndex = jlimit (0, values.size(), startIndex);
694 auto endIndex = jlimit (0, values.size(), startIndex + numberToRemove);
695 numberToRemove = endIndex - startIndex;
696
697 if (numberToRemove > 0)
698 {
699 Array<ObjectClass*> objectsToRemove;
700 objectsToRemove.addArray (values.begin() + startIndex, numberToRemove);
701
702 values.removeElements (startIndex, numberToRemove);
703
704 for (auto& o : objectsToRemove)
705 releaseObject (o);
706
707 if ((values.size() << 1) < values.capacity())
708 minimiseStorageOverheads();
709 }
