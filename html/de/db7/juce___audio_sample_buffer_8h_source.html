415 void setSize (int newNumChannels,
416 int newNumSamples,
417 bool keepExistingContent = false,
418 bool clearExtraSpace = false,
419 bool avoidReallocating = false)
420 {
421 jassert (newNumChannels >= 0);
422 jassert (newNumSamples >= 0);
423
424 if (newNumSamples != size || newNumChannels != numChannels)
425 {
426 auto allocatedSamplesPerChannel = ((size_t) newNumSamples + 3) & ~3u;
427 auto channelListSize = ((static_cast<size_t> (1 + newNumChannels) * sizeof (Type*)) + 15) & ~15u;
428 auto newTotalBytes = ((size_t) newNumChannels * (size_t) allocatedSamplesPerChannel * sizeof (Type))
429 + channelListSize + 32;
430
431 if (keepExistingContent)
432 {
433 if (avoidReallocating && newNumChannels <= numChannels && newNumSamples <= size)
434 {
435 // no need to do any remapping in this case, as the channel pointers will remain correct!
436 }
437 else
438 {
439 HeapBlock<char, true> newData;
440 newData.allocate (newTotalBytes, clearExtraSpace || isClear);
441
442 auto numSamplesToCopy = (size_t) jmin (newNumSamples, size);
443
444 auto newChannels = unalignedPointerCast<Type**> (newData.get());
445 auto newChan = unalignedPointerCast<Type*> (newData + channelListSize);
446
447 for (int j = 0; j < newNumChannels; ++j)
448 {
449 newChannels[j] = newChan;
450 newChan += allocatedSamplesPerChannel;
451 }
452
453 if (! isClear)
454 {
455 auto numChansToCopy = jmin (numChannels, newNumChannels);
456
457 for (int i = 0; i < numChansToCopy; ++i)
458 FloatVectorOperations::copy (newChannels[i], channels[i], (int) numSamplesToCopy);
459 }
460
461 allocatedData.swapWith (newData);
462 allocatedBytes = newTotalBytes;
463 channels = newChannels;
464 }
465 }
466 else
467 {
468 if (avoidReallocating && allocatedBytes >= newTotalBytes)
469 {
470 if (clearExtraSpace || isClear)
471 allocatedData.clear (newTotalBytes);
472 }
473 else
474 {
475 allocatedBytes = newTotalBytes;
476 allocatedData.allocate (newTotalBytes, clearExtraSpace || isClear);
477 channels = unalignedPointerCast<Type**> (allocatedData.get());
478 }
479
480 auto* chan = unalignedPointerCast<Type*> (allocatedData + channelListSize);
481
482 for (int i = 0; i < newNumChannels; ++i)
483 {
484 channels[i] = chan;
485 chan += allocatedSamplesPerChannel;
486 }
487 }
488
489 channels[newNumChannels] = nullptr;
490 size = newNumSamples;
491 numChannels = newNumChannels;
492 }
493 }

1176 {
1177 jassert (isPositiveAndBelow (channel, numChannels));

1226 void allocateData()
1227 {
1228 #if (! JUCE_GCC || (__GNUC__ * 100 + __GNUC_MINOR__) >= 409)
1229 static_assert (alignof (Type) <= detail::maxAlignment,
1230 "AudioBuffer cannot hold types with alignment requirements larger than that guaranteed by malloc");
1231 #endif
1232 jassert (size >= 0);
1233
1234 auto channelListSize = (size_t) (numChannels + 1) * sizeof (Type*);
1235 auto requiredSampleAlignment = std::alignment_of<Type>::value;
1236 size_t alignmentOverflow = channelListSize % requiredSampleAlignment;
1237
1238 if (alignmentOverflow != 0)
1239 channelListSize += requiredSampleAlignment - alignmentOverflow;
1240
1241 allocatedBytes = (size_t) numChannels * (size_t) size * sizeof (Type) + channelListSize + 32;
1242 allocatedData.malloc (allocatedBytes);
1243 channels = unalignedPointerCast<Type**> (allocatedData.get());
1244 auto chan = unalignedPointerCast<Type*> (allocatedData + channelListSize);
1245
