41 void reset() { currentPacketLen = 0; }

49 template <typename PacketCallbackFunction>
50 void dispatch (const uint32_t* begin,
51 const uint32_t* end,
52 double timeStamp,
53 PacketCallbackFunction&& callback)
54 {
55 std::for_each (begin, end, [&] (uint32_t word)
56 {
57 nextPacket[currentPacketLen++] = word;
58
59 if (currentPacketLen == Utils::getNumWordsForMessageType (nextPacket.front()))
60 {
61 callback (View (nextPacket.data()), timeStamp);
62 currentPacketLen = 0;
63 }
64 });
65 }
66
67private:
68 std::array<uint32_t, 4> nextPacket;
69 size_t currentPacketLen = 0;
70};
