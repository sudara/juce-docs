93 void reset()
94 {
95 if (coefficients != nullptr)
96 {
97 auto newSize = coefficients->getFilterOrder() + 1;
98
99 if (newSize != size)
100 {
101 memory.malloc (1 + jmax (newSize, size, static_cast<size_t> (128)));
102
103 fifo = snapPointerToAlignment (memory.getData(), sizeof (SampleType));
104 size = newSize;
105 }
106
107 for (size_t i = 0; i < size; ++i)
108 fifo[i] = SampleType {0};
109 }

123 template <typename ProcessContext>
124 void process (const ProcessContext& context) noexcept
125 {
126 static_assert (std::is_same<typename ProcessContext::SampleType, SampleType>::value,
127 "The sample-type of the FIR filter must match the sample-type supplied to this process callback");
128 check();
129
130 auto&& inputBlock = context.getInputBlock();
131 auto&& outputBlock = context.getOutputBlock();
132
133 // This class can only process mono signals. Use the ProcessorDuplicator class
134 // to apply this filter on a multi-channel audio stream.
135 jassert (inputBlock.getNumChannels() == 1);
136 jassert (outputBlock.getNumChannels() == 1);
137
138 auto numSamples = inputBlock.getNumSamples();
139 auto* src = inputBlock .getChannelPointer (0);
140 auto* dst = outputBlock.getChannelPointer (0);
141
142 auto* fir = coefficients->getRawCoefficients();
143 size_t p = pos;
144
145 if (context.isBypassed)
146 {
147 for (size_t i = 0; i < numSamples; ++i)
148 {
149 fifo[p] = dst[i] = src[i];
150 p = (p == 0 ? size - 1 : p - 1);
151 }
152 }
153 else
154 {
155 for (size_t i = 0; i < numSamples; ++i)
156 dst[i] = processSingleSample (src[i], fifo, fir, size, p);
157 }
158
159 pos = p;
160 }

166 SampleType JUCE_VECTOR_CALLTYPE processSample (SampleType sample) noexcept
167 {
168 check();
169 return processSingleSample (sample, fifo, coefficients->getRawCoefficients(), size, pos);
170 }

172 private:
173 //==============================================================================
174 HeapBlock<SampleType> memory;
175 SampleType* fifo = nullptr;
176 size_t pos = 0, size = 0;

178 //==============================================================================
179 void check()
180 {
181 jassert (coefficients != nullptr);
182
183 if (size != (coefficients->getFilterOrder() + 1))
184 reset();

187 static SampleType JUCE_VECTOR_CALLTYPE processSingleSample (SampleType sample, SampleType* buf,
188 const NumericType* fir, size_t m, size_t& p) noexcept
189 {
190 SampleType out (0);
191
192 buf[p] = sample;
193
194 size_t k;
195 for (k = 0; k < m - p; ++k)
196 out += buf[(p + k)] * fir[k];
197
198 for (size_t j = 0; j < p; ++j)
199 out += buf[j] * fir[j + k];
200
201 p = (p == 0 ? m - 1 : p - 1);
202
203 return out;
204 }
