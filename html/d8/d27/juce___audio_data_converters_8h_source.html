459 template <class OtherPointerType>
460 void convertSamples (OtherPointerType source, int numSamples) const noexcept
461 {
462 // trying to write to a const pointer! For a writeable one, use AudioData::NonConst instead!
463 static_assert (Constness::isConst == 0, "Attempt to write to a const pointer");
464
465 Pointer dest (*this);
466
467 if (source.getRawData() != getRawData() || source.getNumBytesBetweenSamples() >= getNumBytesBetweenSamples())
468 {
469 while (--numSamples >= 0)
470 {
471 Endianness::copyFrom (dest.data, source);
472 dest.advance();
473 ++source;
474 }
475 }
476 else // copy backwards if we're increasing the sample width..
477 {
478 dest += numSamples;
479 source += numSamples;
480
481 while (--numSamples >= 0)
482 Endianness::copyFrom ((--dest).data, --source);
483 }

494 Range<float> findMinAndMax (size_t numSamples) const noexcept
495 {
496 if (numSamples == 0)
497 return Range<float>();
498
499 Pointer dest (*this);
500
501 if (isFloatingPoint())
502 {
503 float mn = dest.getAsFloat();
504 dest.advance();
505 float mx = mn;
506
507 while (--numSamples > 0)
508 {
509 const float v = dest.getAsFloat();
510 dest.advance();
511
512 if (mx < v) mx = v;
513 if (v < mn) mn = v;
514 }
515
516 return Range<float> (mn, mx);
517 }
518
519 int32 mn = dest.getAsInt32();
520 dest.advance();
521 int32 mx = mn;
522
523 while (--numSamples > 0)
524 {
525 const int v = dest.getAsInt32();
526 dest.advance();
527
528 if (mx < v) mx = v;
529 if (v < mn) mn = v;
530 }
531
532 return Range<float> ((float) mn * (float) (1.0 / (1.0 + (double) Int32::maxValue)),
533 (float) mx * (float) (1.0 / (1.0 + (double) Int32::maxValue)));
534 }

721 static void interleaveSamples (NonInterleavedSource<SourceFormat...> source,
722 InterleavedDest<DestFormat...> dest,
723 int numSamples)
724 {
725 using SourceType = typename decltype (source)::PointerType;
726 using DestType = typename decltype (dest) ::PointerType;
727
728 for (int i = 0; i < dest.channels; ++i)
729 {
730 const DestType destType (addBytesToPointer (dest.data, i * DestType::getBytesPerSample()), dest.channels);
731
732 if (i < source.channels)
733 {
734 if (*source.data != nullptr)
735 {
736 destType.convertSamples (SourceType { *source.data }, numSamples);
737 ++source.data;
738 }
739 }
740 else
741 {
742 destType.clearSamples (numSamples);
743 }
744 }
745 }

766 NonInterleavedDest<DestFormat...> dest,
767 int numSamples)
768 {
769 using SourceType = typename decltype (source)::PointerType;
770 using DestType = typename decltype (dest) ::PointerType;
771
772 for (int i = 0; i < dest.channels; ++i)
773 {
774 if (auto* targetChan = dest.data[i])
775 {
776 const DestType destType (targetChan);
777
778 if (i < source.channels)
779 destType.convertSamples (SourceType (addBytesToPointer (source.data, i * SourceType::getBytesPerSample()), source.channels), numSamples);
780 else
781 destType.clearSamples (numSamples);
782 }
