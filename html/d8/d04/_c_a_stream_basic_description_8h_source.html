120 CAStreamBasicDescription( double inSampleRate, UInt32 inNumChannels, CommonPCMFormat pcmf, bool inIsInterleaved) {
121 unsigned wordsize;
122
123 mSampleRate = inSampleRate;
124 mFormatID = kAudioFormatLinearPCM;
125 mFormatFlags = kAudioFormatFlagsNativeEndian | kAudioFormatFlagIsPacked;
126 mFramesPerPacket = 1;
127 mChannelsPerFrame = inNumChannels;
128 mBytesPerFrame = mBytesPerPacket = 0;
129 mReserved = 0;
130
131 switch (pcmf) {
132 default:
133 return;
134 case kPCMFormatFloat32:
135 wordsize = 4;
136 mFormatFlags |= kAudioFormatFlagIsFloat;
137 break;
138 case kPCMFormatFloat64:
139 wordsize = 8;
140 mFormatFlags |= kAudioFormatFlagIsFloat;
141 break;
142 case kPCMFormatInt16:
143 wordsize = 2;
144 mFormatFlags |= kAudioFormatFlagIsSignedInteger;
145 break;
146 case kPCMFormatFixed824:
147 wordsize = 4;
148 mFormatFlags |= kAudioFormatFlagIsSignedInteger | (24 << kLinearPCMFormatFlagsSampleFractionShift);
149 break;
150 }
151 mBitsPerChannel = wordsize * 8;
152 if (inIsInterleaved)
153 mBytesPerFrame = mBytesPerPacket = wordsize * inNumChannels;
154 else {
155 mFormatFlags |= kAudioFormatFlagIsNonInterleaved;
156 mBytesPerFrame = mBytesPerPacket = wordsize;
157 }
158 }

230 bool IdentifyCommonPCMFormat(CommonPCMFormat &outFormat, bool *outIsInterleaved=NULL) const
231 { // return true if it's a valid PCM format.
232
233 outFormat = kPCMFormatOther;
234 // trap out patently invalid formats.
235 if (mFormatID != kAudioFormatLinearPCM || mFramesPerPacket != 1 || mBytesPerFrame != mBytesPerPacket || mBitsPerChannel/8 > mBytesPerFrame || mChannelsPerFrame == 0)
236 return false;
237 bool interleaved = (mFormatFlags & kAudioFormatFlagIsNonInterleaved) == 0;
238 if (outIsInterleaved != NULL) *outIsInterleaved = interleaved;
239 unsigned wordsize = mBytesPerFrame;
240 if (interleaved) {
241 if (wordsize % mChannelsPerFrame != 0) return false;
242 wordsize /= mChannelsPerFrame;
243 }
244
245 if ((mFormatFlags & kAudioFormatFlagIsBigEndian) == kAudioFormatFlagsNativeEndian
246 && wordsize * 8 == mBitsPerChannel) {
247 // packed and native endian, good
248 if (mFormatFlags & kLinearPCMFormatFlagIsFloat) {
249 // float: reject nonsense bits
250 if (mFormatFlags & (kLinearPCMFormatFlagIsSignedInteger | kLinearPCMFormatFlagsSampleFractionMask))
251 return false;
252 if (wordsize == 4)
253 outFormat = kPCMFormatFloat32;
254 if (wordsize == 8)
255 outFormat = kPCMFormatFloat64;
256 } else if (mFormatFlags & kLinearPCMFormatFlagIsSignedInteger) {
257 // signed int
258 unsigned fracbits = (mFormatFlags & kLinearPCMFormatFlagsSampleFractionMask) >> kLinearPCMFormatFlagsSampleFractionShift;
259 if (wordsize == 4 && fracbits == 24)
260 outFormat = kPCMFormatFixed824;
261 else if (wordsize == 2 && fracbits == 0)
262 outFormat = kPCMFormatInt16;
263 }
264 }
265 return true;
266 }

289 void SetCanonical(UInt32 nChannels, bool interleaved)
290 // note: leaves sample rate untouched
291 {
292 mFormatID = kAudioFormatLinearPCM;
293 UInt32 sampleSize = SizeOf32(AudioSampleType);
294 mFormatFlags = kAudioFormatFlagsCanonical;
295 mBitsPerChannel = 8 * sampleSize;
296 mChannelsPerFrame = nChannels;
297 mFramesPerPacket = 1;
298 if (interleaved)
299 mBytesPerPacket = mBytesPerFrame = nChannels * sampleSize;
300 else {
301 mBytesPerPacket = mBytesPerFrame = sampleSize;
302 mFormatFlags |= kAudioFormatFlagIsNonInterleaved;
303 }
304 }
305
306 bool IsCanonical() const
307 {
308 if (mFormatID != kAudioFormatLinearPCM) return false;
309 UInt32 reqFormatFlags;
310 UInt32 flagsMask = (kLinearPCMFormatFlagIsFloat | kLinearPCMFormatFlagIsBigEndian | kLinearPCMFormatFlagIsSignedInteger | kLinearPCMFormatFlagIsPacked | kLinearPCMFormatFlagsSampleFractionMask);
311 bool interleaved = (mFormatFlags & kAudioFormatFlagIsNonInterleaved) == 0;
312 unsigned sampleSize = SizeOf32(AudioSampleType);
313 reqFormatFlags = kAudioFormatFlagsCanonical;
314 UInt32 reqFrameSize = interleaved ? (mChannelsPerFrame * sampleSize) : sampleSize;
315
316 return ((mFormatFlags & flagsMask) == reqFormatFlags
317 && mBitsPerChannel == 8 * sampleSize
318 && mFramesPerPacket == 1
319 && mBytesPerFrame == reqFrameSize
320 && mBytesPerPacket == reqFrameSize);
321 }

323 void SetAUCanonical(UInt32 nChannels, bool interleaved)
324 {
325 mFormatID = kAudioFormatLinearPCM;
326#if CA_PREFER_FIXED_POINT
327 mFormatFlags = kAudioFormatFlagsCanonical | (kAudioUnitSampleFractionBits << kLinearPCMFormatFlagsSampleFractionShift);
328#else
329 mFormatFlags = kAudioFormatFlagsCanonical;
330#endif
331 mChannelsPerFrame = nChannels;
332 mFramesPerPacket = 1;
333 mBitsPerChannel = 8 * SizeOf32(AudioUnitSampleType);
334 if (interleaved)
335 mBytesPerPacket = mBytesPerFrame = nChannels * SizeOf32(AudioUnitSampleType);
336 else {
337 mBytesPerPacket = mBytesPerFrame = SizeOf32(AudioUnitSampleType);
338 mFormatFlags |= kAudioFormatFlagIsNonInterleaved;
339 }
340 }

342 void ChangeNumberChannels(UInt32 nChannels, bool interleaved)
343 // alter an existing format
344 {
345 Assert(IsPCM(), "ChangeNumberChannels only works for PCM formats");
346 UInt32 wordSize = SampleWordSize(); // get this before changing ANYTHING
347 if (wordSize == 0)
348 wordSize = (mBitsPerChannel + 7) / 8;
349 mChannelsPerFrame = nChannels;
350 mFramesPerPacket = 1;
351 if (interleaved) {
352 mBytesPerPacket = mBytesPerFrame = nChannels * wordSize;
353 mFormatFlags &= ~static_cast<UInt32>(kAudioFormatFlagIsNonInterleaved);
354 } else {
355 mBytesPerPacket = mBytesPerFrame = wordSize;
356 mFormatFlags |= kAudioFormatFlagIsNonInterleaved;
357 }
358 }
