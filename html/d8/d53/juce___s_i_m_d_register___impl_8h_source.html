34template <typename Type>
35struct SIMDRegister<Type>::ElementAccess
36{
37 ElementAccess (const ElementAccess&) = default;
38 operator Type() const { return simd.get (idx); }
39 ElementAccess& operator= (Type scalar) noexcept { simd.set (idx, scalar); return *this; }
40 ElementAccess& operator= (const ElementAccess& o) noexcept { return operator= ((Type) o); }
41
42private:
43 friend struct SIMDRegister;
44 ElementAccess (SIMDRegister& owner, size_t index) noexcept : simd (owner), idx (index) {}
45 SIMDRegister& simd;
46 size_t idx;
47};

101// The pure complex version
102template <typename Scalar>
103struct CmplxSIMDOps<std::complex<Scalar>>
104{
105 using vSIMDType = typename SIMDNativeOps<Scalar>::vSIMDType;
106
107 static vSIMDType JUCE_VECTOR_CALLTYPE load (const std::complex<Scalar>* a) noexcept
108 {
109 return SIMDNativeOps<Scalar>::load (reinterpret_cast<const Scalar*> (a));
110 }
111
112 static void JUCE_VECTOR_CALLTYPE store (vSIMDType value, std::complex<Scalar>* dest) noexcept
113 {
114 SIMDNativeOps<Scalar>::store (value, reinterpret_cast<Scalar*> (dest));
115 }
116
117 static vSIMDType JUCE_VECTOR_CALLTYPE expand (std::complex<Scalar> s) noexcept
118 {
119 const int n = sizeof (vSIMDType) / sizeof (Scalar);
120
121 union
122 {
123 vSIMDType v;
124 Scalar floats[(size_t) n];
125 } u;
126
127 for (int i = 0; i < n; ++i)
128 u.floats[i] = (i & 1) == 0 ? s.real() : s.imag();
129
130 return u.v;
131 }
132
133 static std::complex<Scalar> JUCE_VECTOR_CALLTYPE get (vSIMDType v, std::size_t i) noexcept
134 {
135 auto j = i << 1;
136 return std::complex<Scalar> (SIMDNativeOps<Scalar>::get (v, j), SIMDNativeOps<Scalar>::get (v, j + 1));
137 }
138
139 static vSIMDType JUCE_VECTOR_CALLTYPE set (vSIMDType v, std::size_t i, std::complex<Scalar> s) noexcept
140 {
141 auto j = i << 1;
142 return SIMDNativeOps<Scalar>::set (SIMDNativeOps<Scalar>::set (v, j, s.real()), j + 1, s.imag());
143 }
144
145 static std::complex<Scalar> JUCE_VECTOR_CALLTYPE sum (vSIMDType a) noexcept
146 {
147 vSIMDType result = SIMDNativeOps<Scalar>::oddevensum (a);
148 auto* ptr = reinterpret_cast<const Scalar*> (&result);
149 return std::complex<Scalar> (ptr[0], ptr[1]);
150 }
151
152 static vSIMDType JUCE_VECTOR_CALLTYPE mul (vSIMDType a, vSIMDType b) noexcept
153 {
154 return SIMDNativeOps<Scalar>::cmplxmul (a, b);
155 }
156
157 static vSIMDType JUCE_VECTOR_CALLTYPE muladd (vSIMDType a, vSIMDType b, vSIMDType c) noexcept
158 {
159 return SIMDNativeOps<Scalar>::add (a, SIMDNativeOps<Scalar>::cmplxmul (b, c));
160 }
161};
