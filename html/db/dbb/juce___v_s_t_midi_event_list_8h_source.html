41class VSTMidiEventList
42{
43 // "events" is expected to be a const- or non-const-ref to Vst2::VstEvents.
44 template <typename Events>
45 static auto& getEvent (Events& events, int index)
46 {
47 using EventType = decltype (&*events.events);
48
49 // We static cast rather than using a direct array index here to circumvent
50 // UB sanitizer's bounds-checks. The original struct is supposed to contain
51 // a variable-length array, but the declaration uses a size of "2" for this
52 // member.
53 return static_cast<EventType> (events.events)[index];
54 }
55
56 Vst2::VstEvent* const& getEvent (int index) const { return getEvent (*events, index); }
57 Vst2::VstEvent* & getEvent (int index) { return getEvent (*events, index); }
58
59public:
60 //==============================================================================
61 VSTMidiEventList()
62 : numEventsUsed (0), numEventsAllocated (0)
63 {
64 }

66 ~VSTMidiEventList()
67 {
68 freeEvents();

72 void clear()
73 {
74 numEventsUsed = 0;
75
76 if (events != nullptr)
77 events->numEvents = 0;
78 }

80 void addEvent (const void* const midiData, int numBytes, int frameOffset)
81 {
82 ensureSize (numEventsUsed + 1);
83
84 void* const ptr = getEvent (numEventsUsed);
85 events->numEvents = ++numEventsUsed;
86
87 if (numBytes <= 4)
88 {
89 auto* const e = static_cast<Vst2::VstMidiEvent*> (ptr);
90
91 if (e->type == Vst2::kVstSysExType)
92 {
93 delete[] reinterpret_cast<Vst2::VstMidiSysexEvent*> (e)->sysexDump;
94 e->type = Vst2::kVstMidiType;
95 e->byteSize = sizeof (Vst2::VstMidiEvent);
96 e->noteLength = 0;
97 e->noteOffset = 0;
98 e->detune = 0;
99 e->noteOffVelocity = 0;
100 }
101
102 e->deltaFrames = frameOffset;
103 memcpy (e->midiData, midiData, (size_t) numBytes);
104 }
105 else
106 {
107 auto* const se = static_cast<Vst2::VstMidiSysexEvent*> (ptr);
108
109 if (se->type == Vst2::kVstSysExType)
110 delete[] se->sysexDump;
111
112 se->sysexDump = new char [(size_t) numBytes];
113 memcpy (se->sysexDump, midiData, (size_t) numBytes);
114
115 se->type = Vst2::kVstSysExType;
116 se->byteSize = sizeof (Vst2::VstMidiSysexEvent);
117 se->deltaFrames = frameOffset;
118 se->flags = 0;
119 se->dumpBytes = numBytes;
120 se->resvd1 = 0;
121 se->resvd2 = 0;
122 }
123 }

126 // Handy method to pull the events out of an event buffer supplied by the host
127 // or plugin.
128 static void addEventsToMidiBuffer (const Vst2::VstEvents* events, MidiBuffer& dest)
129 {
130 for (int i = 0; i < events->numEvents; ++i)
131 {
132 const auto* const e = getEvent (*events, i);
133
134 if (e != nullptr)
135 {
136 const void* const ptr = e;
137
138 if (e->type == Vst2::kVstMidiType)
139 {
140 dest.addEvent ((const juce::uint8*) static_cast<const Vst2::VstMidiEvent*> (ptr)->midiData,
141 4, e->deltaFrames);
142 }
143 else if (e->type == Vst2::kVstSysExType)
144 {
145 const auto* se = static_cast<const Vst2::VstMidiSysexEvent*> (ptr);
146 dest.addEvent ((const juce::uint8*) se->sysexDump,
147 (int) se->dumpBytes,
148 e->deltaFrames);
149 }
150 }
151 }
152 }

154 //==============================================================================
155 void ensureSize (int numEventsNeeded)
156 {
157 if (numEventsNeeded > numEventsAllocated)
158 {
159 numEventsNeeded = (numEventsNeeded + 32) & ~31;
160
161 const size_t size = 20 + (size_t) numEventsNeeded * sizeof (Vst2::VstEvent*);
162
163 if (events == nullptr)
164 events.calloc (size, 1);
165 else
166 events.realloc (size, 1);
167
168 for (int i = numEventsAllocated; i < numEventsNeeded; ++i)
169 getEvent (i) = allocateVSTEvent();
170
171 numEventsAllocated = numEventsNeeded;
172 }
173 }
174
175 void freeEvents()
176 {
177 if (events != nullptr)
178 {
179 for (int i = numEventsAllocated; --i >= 0;)
180 freeVSTEvent (getEvent (i));
181
182 events.free();
183 numEventsUsed = 0;
184 numEventsAllocated = 0;
185 }
186 }

191private:
192 int numEventsUsed, numEventsAllocated;

194 static Vst2::VstEvent* allocateVSTEvent()
195 {
196 constexpr auto size = jmax (sizeof (Vst2::VstMidiEvent), sizeof (Vst2::VstMidiSysexEvent));
197
198 if (auto* e = static_cast<Vst2::VstEvent*> (std::calloc (1, size)))
199 {
200 e->type = Vst2::kVstMidiType;
201 e->byteSize = sizeof (Vst2::VstMidiEvent);
202 return e;
203 }
204
205 return nullptr;
206 }
207
208 static void freeVSTEvent (Vst2::VstEvent* e)
209 {
210 if (e->type == Vst2::kVstSysExType)
211 {
212 delete[] (reinterpret_cast<Vst2::VstMidiSysexEvent*> (e)->sysexDump);
213 }
214
215 std::free (e);
216 }
