116 template <typename ProcessContext>
117 void process (const ProcessContext& context) noexcept
118 {
119 const auto& inputBlock = context.getInputBlock();
120 auto& outputBlock = context.getOutputBlock();
121 const auto numChannels = outputBlock.getNumChannels();
122 const auto numSamples = outputBlock.getNumSamples();
123
124 jassert (inputBlock.getNumChannels() <= s1.size());
125 jassert (inputBlock.getNumChannels() == numChannels);
126 jassert (inputBlock.getNumSamples() == numSamples);
127
128 if (context.isBypassed)
129 {
130 outputBlock.copyFrom (inputBlock);
131 return;
132 }
133
134 for (size_t channel = 0; channel < numChannels; ++channel)
135 {
136 auto* inputSamples = inputBlock .getChannelPointer (channel);
137 auto* outputSamples = outputBlock.getChannelPointer (channel);
138
139 for (size_t i = 0; i < numSamples; ++i)
140 outputSamples[i] = processSample ((int) channel, inputSamples[i]);
141 }
142
143 #if JUCE_DSP_ENABLE_SNAP_TO_ZERO
144 snapToZero();
145 #endif
146 }
