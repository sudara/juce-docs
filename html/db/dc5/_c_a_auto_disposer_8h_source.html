54inline void* CA_malloc(size_t size)
55{
56 void* p = malloc(size);
57 if (!p && size) throw std::bad_alloc();
58 return p;
59}

61inline void* CA_realloc(void* old, size_t size)
62{
63#if TARGET_OS_WIN32
64 void* p = realloc(old, size);
65#else
66 void* p = reallocf(old, size); // reallocf ensures the old pointer is freed if memory is full (p is NULL).
67#endif
68 if (!p && size) throw std::bad_alloc();
69 return p;
70}

80inline void* CA_calloc(size_t n, size_t size)
81{
82 // ensure that multiplication will not overflow
83 if (n && UINTPTR_MAX / n < size) throw std::bad_alloc();
84
85 size_t nsize = n*size;
86 void* p = malloc(nsize);
87 if (!p && nsize) throw std::bad_alloc();
88
89 memset(p, 0, nsize);
90 return p;
91}

121 CAAutoFree(size_t n, bool clear = false)
122 // this becomes an ambiguous call if n == 0
123 : ptr_(0)
124 {
125 size_t maxItems = ~size_t(0) / sizeof(T);
126 if (n > maxItems)

128
129 ptr_ = static_cast<T*>(clear ? CA_calloc(n, sizeof(T)) : CA_malloc(n * sizeof(T)));

135 {
136 size_t maxItems = ~size_t(0) / sizeof(T);
137 if (numItems > maxItems) throw std::bad_alloc();
138
139 free();
140 ptr_ = static_cast<T*>(clear ? CA_calloc(numItems, sizeof(T)) : CA_malloc(numItems * sizeof(T)));
141 }
142
143 void allocBytes(size_t numBytes, bool clear = false)
144 {
145 free();
