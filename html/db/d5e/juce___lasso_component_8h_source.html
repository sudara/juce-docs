119 void beginLasso (const MouseEvent& e, LassoSource<SelectableItemType>* lassoSource)
120 {
121 jassert (source == nullptr); // this suggests that you didn't call endLasso() after the last drag...
122 jassert (lassoSource != nullptr); // the source can't be null!
123 jassert (getParentComponent() != nullptr); // you need to add this to a parent component for it to work!
124
125 source = lassoSource;
126
127 if (lassoSource != nullptr)
128 originalSelection = lassoSource->getLassoSelection().getItemArray();
129
130 setSize (0, 0);
131 dragStartPos = e.getMouseDownPosition();
132 }
133
146 void dragLasso (const MouseEvent& e)
147 {
148 if (source != nullptr)
149 {
150 setBounds (Rectangle<int> (dragStartPos, e.getPosition()));
151 setVisible (true);
152
153 Array<SelectableItemType> itemsInLasso;
154 source->findLassoItemsInArea (itemsInLasso, getBounds());
155
156 if (e.mods.isShiftDown())
157 {
158 itemsInLasso.removeValuesIn (originalSelection); // to avoid duplicates
159 itemsInLasso.addArray (originalSelection);
160 }
161 else if (e.mods.isCommandDown() || e.mods.isAltDown())
162 {
163 auto originalMinusNew = originalSelection;
164 originalMinusNew.removeValuesIn (itemsInLasso);
165
166 itemsInLasso.removeValuesIn (originalSelection);
167 itemsInLasso.addArray (originalMinusNew);
168 }
169
170 source->getLassoSelection() = SelectedItemSet<SelectableItemType> (itemsInLasso);
171 }
172 }
173
177 void endLasso()
178 {
179 source = nullptr;
180 originalSelection.clear();
181 setVisible (false);
182 }
