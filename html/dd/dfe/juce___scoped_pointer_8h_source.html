57 ScopedPointer& operator= (ScopedPointer& objectToTransferFrom)
58 {
59 if (this != objectToTransferFrom.getAddress())
60 {
61 // Two ScopedPointers should never be able to refer to the same object - if
62 // this happens, you must have done something dodgy!
63 jassert (object == nullptr || object != objectToTransferFrom.object);
64 reset (objectToTransferFrom.release());
65 }
66
67 return *this;
68 }

100 void reset (ObjectType* newObject)
101 {
102 if (object != newObject)
103 {
104 auto* oldObject = object;
105 object = newObject;
106 ContainerDeletePolicy<ObjectType>::destroy (oldObject);
107 }
108 else
109 {
110 // You're trying to reset this ScopedPointer to itself! This will work here as ScopedPointer does an equality check
111 // but be aware that std::unique_ptr won't do this and you could end up with some nasty, subtle bugs!
112 jassert (newObject == nullptr);
113 }
114 }
