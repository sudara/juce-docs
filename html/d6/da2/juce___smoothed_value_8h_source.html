36template <typename SmoothedValueType>
37class SmoothedValueBase
38{
39private:
40 //==============================================================================
41 template <typename T> struct FloatTypeHelper;
42
43 template <template <typename> class SmoothedValueClass, typename FloatType>
44 struct FloatTypeHelper <SmoothedValueClass <FloatType>>
45 {
46 using Type = FloatType;
47 };
48
49 template <template <typename, typename> class SmoothedValueClass, typename FloatType, typename SmoothingType>
50 struct FloatTypeHelper <SmoothedValueClass <FloatType, SmoothingType>>
51 {
52 using Type = FloatType;
53 };
54
55public:
56 using FloatType = typename FloatTypeHelper<SmoothedValueType>::Type;
57
58 //==============================================================================
60 SmoothedValueBase() = default;
61
62 //==============================================================================
64 bool isSmoothing() const noexcept { return countdown > 0; }
65
67 FloatType getCurrentValue() const noexcept { return currentValue; }
68
69 //==============================================================================
71 FloatType getTargetValue() const noexcept { return target; }
72
76 void setCurrentAndTargetValue (FloatType newValue)
77 {
78 target = currentValue = newValue;
79 countdown = 0;
80 }
81
82 //==============================================================================
88 void applyGain (FloatType* samples, int numSamples) noexcept
89 {
90 jassert (numSamples >= 0);
91
92 if (isSmoothing())
93 {
94 for (int i = 0; i < numSamples; ++i)
95 samples[i] *= getNextSmoothedValue();
96 }
97 else
98 {
99 FloatVectorOperations::multiply (samples, target, numSamples);
100 }
101 }
102
109 void applyGain (FloatType* samplesOut, const FloatType* samplesIn, int numSamples) noexcept
110 {
111 jassert (numSamples >= 0);
112
113 if (isSmoothing())
114 {
115 for (int i = 0; i < numSamples; ++i)
116 samplesOut[i] = samplesIn[i] * getNextSmoothedValue();
117 }
118 else
119 {
120 FloatVectorOperations::multiply (samplesOut, samplesIn, target, numSamples);
121 }
122 }
123
125 void applyGain (AudioBuffer<FloatType>& buffer, int numSamples) noexcept
126 {
127 jassert (numSamples >= 0);
128
129 if (isSmoothing())
130 {
131 if (buffer.getNumChannels() == 1)
132 {
133 auto* samples = buffer.getWritePointer (0);
134
135 for (int i = 0; i < numSamples; ++i)
136 samples[i] *= getNextSmoothedValue();
137 }
138 else
139 {
140 for (auto i = 0; i < numSamples; ++i)
141 {
142 auto gain = getNextSmoothedValue();
143
144 for (int channel = 0; channel < buffer.getNumChannels(); channel++)
145 buffer.setSample (channel, i, buffer.getSample (channel, i) * gain);
146 }
147 }
148 }
149 else
150 {
151 buffer.applyGain (0, numSamples, target);
152 }
153 }
154
155private:
156 //==============================================================================
157 FloatType getNextSmoothedValue() noexcept
158 {
159 return static_cast <SmoothedValueType*> (this)->getNextValue();
160 }
161
162protected:
163 //==============================================================================
164 FloatType currentValue = 0;
165 FloatType target = currentValue;
166 int countdown = 0;
167};

226template <typename FloatType, typename SmoothingType = ValueSmoothingTypes::Linear>
227class SmoothedValue : public SmoothedValueBase <SmoothedValue <FloatType, SmoothingType>>
228{
229public:
230 //==============================================================================
232 SmoothedValue() noexcept
233 : SmoothedValue ((FloatType) (std::is_same<SmoothingType, ValueSmoothingTypes::Linear>::value ? 0 : 1))
234 {
235 }
236
238 SmoothedValue (FloatType initialValue) noexcept
239 {
240 // Multiplicative smoothed values cannot ever reach 0!
241 jassert (! (std::is_same<SmoothingType, ValueSmoothingTypes::Multiplicative>::value && initialValue == 0));
242
243 // Visual Studio can't handle base class initialisation with CRTP
244 this->currentValue = initialValue;
245 this->target = this->currentValue;
246 }
247
248 //==============================================================================
253 void reset (double sampleRate, double rampLengthInSeconds) noexcept
254 {
255 jassert (sampleRate > 0 && rampLengthInSeconds >= 0);
256 reset ((int) std::floor (rampLengthInSeconds * sampleRate));
257 }
258
262 void reset (int numSteps) noexcept
263 {
264 stepsToTarget = numSteps;
265 this->setCurrentAndTargetValue (this->target);
266 }
267
268 //==============================================================================
272 void setTargetValue (FloatType newValue) noexcept
273 {
274 if (newValue == this->target)
275 return;
276
277 if (stepsToTarget <= 0)
278 {
279 this->setCurrentAndTargetValue (newValue);
280 return;
281 }
282
283 // Multiplicative smoothed values cannot ever reach 0!
284 jassert (! (std::is_same<SmoothingType, ValueSmoothingTypes::Multiplicative>::value && newValue == 0));
285
286 this->target = newValue;
287 this->countdown = stepsToTarget;
288
289 setStepSize();
290 }
291
292 //==============================================================================
296 FloatType getNextValue() noexcept
297 {
298 if (! this->isSmoothing())
299 return this->target;
300
301 --(this->countdown);
302
303 if (this->isSmoothing())
304 setNextValue();
305 else
306 this->currentValue = this->target;
307
308 return this->currentValue;
309 }
310
311 //==============================================================================
317 FloatType skip (int numSamples) noexcept
318 {
319 if (numSamples >= this->countdown)
320 {
321 this->setCurrentAndTargetValue (this->target);
322 return this->target;
323 }
324
325 skipCurrentValue (numSamples);
326
327 this->countdown -= numSamples;
328 return this->currentValue;
329 }
330
331 //==============================================================================
332 #ifndef DOXYGEN
341 [[deprecated ("Use setTargetValue and setCurrentAndTargetValue instead.")]]
342 void setValue (FloatType newValue, bool force = false) noexcept
343 {
344 if (force)
345 {
346 this->setCurrentAndTargetValue (newValue);
347 return;
348 }
349
350 setTargetValue (newValue);
351 }
352 #endif
353
354private:
355 //==============================================================================
356 template <typename T>
357 using LinearVoid = typename std::enable_if <std::is_same <T, ValueSmoothingTypes::Linear>::value, void>::type;
358
359 template <typename T>
360 using MultiplicativeVoid = typename std::enable_if <std::is_same <T, ValueSmoothingTypes::Multiplicative>::value, void>::type;
361
362 //==============================================================================
363 template <typename T = SmoothingType>
364 LinearVoid<T> setStepSize() noexcept
365 {
366 step = (this->target - this->currentValue) / (FloatType) this->countdown;
367 }
368
369 template <typename T = SmoothingType>
370 MultiplicativeVoid<T> setStepSize()
371 {
372 step = std::exp ((std::log (std::abs (this->target)) - std::log (std::abs (this->currentValue))) / (FloatType) this->countdown);
373 }
374
375 //==============================================================================
376 template <typename T = SmoothingType>
377 LinearVoid<T> setNextValue() noexcept
378 {
379 this->currentValue += step;
380 }
381
382 template <typename T = SmoothingType>
383 MultiplicativeVoid<T> setNextValue() noexcept
384 {
385 this->currentValue *= step;
386 }
387
388 //==============================================================================
389 template <typename T = SmoothingType>
390 LinearVoid<T> skipCurrentValue (int numSamples) noexcept
391 {
392 this->currentValue += step * (FloatType) numSamples;
393 }
394
395 template <typename T = SmoothingType>
396 MultiplicativeVoid<T> skipCurrentValue (int numSamples)
397 {
398 this->currentValue *= (FloatType) std::pow (step, numSamples);
399 }
400
401 //==============================================================================
402 FloatType step = FloatType();
403 int stepsToTarget = 0;
404};
