35struct SpeakerMappings : private AudioChannelSet // (inheritance only to give easier access to items in the namespace)

38 struct Mapping
39 {
40 int32 vst2;
41 ChannelType channels[13];
42
43 bool matches (const Array<ChannelType>& chans) const noexcept
44 {
45 auto n = static_cast<int> (sizeof (channels) / sizeof (ChannelType));
46
47 for (int i = 0; i < n; ++i)
48 {
49 if (channels[i] == unknown) return (i == chans.size());
50 if (i == chans.size()) return (channels[i] == unknown);
51
52 if (channels[i] != chans.getUnchecked (i))
53 return false;
54 }
55
56 return true;
57 }
58 };

60 static AudioChannelSet vstArrangementTypeToChannelSet (int32 arr, int fallbackNumChannels)
61 {
62 if (arr == Vst2::kSpeakerArrEmpty) return AudioChannelSet::disabled();
63 else if (arr == Vst2::kSpeakerArrMono) return AudioChannelSet::mono();
64 else if (arr == Vst2::kSpeakerArrStereo) return AudioChannelSet::stereo();
65 else if (arr == Vst2::kSpeakerArr30Cine) return AudioChannelSet::createLCR();
66 else if (arr == Vst2::kSpeakerArr30Music) return AudioChannelSet::createLRS();
67 else if (arr == Vst2::kSpeakerArr40Cine) return AudioChannelSet::createLCRS();
68 else if (arr == Vst2::kSpeakerArr50) return AudioChannelSet::create5point0();
69 else if (arr == Vst2::kSpeakerArr51) return AudioChannelSet::create5point1();
70 else if (arr == Vst2::kSpeakerArr60Cine) return AudioChannelSet::create6point0();
71 else if (arr == Vst2::kSpeakerArr61Cine) return AudioChannelSet::create6point1();
72 else if (arr == Vst2::kSpeakerArr60Music) return AudioChannelSet::create6point0Music();
73 else if (arr == Vst2::kSpeakerArr61Music) return AudioChannelSet::create6point1Music();
74 else if (arr == Vst2::kSpeakerArr70Music) return AudioChannelSet::create7point0();
75 else if (arr == Vst2::kSpeakerArr70Cine) return AudioChannelSet::create7point0SDDS();
76 else if (arr == Vst2::kSpeakerArr71Music) return AudioChannelSet::create7point1();
77 else if (arr == Vst2::kSpeakerArr71Cine) return AudioChannelSet::create7point1SDDS();
78 else if (arr == Vst2::kSpeakerArr40Music) return AudioChannelSet::quadraphonic();
79
80 for (const Mapping* m = getMappings(); m->vst2 != Vst2::kSpeakerArrEmpty; ++m)
81 {
82 if (m->vst2 == arr)
83 {
84 AudioChannelSet s;
85
86 for (int i = 0; m->channels[i] != 0; ++i)
87 s.addChannel (m->channels[i]);
88
89 return s;
90 }
91 }
92
93 return AudioChannelSet::discreteChannels (fallbackNumChannels);
94 }

101 static int32 channelSetToVstArrangementType (AudioChannelSet channels)
102 {
103 if (channels == AudioChannelSet::disabled()) return Vst2::kSpeakerArrEmpty;
104 else if (channels == AudioChannelSet::mono()) return Vst2::kSpeakerArrMono;
105 else if (channels == AudioChannelSet::stereo()) return Vst2::kSpeakerArrStereo;
106 else if (channels == AudioChannelSet::createLCR()) return Vst2::kSpeakerArr30Cine;
107 else if (channels == AudioChannelSet::createLRS()) return Vst2::kSpeakerArr30Music;
108 else if (channels == AudioChannelSet::createLCRS()) return Vst2::kSpeakerArr40Cine;
109 else if (channels == AudioChannelSet::create5point0()) return Vst2::kSpeakerArr50;
110 else if (channels == AudioChannelSet::create5point1()) return Vst2::kSpeakerArr51;
111 else if (channels == AudioChannelSet::create6point0()) return Vst2::kSpeakerArr60Cine;
112 else if (channels == AudioChannelSet::create6point1()) return Vst2::kSpeakerArr61Cine;
113 else if (channels == AudioChannelSet::create6point0Music()) return Vst2::kSpeakerArr60Music;
114 else if (channels == AudioChannelSet::create6point1Music()) return Vst2::kSpeakerArr61Music;
115 else if (channels == AudioChannelSet::create7point0()) return Vst2::kSpeakerArr70Music;
116 else if (channels == AudioChannelSet::create7point0SDDS()) return Vst2::kSpeakerArr70Cine;
117 else if (channels == AudioChannelSet::create7point1()) return Vst2::kSpeakerArr71Music;
118 else if (channels == AudioChannelSet::create7point1SDDS()) return Vst2::kSpeakerArr71Cine;
119 else if (channels == AudioChannelSet::quadraphonic()) return Vst2::kSpeakerArr40Music;
120
121 if (channels == AudioChannelSet::disabled())
122 return Vst2::kSpeakerArrEmpty;
123
124 auto chans = channels.getChannelTypes();
125
126 for (auto* m = getMappings(); m->vst2 != Vst2::kSpeakerArrEmpty; ++m)
127 if (m->matches (chans))
128 return m->vst2;
129
130 return Vst2::kSpeakerArrUserDefined;
131 }

133 static void channelSetToVstArrangement (const AudioChannelSet& channels, Vst2::VstSpeakerArrangement& result)
134 {
135 result.type = channelSetToVstArrangementType (channels);
136 result.numChannels = channels.size();
137
138 for (int i = 0; i < result.numChannels; ++i)
139 {
140 auto& speaker = result.speakers[i];
141
142 zeromem (&speaker, sizeof (Vst2::VstSpeakerProperties));
143 speaker.type = getSpeakerType (channels.getTypeOfChannel (i));
144 }
145 }

172 VstSpeakerConfigurationHolder (const AudioChannelSet& channels)
173 {
174 auto numberOfChannels = channels.size();
175 Vst2::VstSpeakerArrangement& dst = *allocate (numberOfChannels);
176
177 dst.type = channelSetToVstArrangementType (channels);
178 dst.numChannels = numberOfChannels;
179
180 for (int i = 0; i < dst.numChannels; ++i)
181 {
182 Vst2::VstSpeakerProperties& speaker = dst.speakers[i];
183
184 zeromem (&speaker, sizeof (Vst2::VstSpeakerProperties));
185 speaker.type = getSpeakerType (channels.getTypeOfChannel (i));
186 }
187 }

189 VstSpeakerConfigurationHolder& operator= (const VstSpeakerConfigurationHolder& vstConfig) { return operator=(vstConfig.get()); }
190 VstSpeakerConfigurationHolder& operator= (const Vst2::VstSpeakerArrangement& vstConfig)
191 {
192 Vst2::VstSpeakerArrangement& dst = *allocate (vstConfig.numChannels);
193
194 dst.type = vstConfig.type;
195 dst.numChannels = vstConfig.numChannels;
196
197 for (int i = 0; i < dst.numChannels; ++i)
198 dst.speakers[i] = vstConfig.speakers[i];
199
200 return *this;
201 }

209 }

218 Vst2::VstSpeakerArrangement* allocate (int numChannels)
219 {
220 auto arrangementSize = (size_t) (jmax (8, numChannels) - 8) * sizeof (Vst2::VstSpeakerProperties)
221 + sizeof (Vst2::VstSpeakerArrangement);
222
223 storage.malloc (1, arrangementSize);
