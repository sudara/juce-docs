32class OpenGLContext::NativeContext : private ComponentPeer::ScaleFactorListener
33{
34public:
35 NativeContext (Component& component,
36 const OpenGLPixelFormat& pixelFormat,
37 void* contextToShareWithIn,
38 bool /*useMultisampling*/,
39 OpenGLVersion version)
40 {
41 dummyComponent.reset (new DummyComponent (*this));
42 createNativeWindow (component);
43
44 PIXELFORMATDESCRIPTOR pfd;
45 initialisePixelFormatDescriptor (pfd, pixelFormat);
46
47 auto pixFormat = ChoosePixelFormat (dc, &pfd);
48
49 if (pixFormat != 0)
50 SetPixelFormat (dc, pixFormat, &pfd);
51
52 initialiseWGLExtensions (dc);
53 renderContext = createRenderContext (version, dc);
54
55 if (renderContext != nullptr)
56 {
57 makeActive();
58
59 auto wglFormat = wglChoosePixelFormatExtension (pixelFormat);
60 deactivateCurrentContext();
61
62 if (wglFormat != pixFormat && wglFormat != 0)
63 {
64 // can't change the pixel format of a window, so need to delete the
65 // old one and create a new one..
66 releaseDC();
67 nativeWindow = nullptr;
68 createNativeWindow (component);
69
70 if (SetPixelFormat (dc, wglFormat, &pfd))
71 {
72 deleteRenderContext();
73 renderContext = createRenderContext (version, dc);
74 }
75 }
76
77 if (contextToShareWithIn != nullptr)
78 wglShareLists ((HGLRC) contextToShareWithIn, renderContext);
79
80 component.getTopLevelComponent()->repaint();
81 component.repaint();
82 }
83 }
84
85 ~NativeContext() override
86 {
87 deleteRenderContext();
88 releaseDC();
89
90 if (safeComponent != nullptr)
91 if (auto* peer = safeComponent->getTopLevelComponent()->getPeer())
92 peer->removeScaleFactorListener (this);
93 }

127 {
128 if (nativeWindow != nullptr)

130 if (! approximatelyEqual (nativeScaleFactor, 1.0))
131 bounds = (bounds.toDouble() * nativeScaleFactor).toNearestInt();

134 bounds.getX(), bounds.getY(), bounds.getWidth(), bounds.getHeight(),
135 SWP_NOACTIVATE | SWP_NOZORDER | SWP_NOOWNERZORDER);
136 }

161 static void initialiseWGLExtensions (HDC dcIn)
162 {
163 static bool initialised = false;
164
165 if (initialised)
166 return;
167
168 initialised = true;
169
170 const auto dummyContext = wglCreateContext (dcIn);
171 wglMakeCurrent (dcIn, dummyContext);

173 #define JUCE_INIT_WGL_FUNCTION(name) name = (type_ ## name) OpenGLHelpers::getExtensionFunction (#name);
174 JUCE_INIT_WGL_FUNCTION (wglChoosePixelFormatARB)
175 JUCE_INIT_WGL_FUNCTION (wglSwapIntervalEXT)
176 JUCE_INIT_WGL_FUNCTION (wglGetSwapIntervalEXT)
177 JUCE_INIT_WGL_FUNCTION (wglCreateContextAttribsARB)
178 #undef JUCE_INIT_WGL_FUNCTION
179
180 wglMakeCurrent (nullptr, nullptr);
181 wglDeleteContext (dummyContext);
182 }

184 static void initialisePixelFormatDescriptor (PIXELFORMATDESCRIPTOR& pfd, const OpenGLPixelFormat& pixelFormat)
185 {
186 zerostruct (pfd);
187 pfd.nSize = sizeof (pfd);
188 pfd.nVersion = 1;
189 pfd.dwFlags = PFD_SUPPORT_OPENGL | PFD_DRAW_TO_WINDOW | PFD_DOUBLEBUFFER;
190 pfd.iPixelType = PFD_TYPE_RGBA;
191 pfd.iLayerType = PFD_MAIN_PLANE;
192 pfd.cColorBits = (BYTE) (pixelFormat.redBits + pixelFormat.greenBits + pixelFormat.blueBits);
193 pfd.cRedBits = (BYTE) pixelFormat.redBits;
194 pfd.cGreenBits = (BYTE) pixelFormat.greenBits;
195 pfd.cBlueBits = (BYTE) pixelFormat.blueBits;
196 pfd.cAlphaBits = (BYTE) pixelFormat.alphaBits;
197 pfd.cDepthBits = (BYTE) pixelFormat.depthBufferBits;
198 pfd.cStencilBits = (BYTE) pixelFormat.stencilBufferBits;
199 pfd.cAccumBits = (BYTE) (pixelFormat.accumulationBufferRedBits + pixelFormat.accumulationBufferGreenBits
200 + pixelFormat.accumulationBufferBlueBits + pixelFormat.accumulationBufferAlphaBits);
201 pfd.cAccumRedBits = (BYTE) pixelFormat.accumulationBufferRedBits;
202 pfd.cAccumGreenBits = (BYTE) pixelFormat.accumulationBufferGreenBits;
203 pfd.cAccumBlueBits = (BYTE) pixelFormat.accumulationBufferBlueBits;
204 pfd.cAccumAlphaBits = (BYTE) pixelFormat.accumulationBufferAlphaBits;
205 }

207 static HGLRC createRenderContext (OpenGLVersion version, HDC dcIn)
208 {
209 if (version >= openGL3_2 && wglCreateContextAttribsARB != nullptr)
210 {
211 const int attribs[] =
212 {
213 WGL_CONTEXT_MAJOR_VERSION_ARB, 3,
214 WGL_CONTEXT_MINOR_VERSION_ARB, 2,
215 WGL_CONTEXT_PROFILE_MASK_ARB, WGL_CONTEXT_CORE_PROFILE_BIT_ARB,
216 0
217 };
218
219 const auto c = wglCreateContextAttribsARB (dcIn, nullptr, attribs);
220
221 if (c != nullptr)
222 return c;
223 }
224
225 return wglCreateContext (dcIn);
226 }

240 void nativeScaleFactorChanged (double newScaleFactor) override
241 {
242 if (approximatelyEqual (newScaleFactor, nativeScaleFactor)
243 || safeComponent == nullptr)
244 return;
245
246 if (auto* peer = safeComponent->getTopLevelComponent()->getPeer())
247 {
248 nativeScaleFactor = newScaleFactor;
249 updateWindowPosition (peer->getAreaCoveredBy (*safeComponent));
250 }

253 void createNativeWindow (Component& component)
254 {
255 auto* topComp = component.getTopLevelComponent();
256
257 {
258 auto* parentHWND = topComp->getWindowHandle();
259
260 ScopedThreadDPIAwarenessSetter setter { parentHWND };
261 nativeWindow.reset (createNonRepaintingEmbeddedWindowsPeer (*dummyComponent, parentHWND));
262 }
263
264 if (auto* peer = topComp->getPeer())
265 {
266 safeComponent = Component::SafePointer<Component> (&component);
267
268 nativeScaleFactor = peer->getPlatformScaleFactor();
269 updateWindowPosition (peer->getAreaCoveredBy (component));
270 peer->addScaleFactorListener (this);
271 }
272
273 nativeWindow->setVisible (true);
274 dc = GetDC ((HWND) nativeWindow->getNativeHandle());
275 }

291 int wglChoosePixelFormatExtension (const OpenGLPixelFormat& pixelFormat) const
292 {
293 int format = 0;
294
295 if (wglChoosePixelFormatARB != nullptr)
296 {
297 int atts[64];
298 int n = 0;
299
300 atts[n++] = WGL_DRAW_TO_WINDOW_ARB; atts[n++] = GL_TRUE;
301 atts[n++] = WGL_SUPPORT_OPENGL_ARB; atts[n++] = GL_TRUE;
302 atts[n++] = WGL_DOUBLE_BUFFER_ARB; atts[n++] = GL_TRUE;
303 atts[n++] = WGL_PIXEL_TYPE_ARB; atts[n++] = WGL_TYPE_RGBA_ARB;
304 atts[n++] = WGL_ACCELERATION_ARB;
305 atts[n++] = WGL_FULL_ACCELERATION_ARB;
306
307 atts[n++] = WGL_COLOR_BITS_ARB; atts[n++] = pixelFormat.redBits + pixelFormat.greenBits + pixelFormat.blueBits;
308 atts[n++] = WGL_RED_BITS_ARB; atts[n++] = pixelFormat.redBits;
309 atts[n++] = WGL_GREEN_BITS_ARB; atts[n++] = pixelFormat.greenBits;
310 atts[n++] = WGL_BLUE_BITS_ARB; atts[n++] = pixelFormat.blueBits;
311 atts[n++] = WGL_ALPHA_BITS_ARB; atts[n++] = pixelFormat.alphaBits;
312 atts[n++] = WGL_DEPTH_BITS_ARB; atts[n++] = pixelFormat.depthBufferBits;
313
314 atts[n++] = WGL_STENCIL_BITS_ARB; atts[n++] = pixelFormat.stencilBufferBits;
315 atts[n++] = WGL_ACCUM_RED_BITS_ARB; atts[n++] = pixelFormat.accumulationBufferRedBits;
316 atts[n++] = WGL_ACCUM_GREEN_BITS_ARB; atts[n++] = pixelFormat.accumulationBufferGreenBits;
317 atts[n++] = WGL_ACCUM_BLUE_BITS_ARB; atts[n++] = pixelFormat.accumulationBufferBlueBits;
318 atts[n++] = WGL_ACCUM_ALPHA_BITS_ARB; atts[n++] = pixelFormat.accumulationBufferAlphaBits;
319
320 if (pixelFormat.multisamplingLevel > 0
321 && OpenGLHelpers::isExtensionSupported ("GL_ARB_multisample"))
322 {
323 atts[n++] = WGL_SAMPLE_BUFFERS_ARB;
324 atts[n++] = 1;
325 atts[n++] = WGL_SAMPLES_ARB;
326 atts[n++] = pixelFormat.multisamplingLevel;
327 }
328
329 atts[n++] = 0;
330 jassert (n <= numElementsInArray (atts));
331
332 UINT formatsCount = 0;
333 wglChoosePixelFormatARB (dc, atts, nullptr, 1, &format, &formatsCount);
334 }
335
336 return format;
337 }
