1391 bool isSeekInProgress() const noexcept
1392 {
1393 if (pendingSeekRequest)
1394 return true;
1395
1396 if (! usesBuffering)
1397 return false;
1398
1399 // NB: player sometimes notifies us about buffering, but only for regions that
1400 // were previously buffered already. For buffering happening for the first time,
1401 // we don't get such notification...
1402 if (playerBufferingInProgress)
1403 return true;
1404
1405 auto playPos = player.getPlayPosition();
1406 auto durationMs = player.getVideoDuration();
1407 auto playPosPercent = (int) (100.0 * playPos / static_cast<double> (durationMs));
1408
1409 // NB: assuming the playback will start roughly when there is 5% of content loaded...
1410 return ! bufferedRegions.containsRange (Range<int> (playPosPercent, jmin (101, playPosPercent + 5)));
1411 }
