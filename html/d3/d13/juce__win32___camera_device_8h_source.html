26struct CameraDevice::Pimpl : public ChangeBroadcaster
27{
28 Pimpl (CameraDevice& ownerToUse, const String&, int index,
29 int minWidth, int minHeight, int maxWidth, int maxHeight,
30 bool /*highQuality*/)
31 : owner (ownerToUse)
32 {
33 HRESULT hr = captureGraphBuilder.CoCreateInstance (ComTypes::CLSID_CaptureGraphBuilder2);
34 if (FAILED (hr))
35 return;
36
37 filter = enumerateCameras (nullptr, index);
38 if (filter == nullptr)
39 return;
40
41 hr = graphBuilder.CoCreateInstance (ComTypes::CLSID_FilterGraph);
42 if (FAILED (hr))
43 return;
44
45 hr = captureGraphBuilder->SetFiltergraph (graphBuilder);
46 if (FAILED (hr))
47 return;
48
49 mediaControl = graphBuilder.getInterface<ComTypes::IMediaControl>();
50 if (mediaControl == nullptr)
51 return;
52
53 {
54 ComSmartPtr<ComTypes::IAMStreamConfig> streamConfig;
55
56 JUCE_BEGIN_IGNORE_WARNINGS_GCC_LIKE ("-Wlanguage-extension-token")
57 hr = captureGraphBuilder->FindInterface (&ComTypes::PIN_CATEGORY_CAPTURE, nullptr, filter,
58 __uuidof (ComTypes::IAMStreamConfig), (void**) streamConfig.resetAndGetPointerAddress());
59 JUCE_END_IGNORE_WARNINGS_GCC_LIKE
60
61 if (streamConfig != nullptr)
62 {
63 getVideoSizes (streamConfig);
64
65 if (! selectVideoSize (streamConfig, minWidth, minHeight, maxWidth, maxHeight))
66 return;
67 }
68 }
69
70 hr = graphBuilder->AddFilter (filter, _T("Video Capture"));
71 if (FAILED (hr))
72 return;
73
74 hr = smartTee.CoCreateInstance (ComTypes::CLSID_SmartTee);
75 if (FAILED (hr))
76 return;
77
78 hr = graphBuilder->AddFilter (smartTee, _T("Smart Tee"));
79 if (FAILED (hr))
80 return;
81
82 if (! connectFilters (filter, smartTee))
83 return;
84
85 ComSmartPtr<ComTypes::IBaseFilter> sampleGrabberBase;
86 hr = sampleGrabberBase.CoCreateInstance (ComTypes::CLSID_SampleGrabber);
87 if (FAILED (hr))
88 return;
89
90 JUCE_BEGIN_IGNORE_WARNINGS_GCC_LIKE ("-Wlanguage-extension-token")
91 hr = sampleGrabberBase.QueryInterface (__uuidof (ComTypes::ISampleGrabber), sampleGrabber);
92 JUCE_END_IGNORE_WARNINGS_GCC_LIKE
93
94 if (FAILED (hr))
95 return;
96
97 {
98 ComTypes::AM_MEDIA_TYPE mt = {};
99 mt.majortype = ComTypes::MEDIATYPE_Video;
100 mt.subtype = ComTypes::MEDIASUBTYPE_RGB24;
101 mt.formattype = ComTypes::FORMAT_VideoInfo;
102 sampleGrabber->SetMediaType (&mt);
103 }
104
105 callback = new GrabberCallback (*this);
106 hr = sampleGrabber->SetCallback (callback, 1);
107
108 hr = graphBuilder->AddFilter (sampleGrabberBase, _T("Sample Grabber"));
109 if (FAILED (hr))
110 return;
111
112 ComSmartPtr<ComTypes::IPin> grabberInputPin;
113 if (! (getPin (smartTee, ComTypes::PINDIR_OUTPUT, smartTeeCaptureOutputPin, "capture")
114 && getPin (smartTee, ComTypes::PINDIR_OUTPUT, smartTeePreviewOutputPin, "preview")
115 && getPin (sampleGrabberBase, ComTypes::PINDIR_INPUT, grabberInputPin)))
116 return;
117
118 hr = graphBuilder->Connect (smartTeePreviewOutputPin, grabberInputPin);
119 if (FAILED (hr))
120 return;
121
122 ComTypes::AM_MEDIA_TYPE mt = {};
123 hr = sampleGrabber->GetConnectedMediaType (&mt);
124
125 if (auto* pVih = unalignedPointerCast<ComTypes::VIDEOINFOHEADER*> (mt.pbFormat))
126 {
127 width = pVih->bmiHeader.biWidth;
128 height = pVih->bmiHeader.biHeight;
129 }
130
131 ComSmartPtr<ComTypes::IBaseFilter> nullFilter;
132 hr = nullFilter.CoCreateInstance (ComTypes::CLSID_NullRenderer);
133 hr = graphBuilder->AddFilter (nullFilter, _T("Null Renderer"));
134
135 if (connectFilters (sampleGrabberBase, nullFilter)
136 && addGraphToRot())
137 {

142 }

267 void handleFrame (double /*time*/, BYTE* buffer, long /*bufferSize*/)
268 {
269 if (recordNextFrameTime)
270 {
271 const double defaultCameraLatency = 0.1;
272
273 firstRecordedTime = Time::getCurrentTime() - RelativeTime (defaultCameraLatency);
274 recordNextFrameTime = false;
275
276 ComSmartPtr<ComTypes::IPin> pin;
277 if (getPin (filter, ComTypes::PINDIR_OUTPUT, pin))
278 {
279 if (auto pushSource = pin.getInterface<ComTypes::IAMPushSource>())
280 {
281 ComTypes::REFERENCE_TIME latency = 0;
282 pushSource->GetLatency (&latency);
283
284 firstRecordedTime = firstRecordedTime - RelativeTime ((double) latency);
285 }
286 }
287 }
288
289 {
290 const int lineStride = width * 3;
291 const ScopedLock sl (imageSwapLock);
292
293 {
294 loadingImage.duplicateIfShared();
295 const Image::BitmapData destData (loadingImage, 0, 0, width, height, Image::BitmapData::writeOnly);
296
297 for (int i = 0; i < height; ++i)
298 memcpy (destData.getLinePointer ((height - 1) - i),
299 buffer + lineStride * i,
300 (size_t) lineStride);
301 }
302
303 imageNeedsFlipping = true;
304 }
305
306 if (listeners.size() > 0)
307 callListeners (loadingImage);
308
309 notifyPictureTakenIfNeeded (loadingImage);
310
311 sendChangeMessage();
312 }

314 void drawCurrentImage (Graphics& g, Rectangle<int> area)
315 {
316 if (imageNeedsFlipping)
317 {
318 const ScopedLock sl (imageSwapLock);
319 std::swap (loadingImage, activeImage);
320 imageNeedsFlipping = false;
321 }
322
323 Rectangle<int> centred (RectanglePlacement (RectanglePlacement::centred)
324 .appliedTo (Rectangle<int> (width, height), area));
325
326 RectangleList<int> borders (area);
327 borders.subtract (centred);
328 g.setColour (Colours::black);
329 g.fillRectList (borders);
330
331 g.drawImage (activeImage, centred.getX(), centred.getY(),
332 centred.getWidth(), centred.getHeight(), 0, 0, width, height);
333 }

335 bool createFileCaptureFilter (const File& file, int quality)
336 {
337 removeFileCaptureFilter();
338 file.deleteFile();
339 mediaControl->Stop();
340 firstRecordedTime = Time();
341 recordNextFrameTime = true;
342 previewMaxFPS = 60;
343
344 HRESULT hr = asfWriter.CoCreateInstance (ComTypes::CLSID_WMAsfWriter);
345
346 if (SUCCEEDED (hr))
347 {
348 if (auto fileSink = asfWriter.getInterface<ComTypes::IFileSinkFilter>())
349 {
350 hr = fileSink->SetFileName (file.getFullPathName().toWideCharPointer(), nullptr);
351
352 if (SUCCEEDED (hr))
353 {
354 hr = graphBuilder->AddFilter (asfWriter, _T("AsfWriter"));
355
356 if (SUCCEEDED (hr))
357 {
358 if (auto asfConfig = asfWriter.getInterface<ComTypes::IConfigAsfWriter>())
359 {
360 asfConfig->SetIndexMode (true);
361 ComSmartPtr<IWMProfileManager> profileManager;
362
363 using Fn = HRESULT (*) (IWMProfileManager**);
364
365 // This function is available on Windows 2000 and up, but we load it at runtime anyway
366 // because some versions of MinGW ship with libraries that don't include this symbol.
367 if (auto* fn = reinterpret_cast<Fn> (wmvcoreLibrary.getFunction ("WMCreateProfileManager")))
368 hr = fn (profileManager.resetAndGetPointerAddress());
369 else
370 jassertfalse;
371
372 // This gibberish is the DirectShow profile for a video-only wmv file.
373 String prof ("<profile version=\"589824\" storageformat=\"1\" name=\"Quality\" description=\"Quality type for output.\">"
374 "<streamconfig majortype=\"{73646976-0000-0010-8000-00AA00389B71}\" streamnumber=\"1\" "
375 "streamname=\"Video Stream\" inputname=\"Video409\" bitrate=\"894960\" "
376 "bufferwindow=\"0\" reliabletransport=\"1\" decodercomplexity=\"AU\" rfc1766langid=\"en-us\">"
377 "<videomediaprops maxkeyframespacing=\"50000000\" quality=\"90\"/>"
378 "<wmmediatype subtype=\"{33564D57-0000-0010-8000-00AA00389B71}\" bfixedsizesamples=\"0\" "
379 "btemporalcompression=\"1\" lsamplesize=\"0\">"
380 "<videoinfoheader dwbitrate=\"894960\" dwbiterrorrate=\"0\" avgtimeperframe=\"$AVGTIMEPERFRAME\">"
381 "<rcsource left=\"0\" top=\"0\" right=\"$WIDTH\" bottom=\"$HEIGHT\"/>"
382 "<rctarget left=\"0\" top=\"0\" right=\"$WIDTH\" bottom=\"$HEIGHT\"/>"
383 "<bitmapinfoheader biwidth=\"$WIDTH\" biheight=\"$HEIGHT\" biplanes=\"1\" bibitcount=\"24\" "
384 "bicompression=\"WMV3\" bisizeimage=\"0\" bixpelspermeter=\"0\" biypelspermeter=\"0\" "
385 "biclrused=\"0\" biclrimportant=\"0\"/>"
386 "</videoinfoheader>"
387 "</wmmediatype>"
388 "</streamconfig>"
389 "</profile>");
390
391 const int fps[] = { 10, 15, 30 };
392 int maxFramesPerSecond = fps[jlimit (0, numElementsInArray (fps) - 1, quality & 0xff)];
393
394 if (((uint32_t) quality & 0xff000000) != 0) // (internal hacky way to pass explicit frame rates for testing)
395 maxFramesPerSecond = (quality >> 24) & 0xff;
396
397 prof = prof.replace ("$WIDTH", String (width))
398 .replace ("$HEIGHT", String (height))
399 .replace ("$AVGTIMEPERFRAME", String (10000000 / maxFramesPerSecond));
400
401 ComSmartPtr<IWMProfile> currentProfile;
402 hr = profileManager->LoadProfileByData (prof.toWideCharPointer(), currentProfile.resetAndGetPointerAddress());
403 hr = asfConfig->ConfigureFilterUsingProfile (currentProfile);
404
405 if (SUCCEEDED (hr))
406 {
407 ComSmartPtr<ComTypes::IPin> asfWriterInputPin;
408
409 if (getPin (asfWriter, ComTypes::PINDIR_INPUT, asfWriterInputPin, "Video Input 01"))
410 {
411 hr = graphBuilder->Connect (smartTeeCaptureOutputPin, asfWriterInputPin);
412
413 if (SUCCEEDED (hr) && openedSuccessfully && activeUsers > 0
414 && SUCCEEDED (mediaControl->Run()))
415 {
416 previewMaxFPS = (quality < 2) ? 15 : 25; // throttle back the preview comps to try to leave the cpu free for encoding
417
418 if ((quality & 0x00ff0000) != 0) // (internal hacky way to pass explicit frame rates for testing)
419 previewMaxFPS = (quality >> 16) & 0xff;
420
421 return true;
422 }
423 }
424 }
425 }
426 }
427 }
428 }
429 }

455 static ComSmartPtr<ComTypes::IBaseFilter> enumerateCameras (StringArray* names, const int deviceIndexToOpen)
456 {
457 int index = 0;
458 ComSmartPtr<ComTypes::ICreateDevEnum> pDevEnum;
459
460 struct Deleter
461 {
462 void operator() (IUnknown* ptr) const noexcept { ptr->Release(); }
463 };
464
465 using ContextPtr = std::unique_ptr<IBindCtx, Deleter>;
466
467 if (SUCCEEDED (pDevEnum.CoCreateInstance (ComTypes::CLSID_SystemDeviceEnum)))
468 {
469 ComSmartPtr<IEnumMoniker> enumerator;
470 HRESULT hr = pDevEnum->CreateClassEnumerator (ComTypes::CLSID_VideoInputDeviceCategory, enumerator.resetAndGetPointerAddress(), 0);
471
472 if (SUCCEEDED (hr) && enumerator != nullptr)
473 {
474 ComSmartPtr<IMoniker> moniker;
475 ULONG fetched;
476
477 while (enumerator->Next (1, moniker.resetAndGetPointerAddress(), &fetched) == S_OK)
478 {
479 auto context = []
480 {
481 IBindCtx* ptr = nullptr;
482 ignoreUnused (CreateBindCtx (0, &ptr));
483 return ContextPtr (ptr);
484 }();
485
486 ComSmartPtr<ComTypes::IBaseFilter> captureFilter;
487 JUCE_BEGIN_IGNORE_WARNINGS_GCC_LIKE ("-Wlanguage-extension-token")
488 hr = moniker->BindToObject (context.get(), nullptr, __uuidof (ComTypes::IBaseFilter), (void**) captureFilter.resetAndGetPointerAddress());
489 JUCE_END_IGNORE_WARNINGS_GCC_LIKE
490
491 if (SUCCEEDED (hr))
492 {
493 ComSmartPtr<IPropertyBag> propertyBag;
494 hr = moniker->BindToStorage (context.get(), nullptr, IID_IPropertyBag, (void**) propertyBag.resetAndGetPointerAddress());
495
496 if (SUCCEEDED (hr))
497 {
498 VARIANT var;
499 var.vt = VT_BSTR;
500
501 hr = propertyBag->Read (_T("FriendlyName"), &var, nullptr);
502 propertyBag = nullptr;
503
504 if (SUCCEEDED (hr))
505 {
506 if (names != nullptr)
507 names->add (var.bstrVal);
508
509 if (index == deviceIndexToOpen)
510 return captureFilter;
511
512 ++index;
513 }
514 }
515 }
516 }
517 }
518 }
519
520 return nullptr;
521 }

595 void getVideoSizes (ComTypes::IAMStreamConfig* const streamConfig)
596 {
597 widths.clear();
598 heights.clear();
599
600 int count = 0, size = 0;
601 streamConfig->GetNumberOfCapabilities (&count, &size);
602
603 if (size == (int) sizeof (ComTypes::VIDEO_STREAM_CONFIG_CAPS))
604 {
605 for (int i = 0; i < count; ++i)
606 {
607 ComTypes::VIDEO_STREAM_CONFIG_CAPS scc;
608 ComTypes::AM_MEDIA_TYPE* config;
609
610 HRESULT hr = streamConfig->GetStreamCaps (i, &config, (BYTE*) &scc);
611
612 if (SUCCEEDED (hr))
613 {
614 const int w = scc.InputSize.cx;
615 const int h = scc.InputSize.cy;
616
617 bool duplicate = false;
618
619 for (int j = widths.size(); --j >= 0;)
620 {
621 if (w == widths.getUnchecked (j) && h == heights.getUnchecked (j))
622 {
623 duplicate = true;
624 break;
625 }
626 }
627
628 if (! duplicate)
629 {
630 widths.add (w);
631 heights.add (h);
632 }
633
634 deleteMediaType (config);
635 }
636 }
637 }

640 bool selectVideoSize (ComTypes::IAMStreamConfig* const streamConfig,
641 const int minWidth, const int minHeight,
642 const int maxWidth, const int maxHeight)
643 {
644 int count = 0, size = 0, bestArea = 0, bestIndex = -1;
645 streamConfig->GetNumberOfCapabilities (&count, &size);
646
647 if (size == (int) sizeof (ComTypes::VIDEO_STREAM_CONFIG_CAPS))
648 {
649 ComTypes::AM_MEDIA_TYPE* config;
650 ComTypes::VIDEO_STREAM_CONFIG_CAPS scc;
651
652 for (int i = 0; i < count; ++i)
653 {
654 HRESULT hr = streamConfig->GetStreamCaps (i, &config, (BYTE*) &scc);
655
656 if (SUCCEEDED (hr))
657 {
658 if (scc.InputSize.cx >= minWidth
659 && scc.InputSize.cy >= minHeight
660 && scc.InputSize.cx <= maxWidth
661 && scc.InputSize.cy <= maxHeight)
662 {
663 int area = scc.InputSize.cx * scc.InputSize.cy;
664 if (area > bestArea)
665 {
666 bestIndex = i;
667 bestArea = area;
668 }
669 }
670
671 deleteMediaType (config);
672 }
673 }
674
675 if (bestIndex >= 0)
676 {
677 HRESULT hr = streamConfig->GetStreamCaps (bestIndex, &config, (BYTE*) &scc);
678
679 hr = streamConfig->SetFormat (config);
680 deleteMediaType (config);
681 return SUCCEEDED (hr);
682 }
683 }
684
685 return false;
686 }

688 static bool getPin (ComTypes::IBaseFilter* filter, const ComTypes::PIN_DIRECTION wantedDirection,
689 ComSmartPtr<ComTypes::IPin>& result, const char* pinName = nullptr)
690 {
691 ComSmartPtr<ComTypes::IEnumPins> enumerator;
692 ComSmartPtr<ComTypes::IPin> pin;
693
694 filter->EnumPins (enumerator.resetAndGetPointerAddress());
695
696 while (enumerator->Next (1, pin.resetAndGetPointerAddress(), nullptr) == S_OK)
697 {
698 ComTypes::PIN_DIRECTION dir;
699 pin->QueryDirection (&dir);
700
701 if (wantedDirection == dir)
702 {
703 ComTypes::PIN_INFO info = {};
704 pin->QueryPinInfo (&info);
705
706 if (pinName == nullptr || String (pinName).equalsIgnoreCase (String (info.achName)))
707 {
708 result = pin;
709 return true;
710 }
711 }
712 }
713
714 return false;
715 }
