36 //==============================================================================
37 JUCE_COMRESULT SetValue (LPCWSTR val) override
38 {
39 if (! isElementValid())
40 return (HRESULT) UIA_E_ELEMENTNOTAVAILABLE;
41
42 const auto& handler = getHandler();
43 auto& valueInterface = *handler.getValueInterface();
44
45 if (valueInterface.isReadOnly())
46 return (HRESULT) UIA_E_NOTSUPPORTED;
47
48 valueInterface.setValueAsString (String (val));
49
50 VARIANT newValue;
51 VariantHelpers::setString (valueInterface.getCurrentValueAsString(), &newValue);
52
53 sendAccessibilityPropertyChangedEvent (handler, UIA_ValueValuePropertyId, newValue);
54
55 return S_OK;
56 }
57
58 JUCE_COMRESULT get_Value (BSTR* pRetVal) override
59 {
60 return withCheckedComArgs (pRetVal, *this, [&]
61 {
62 auto currentValueString = getHandler().getValueInterface()->getCurrentValueAsString();
63
64 *pRetVal = SysAllocString ((const OLECHAR*) currentValueString.toWideCharPointer());
65 return S_OK;
66 });
67 }
68
69 JUCE_COMRESULT get_IsReadOnly (BOOL* pRetVal) override
70 {
71 return withCheckedComArgs (pRetVal, *this, [&]
72 {
73 *pRetVal = getHandler().getValueInterface()->isReadOnly();
74 return S_OK;
75 });
76 }
77
