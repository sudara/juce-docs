71 SelectedItemSet& operator= (const SelectedItemSet& other)
72 {
73 if (selectedItems != other.selectedItems)
74 {
75 changed();
76
77 for (int i = selectedItems.size(); --i >= 0;)
78 if (! other.isSelected (selectedItems.getReference (i)))
79 itemDeselected (selectedItems.removeAndReturn (i));
80
81 for (auto& i : other.selectedItems)
82 {

102 void selectOnly (ParameterType item)
103 {
104 if (isSelected (item))
105 {
106 for (int i = selectedItems.size(); --i >= 0;)
107 {
108 if (selectedItems.getUnchecked(i) != item)
109 {
110 deselect (selectedItems.getUnchecked(i));

161 void addToSelectionBasedOnModifiers (ParameterType item,
162 ModifierKeys modifiers)
163 {
164 if (modifiers.isShiftDown())
165 {

167 }
168 else if (modifiers.isCommandDown())
169 {
170 if (isSelected (item))

175 else

198 bool addToSelectionOnMouseDown (ParameterType item,
199 ModifierKeys modifiers)
200 {

203
204 addToSelectionBasedOnModifiers (item, modifiers);
