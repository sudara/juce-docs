42/*
43 * These typedefs make it easier to ensure that integer versions of
44 * the library really only contain integer operations. All the code
45 * in libFLAC should use FLAC__float and FLAC__double in place of
46 * float and double, and be protected by checks of the macro
47 * FLAC__INTEGER_ONLY_LIBRARY.
48 *
49 * FLAC__real is the basic floating point type used in LPC analysis.
50 */
51#ifndef FLAC__INTEGER_ONLY_LIBRARY
52typedef double FLAC__double;
53typedef float FLAC__float;
54/*
55 * WATCHOUT: changing FLAC__real will change the signatures of many
56 * functions that have assembly language equivalents and break them.
57 */
58typedef float FLAC__real;

77/*
78 * FLAC__fixedpoint_log2()
79 * --------------------------------------------------------------------
80 * Returns the base-2 logarithm of the fixed-point number 'x' using an
81 * algorithm by Knuth for x >= 1.0
82 *
83 * 'fracbits' is the number of fractional bits of 'x'. 'fracbits' must
84 * be < 32 and evenly divisible by 4 (0 is OK but not very precise).
85 *
86 * 'precision' roughly limits the number of iterations that are done;
87 * use (unsigned)(-1) for maximum precision.
88 *
89 * If 'x' is less than one -- that is, x < (1<<fracbits) -- then this
90 * function will punt and return 0.
91 *
92 * The return value will also have 'fracbits' fractional bits.
93 */
94FLAC__uint32 FLAC__fixedpoint_log2(FLAC__uint32 x, unsigned fracbits, unsigned precision);
