61inline JUCE_COMRESULT addHandlersToArray (const std::vector<const AccessibilityHandler*>& handlers, SAFEARRAY** pRetVal)
62{
63 auto numHandlers = handlers.size();
64
65 *pRetVal = SafeArrayCreateVector (VT_UNKNOWN, 0, (ULONG) numHandlers);
66
67 if (pRetVal != nullptr)
68 {
69 for (LONG i = 0; i < (LONG) numHandlers; ++i)
70 {
71 auto* handler = handlers[(size_t) i];
72
73 if (handler == nullptr)
74 continue;
75
76 ComSmartPtr<IRawElementProviderSimple> provider;
77 JUCE_BEGIN_IGNORE_WARNINGS_GCC_LIKE ("-Wlanguage-extension-token")
78 handler->getNativeImplementation()->QueryInterface (IID_PPV_ARGS (provider.resetAndGetPointerAddress()));
79 JUCE_END_IGNORE_WARNINGS_GCC_LIKE
80
81 auto hr = SafeArrayPutElement (*pRetVal, &i, provider);
82
83 if (FAILED (hr))
84 return E_FAIL;
85 }
86 }
87
88 return S_OK;
89}
