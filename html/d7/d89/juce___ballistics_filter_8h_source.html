98 template <typename ProcessContext>
99 void process (const ProcessContext& context) noexcept
100 {
101 const auto& inputBlock = context.getInputBlock();
102 auto& outputBlock = context.getOutputBlock();
103 const auto numChannels = outputBlock.getNumChannels();
104 const auto numSamples = outputBlock.getNumSamples();
105
106 jassert (inputBlock.getNumChannels() <= yold.size());
107 jassert (inputBlock.getNumChannels() == numChannels);
108 jassert (inputBlock.getNumSamples() == numSamples);
109
110 if (context.isBypassed)
111 {
112 outputBlock.copyFrom (inputBlock);
113 return;
114 }
115
116 for (size_t channel = 0; channel < numChannels; ++channel)
117 {
118 auto* inputSamples = inputBlock .getChannelPointer (channel);
119 auto* outputSamples = outputBlock.getChannelPointer (channel);
120
121 for (size_t i = 0; i < numSamples; ++i)
122 outputSamples[i] = processSample ((int) channel, inputSamples[i]);
123 }
124
125 #if JUCE_DSP_ENABLE_SNAP_TO_ZERO
126 snapToZero();
127 #endif
128 }
