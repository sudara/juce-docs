194 {

196 getLastFile(),
197 getFilePatterns (fileSuffix));

199 | FileBrowserComponent::canSelectFiles
200 | FileBrowserComponent::warnAboutOverwriting;
201
202 stateFileChooser->launchAsync (flags, [this] (const FileChooser& fc)
203 {
204 if (fc.getResult() == File{})
205 return;
206
207 setLastFile (fc);
208
209 MemoryBlock data;
210 processor->getStateInformation (data);
211
212 if (! fc.getResult().replaceWithData (data.getData(), data.getSize()))
213 AlertWindow::showMessageBoxAsync (AlertWindow::WarningIcon,
214 TRANS("Error whilst saving"),
215 TRANS("Couldn't write to the specified file!"));
216 });
217 }

220 void askUserToLoadState (const String& fileSuffix = String())
221 {
222 stateFileChooser = std::make_unique<FileChooser> (TRANS("Load a saved state"),
223 getLastFile(),
224 getFilePatterns (fileSuffix));
225 auto flags = FileBrowserComponent::openMode
226 | FileBrowserComponent::canSelectFiles;
227
228 stateFileChooser->launchAsync (flags, [this] (const FileChooser& fc)
229 {
230 if (fc.getResult() == File{})
231 return;
232
233 setLastFile (fc);
234
235 MemoryBlock data;
236
237 if (fc.getResult().loadFileAsData (data))
238 processor->setStateInformation (data.getData(), (int) data.getSize());
239 else
240 AlertWindow::showMessageBoxAsync (AlertWindow::WarningIcon,
241 TRANS("Error whilst loading"),
242 TRANS("Couldn't read from the specified file!"));

423private:
424 /* This class can be used to ensure that audio callbacks use buffers with a
425 predictable maximum size.
426
427 On some platforms (such as iOS 10), the expected buffer size reported in
428 audioDeviceAboutToStart may be smaller than the blocks passed to
429 audioDeviceIOCallback. This can lead to out-of-bounds reads if the render
430 callback depends on additional buffers which were initialised using the
431 smaller size.
432
433 As a workaround, this class will ensure that the render callback will
434 only ever be called with a block with a length less than or equal to the
435 expected block size.
436 */
437 class CallbackMaxSizeEnforcer : public AudioIODeviceCallback
438 {
439 public:
440 explicit CallbackMaxSizeEnforcer (AudioIODeviceCallback& callbackIn)
441 : inner (callbackIn) {}
442
443 void audioDeviceAboutToStart (AudioIODevice* device) override
444 {
445 maximumSize = device->getCurrentBufferSizeSamples();
446 storedInputChannels .resize ((size_t) device->getActiveInputChannels() .countNumberOfSetBits());
447 storedOutputChannels.resize ((size_t) device->getActiveOutputChannels().countNumberOfSetBits());
448
449 inner.audioDeviceAboutToStart (device);
450 }
451
452 void audioDeviceIOCallback (const float** inputChannelData,
453 int numInputChannels,
454 float** outputChannelData,
455 int numOutputChannels,
456 int numSamples) override
457 {
458 jassertquiet ((int) storedInputChannels.size() == numInputChannels);
459 jassertquiet ((int) storedOutputChannels.size() == numOutputChannels);
460
461 int position = 0;
462
463 while (position < numSamples)
464 {
465 const auto blockLength = jmin (maximumSize, numSamples - position);
466
467 initChannelPointers (inputChannelData, storedInputChannels, position);
468 initChannelPointers (outputChannelData, storedOutputChannels, position);
469
470 inner.audioDeviceIOCallback (storedInputChannels.data(),
471 (int) storedInputChannels.size(),
472 storedOutputChannels.data(),
473 (int) storedOutputChannels.size(),
474 blockLength);
475
476 position += blockLength;
477 }
478 }
479
480 void audioDeviceStopped() override
481 {
482 inner.audioDeviceStopped();
483 }
484
485 private:
486 struct GetChannelWithOffset
487 {
488 int offset;
489
490 template <typename Ptr>
491 auto operator() (Ptr ptr) const noexcept -> Ptr { return ptr + offset; }
492 };
493
494 template <typename Ptr, typename Vector>
495 void initChannelPointers (Ptr&& source, Vector&& target, int offset)
496 {
497 std::transform (source, source + target.size(), target.begin(), GetChannelWithOffset { offset });
498 }
499
500 AudioIODeviceCallback& inner;
501 int maximumSize = 0;
502 std::vector<const float*> storedInputChannels;
503 std::vector<float*> storedOutputChannels;
504 };
505
506 CallbackMaxSizeEnforcer maxSizeEnforcer { *this };

1060 /* This custom constrainer checks with the AudioProcessorEditor (which might itself be
1061 constrained) to ensure that any size we choose for the standalone window will be suitable
1062 for the editor too.
1063
1064 Without this constrainer, attempting to resize the standalone window may set bounds on the
1065 peer that are unsupported by the inner editor. In this scenario, the peer will be set to a
1066 'bad' size, then the inner editor will be resized. The editor will check the new bounds with
1067 its own constrainer, and may set itself to a more suitable size. After that, the resizable
1068 window will see that its content component has changed size, and set the bounds of the peer
1069 accordingly. The end result is that the peer is resized twice in a row to different sizes,
1070 which can appear glitchy/flickery to the user.
1071 */
1072 struct DecoratorConstrainer : public ComponentBoundsConstrainer
1073 {
1074 void checkBounds (Rectangle<int>& bounds,
1075 const Rectangle<int>& previousBounds,
1076 const Rectangle<int>& limits,
1077 bool isStretchingTop,
1078 bool isStretchingLeft,
1079 bool isStretchingBottom,
1080 bool isStretchingRight) override
1081 {
1082 auto* decorated = contentComponent != nullptr ? contentComponent->getEditorConstrainer()
1083 : nullptr;
1084
1085 if (decorated != nullptr)
1086 {
1087 const auto border = contentComponent->computeBorder();
1088 const auto requestedBounds = bounds;
1089
1090 border.subtractFrom (bounds);
1091 decorated->checkBounds (bounds,
1092 border.subtractedFrom (previousBounds),
1093 limits,
1094 isStretchingTop,
1095 isStretchingLeft,
1096 isStretchingBottom,
1097 isStretchingRight);
1098 border.addTo (bounds);
1099 bounds = bounds.withPosition (requestedBounds.getPosition());
1100
1101 if (isStretchingTop && ! isStretchingBottom)
1102 bounds = bounds.withBottomY (previousBounds.getBottom());
1103
1104 if (! isStretchingTop && isStretchingBottom)
1105 bounds = bounds.withY (previousBounds.getY());
1106
1107 if (isStretchingLeft && ! isStretchingRight)
1108 bounds = bounds.withRightX (previousBounds.getRight());
1109
1110 if (! isStretchingLeft && isStretchingRight)
1111 bounds = bounds.withX (previousBounds.getX());
1112 }
1113 else
1114 {
1115 ComponentBoundsConstrainer::checkBounds (bounds,
1116 previousBounds,
1117 limits,
1118 isStretchingTop,
1119 isStretchingLeft,
1120 isStretchingBottom,
1121 isStretchingRight);
1122 }
1123 }
1124
1125 void setMainContentComponent (MainContentComponent* in) { contentComponent = in; }
1126
1127 private:
1128 MainContentComponent* contentComponent = nullptr;
1129 };
