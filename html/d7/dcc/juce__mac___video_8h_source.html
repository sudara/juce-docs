218 static PlayerControllerBase& getOwner (id self) { return *getIvar<PlayerControllerBase*> (self, "owner"); }
219 static void setOwner (id self, PlayerControllerBase* p) { object_setInstanceVariable (self, "owner", p); }
220
221 private:
222 static void valueChanged (id self, SEL, NSString* keyPath, id,
223 NSDictionary<NSString*, id>* change, void*)
224 {
225 auto& owner = getOwner (self);
226
227 if ([keyPath isEqualToString: nsStringLiteral ("rate")])
228 {
229 auto oldRate = [change[NSKeyValueChangeOldKey] floatValue];
230 auto newRate = [change[NSKeyValueChangeNewKey] floatValue];
231
232 if (oldRate == 0 && newRate != 0)
233 owner.playbackStarted();
234 else if (oldRate != 0 && newRate == 0)
235 owner.playbackStopped();
236 }
237 else if ([keyPath isEqualToString: nsStringLiteral ("status")])
238 {
239 auto status = [change[NSKeyValueChangeNewKey] intValue];
240
241 if (status == AVPlayerStatusFailed)
242 owner.errorOccurred();
243 }
244 }
245 };

306 {
307 JUCE_BEGIN_IGNORE_WARNINGS_GCC_LIKE ("-Wundeclared-selector")
308 addMethod (@selector (observeValueForKeyPath:ofObject:change:context:), valueChanged);

316 //==============================================================================
317 static PlayerAsyncInitialiser& getOwner (id self) { return *getIvar<PlayerAsyncInitialiser*> (self, "owner"); }
318 static void setOwner (id self, PlayerAsyncInitialiser* p) { object_setInstanceVariable (self, "owner", p); }
319
320 private:
321 static void valueChanged (id self, SEL, NSString*, id object,
322 NSDictionary<NSString*, id>* change, void* context)
323 {
324 auto& owner = getOwner (self);
325
326 if (context == &owner)
327 {
328 auto* playerItem = (AVPlayerItem*) object;
329 auto* urlAsset = (AVURLAsset*) playerItem.asset;
330
331 URL url (nsStringToJuce (urlAsset.URL.absoluteString));
332 auto oldStatus = [change[NSKeyValueChangeOldKey] intValue];
333 auto newStatus = [change[NSKeyValueChangeNewKey] intValue];
334
335 // Ignore spurious notifications
336 if (oldStatus == newStatus)
337 return;
338
339 if (newStatus == AVPlayerItemStatusFailed)
340 {
341 auto errorMessage = playerItem.error != nil
342 ? nsStringToJuce (playerItem.error.localizedDescription)
343 : String();
344
345 owner.notifyOwnerPreparationFinished (url, Result::fail (errorMessage), nullptr);
346 }
347 else if (newStatus == AVPlayerItemStatusReadyToPlay)
348 {
349 owner.notifyOwnerPreparationFinished (url, Result::ok(), owner.player.get());
350 }
351 else
352 {
353 jassertfalse;
354 }
355 }
356 }
357 };

369 void checkAllKeysReadyFor (AVAsset* assetToCheck, const URL& url)
370 {
371 NSError* error = nil;
372
373 int successCount = 0;
374
375 for (NSString* key : assetKeys.get())
376 {
377 switch ([assetToCheck statusOfValueForKey: key error: &error])
378 {
379 case AVKeyValueStatusLoaded:
380 {
381 ++successCount;
382 break;
383 }
384 case AVKeyValueStatusCancelled:
385 {
386 notifyOwnerPreparationFinished (url, Result::fail ("Loading cancelled"), nullptr);
387 return;
388 }
389 case AVKeyValueStatusFailed:
390 {
391 auto errorMessage = error != nil ? nsStringToJuce (error.localizedDescription) : String();
392 notifyOwnerPreparationFinished (url, Result::fail (errorMessage), nullptr);
393 return;
394 }
395
396 case AVKeyValueStatusUnknown:
397 case AVKeyValueStatusLoading:
398 default:
399 break;
400 }
401 }
402
403 jassert (successCount == (int) [assetKeys.get() count]);
404 preparePlayerItem();
405 }

418 {
419 [playerItem.get() addObserver: playerItemPreparationStatusObserver.get()
420 forKeyPath: nsStringLiteral ("status")
421 options: NSKeyValueObservingOptionOld | NSKeyValueObservingOptionNew
422 context: this];
423 }
424

426 {
427 if (playerItem != nullptr && playerItemPreparationStatusObserver != nullptr)
428 {
429 [playerItem.get() removeObserver: playerItemPreparationStatusObserver.get()
430 forKeyPath: nsStringLiteral ("status")
431 context: this];
432 }
433 }

439 safeThis = WeakReference<PlayerAsyncInitialiser> { this }]() mutable
440 {
441 if (safeThis != nullptr)

443 });
444 }
