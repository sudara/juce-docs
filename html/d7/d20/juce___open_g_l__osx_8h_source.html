77 static void createAttribs (NSOpenGLPixelFormatAttribute* attribs, OpenGLVersion version,
78 const OpenGLPixelFormat& pixFormat, bool shouldUseMultisampling)
79 {
80 ignoreUnused (version);
81 int numAttribs = 0;
82
83 attribs[numAttribs++] = NSOpenGLPFAOpenGLProfile;
84 attribs[numAttribs++] = version >= openGL3_2 ? NSOpenGLProfileVersion3_2Core
85 : NSOpenGLProfileVersionLegacy;
86
87 attribs[numAttribs++] = NSOpenGLPFADoubleBuffer;
88 attribs[numAttribs++] = NSOpenGLPFAClosestPolicy;
89 attribs[numAttribs++] = NSOpenGLPFANoRecovery;
90 attribs[numAttribs++] = NSOpenGLPFAColorSize;
91 attribs[numAttribs++] = (NSOpenGLPixelFormatAttribute) (pixFormat.redBits + pixFormat.greenBits + pixFormat.blueBits);
92 attribs[numAttribs++] = NSOpenGLPFAAlphaSize;
93 attribs[numAttribs++] = (NSOpenGLPixelFormatAttribute) pixFormat.alphaBits;
94 attribs[numAttribs++] = NSOpenGLPFADepthSize;
95 attribs[numAttribs++] = (NSOpenGLPixelFormatAttribute) pixFormat.depthBufferBits;
96 attribs[numAttribs++] = NSOpenGLPFAStencilSize;
97 attribs[numAttribs++] = (NSOpenGLPixelFormatAttribute) pixFormat.stencilBufferBits;
98 attribs[numAttribs++] = NSOpenGLPFAAccumSize;
99 attribs[numAttribs++] = (NSOpenGLPixelFormatAttribute) (pixFormat.accumulationBufferRedBits + pixFormat.accumulationBufferGreenBits
100 + pixFormat.accumulationBufferBlueBits + pixFormat.accumulationBufferAlphaBits);
101
102 if (shouldUseMultisampling)
103 {
104 attribs[numAttribs++] = NSOpenGLPFAMultisample;
105 attribs[numAttribs++] = NSOpenGLPFASampleBuffers;
106 attribs[numAttribs++] = (NSOpenGLPixelFormatAttribute) 1;
107 attribs[numAttribs++] = NSOpenGLPFASamples;
108 attribs[numAttribs++] = (NSOpenGLPixelFormatAttribute) pixFormat.multisamplingLevel;
109 }
110 }

161 void swapBuffers()
162 {
163 auto now = Time::getMillisecondCounterHiRes();
164 [renderContext flushBuffer];
165
166 if (minSwapTimeMs > 0)
167 {
168 // When our window is entirely occluded by other windows, flushBuffer
169 // fails to wait for the swap interval, so the render loop spins at full
170 // speed, burning CPU. This hack detects when things are going too fast
171 // and sleeps if necessary.
172
173 auto swapTime = Time::getMillisecondCounterHiRes() - now;
174 auto frameTime = (int) (now - lastSwapTime);
175
176 if (swapTime < 0.5 && frameTime < minSwapTimeMs - 3)
177 {
178 if (underrunCounter > 3)
179 {
180 Thread::sleep (2 * (minSwapTimeMs - frameTime));
181 now = Time::getMillisecondCounterHiRes();
182 }
183 else
184 {
185 ++underrunCounter;
186 }
187 }
188 else
189 {
190 if (underrunCounter > 0)
191 --underrunCounter;
192 }
193 }
194
195 lastSwapTime = now;
196 }

200 bool setSwapInterval (int numFramesPerSwapIn)
201 {
202 numFramesPerSwap = numFramesPerSwapIn;
203
204 // The macOS OpenGL programming guide says that numFramesPerSwap
205 // can only be 0 or 1.
206 jassert (isPositiveAndBelow (numFramesPerSwap, 2));
207
208 [renderContext setValues: (const GLint*) &numFramesPerSwap
209 forParameter: getSwapIntervalParameter()];
210
211 updateMinSwapTime();
212
213 return true;
214 }
215
216 int getSwapInterval() const
217 {
218 GLint numFrames = 0;
219 [renderContext getValues: &numFrames
220 forParameter: getSwapIntervalParameter()];
221
222 return numFrames;
223 }
