51 template <typename PacketCallback>
52 void dispatch (const View& v, PacketCallback&& callback)
53 {
54 const auto firstWord = v[0];
55 const auto messageType = Utils::getMessageType (firstWord);
56
57 if (messageType != 0x2)
58 {
59 callback (v);
60 return;
61 }
62
63 const HelperValues helperValues
64 {
65 (uint8_t) ((0x4 << 0x4) | Utils::getGroup (firstWord)),
66 (uint8_t) ((firstWord >> 0x10) & 0xff),
67 (uint8_t) ((firstWord >> 0x08) & 0x7f),
68 (uint8_t) ((firstWord >> 0x00) & 0x7f),
69 };
70
71 switch (Utils::getStatus (firstWord))
72 {
73 case 0x8:
74 case 0x9:
75 {
76 const auto packet = processNoteOnOrOff (helperValues);
77 callback (View (packet.data()));
78 return;
79 }
80
81 case 0xa:
82 {
83 const auto packet = processPolyPressure (helperValues);
84 callback (View (packet.data()));
85 return;
86 }
87
88 case 0xb:
89 {
90 PacketX2 packet;
91
92 if (processControlChange (helperValues, packet))
93 callback (View (packet.data()));
94
95 return;
96 }
97
98 case 0xc:
99 {
100 const auto packet = processProgramChange (helperValues);
101 callback (View (packet.data()));
102 return;
103 }
104
105 case 0xd:
106 {
107 const auto packet = processChannelPressure (helperValues);
108 callback (View (packet.data()));
109 return;
110 }
111
112 case 0xe:
113 {
114 const auto packet = processPitchBend (helperValues);
115 callback (View (packet.data()));
116 return;
117 }
118 }
