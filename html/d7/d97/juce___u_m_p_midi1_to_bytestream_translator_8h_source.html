63 template <typename MessageCallback>
64 void dispatch (const View& packet, double time, MessageCallback&& callback)
65 {
66 const auto firstWord = *packet.data();
67
68 if (! pendingSysExData.empty() && shouldPacketTerminateSysExEarly (firstWord))
69 pendingSysExData.clear();
70
71 switch (packet.size())
72 {
73 case 1:
74 {
75 // Utility messages don't translate to bytestream format
76 if (Utils::getMessageType (firstWord) != 0x00)
77 callback (fromUmp (PacketX1 { firstWord }, time));
78
79 break;
80 }
81
82 case 2:
83 {
84 if (Utils::getMessageType (firstWord) == 0x3)
85 processSysEx (PacketX2 { packet[0], packet[1] }, time, callback);
86
87 break;
88 }
89
90 case 3: // no 3-word packets in the current spec
91 case 4: // no 4-word packets translate to bytestream format
92 default:
93 break;
94 }
95 }

107 static MidiMessage fromUmp (const PacketX1& m, double time = 0)
108 {
109 const auto word = m.front();
110 jassert (Utils::getNumWordsForMessageType (word) == 1);
111
112 const std::array<uint8_t, 3> bytes { { uint8_t ((word >> 0x10) & 0xff),
113 uint8_t ((word >> 0x08) & 0xff),
114 uint8_t ((word >> 0x00) & 0xff) } };
115 const auto numBytes = MidiMessage::getMessageLengthFromFirstByte (bytes.front());
116 return MidiMessage (bytes.data(), numBytes, time);
117 }

120 template <typename MessageCallback>
121 void processSysEx (const PacketX2& packet,
122 double time,
123 MessageCallback&& callback)
124 {
125 switch (getSysEx7Kind (packet[0]))
126 {
127 case SysEx7::Kind::complete:
128 startSysExMessage (time);
129 pushBytes (packet);
130 terminateSysExMessage (callback);
131 break;
132
133 case SysEx7::Kind::begin:
134 startSysExMessage (time);
135 pushBytes (packet);
136 break;
137
138 case SysEx7::Kind::continuation:
139 if (pendingSysExData.empty())
140 break;
141
142 pushBytes (packet);
143 break;
144
145 case SysEx7::Kind::end:
146 if (pendingSysExData.empty())
147 break;
148
149 pushBytes (packet);
150 terminateSysExMessage (callback);
151 break;
152 }

179 static bool shouldPacketTerminateSysExEarly (uint32_t firstWord)
180 {
181 return ! (isSysExContinuation (firstWord)
182 || isSystemRealTime (firstWord)
183 || isJROrNOP (firstWord));
184 }
185
186 static SysEx7::Kind getSysEx7Kind (uint32_t word)
187 {
188 return SysEx7::Kind ((word >> 0x14) & 0xf);
189 }
190
191 static bool isJROrNOP (uint32_t word)
192 {
193 return Utils::getMessageType (word) == 0x0;
194 }
195
196 static bool isSysExContinuation (uint32_t word)
197 {
198 if (Utils::getMessageType (word) != 0x3)
199 return false;
200
201 const auto kind = getSysEx7Kind (word);
202 return kind == SysEx7::Kind::continuation || kind == SysEx7::Kind::end;
203 }
204
205 static bool isSystemRealTime (uint32_t word)
206 {
207 return Utils::getMessageType (word) == 0x1 && ((word >> 0x10) & 0xff) >= 0xf8;
208 }
