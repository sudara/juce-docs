271 static String varArrayToDelimitedString (const Array<var>& input, StringRef delim)
272 {
273 // if you are trying to control a var that is an array then you need to
274 // set a delimiter string that will be used when writing to XML!
275 jassert (delim.isNotEmpty());
276
277 StringArray elements;
278
279 for (auto& v : input)
280 elements.add (v.toString());
281
282 return elements.joinIntoString (delim);
283 }
284
285 static Array<var> delimitedStringToVarArray (StringRef input, StringRef delim)
286 {
287 Array<var> arr;
288
289 for (auto t : StringArray::fromTokens (input, delim, {}))
290 arr.add (t);
291
292 return arr;
293 }
