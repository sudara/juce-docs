43 template <typename PacketCallbackFunction>
44 static void toMidi1 (const MidiMessage& m, PacketCallbackFunction&& callback)
45 {
46 const auto* data = m.getRawData();
47 const auto firstByte = data[0];
48 const auto size = m.getRawDataSize();
49
50 if (firstByte != 0xf0)
51 {
52 const auto mask = [size]() -> uint32_t
53 {
54 switch (size)
55 {
56 case 0: return 0xff000000;
57 case 1: return 0xffff0000;
58 case 2: return 0xffffff00;
59 case 3: return 0xffffffff;
60 }
61
62 return 0x00000000;
63 }();
64
65 const auto extraByte = (uint8_t) ((((firstByte & 0xf0) == 0xf0) ? 0x1 : 0x2) << 0x4);
66 const PacketX1 packet { mask & Utils::bytesToWord (extraByte, data[0], data[1], data[2]) };
67 callback (View (packet.data()));
68 return;
69 }
70
71 const auto numSysExBytes = m.getSysExDataSize();
72 const auto numMessages = SysEx7::getNumPacketsRequiredForDataSize ((uint32_t) numSysExBytes);
73 auto* dataOffset = m.getSysExData();
74
75 if (numMessages <= 1)
76 {
77 const auto packet = Factory::makeSysExIn1Packet (0, (uint8_t) numSysExBytes, dataOffset);
78 callback (View (packet.data()));
79 return;
80 }
81
82 constexpr auto byteIncrement = 6;
83
84 for (auto i = numSysExBytes; i > 0; i -= byteIncrement, dataOffset += byteIncrement)
85 {
86 const auto func = [&]
87 {
88 if (i == numSysExBytes)
89 return Factory::makeSysExStart;
90
91 if (i <= byteIncrement)
92 return Factory::makeSysExEnd;
93
94 return Factory::makeSysExContinue;
95 }();
96
97 const auto bytesNow = std::min (byteIncrement, i);
98 const auto packet = func (0, (uint8_t) bytesNow, dataOffset);
99 callback (View (packet.data()));
100 }
101 }

118 static uint8_t scaleTo8 (uint8_t word7Bit)
119 {
120 const auto shifted = (uint8_t) (word7Bit << 0x1);
121 const auto repeat = (uint8_t) (word7Bit & 0x3f);
122 const auto mask = (uint8_t) (word7Bit <= 0x40 ? 0x0 : 0xff);
123 return (uint8_t) (shifted | ((repeat >> 5) & mask));
124 }

127 static uint16_t scaleTo16 (uint8_t word7Bit)
128 {
129 const auto shifted = (uint16_t) (word7Bit << 0x9);
130 const auto repeat = (uint16_t) (word7Bit & 0x3f);
131 const auto mask = (uint16_t) (word7Bit <= 0x40 ? 0x0 : 0xffff);
132 return (uint16_t) (shifted | (((repeat << 3) | (repeat >> 3)) & mask));
133 }

136 static uint16_t scaleTo16 (uint16_t word14Bit)

138 const auto shifted = (uint16_t) (word14Bit << 0x2);

140 const auto mask = (uint16_t) (word14Bit <= 0x2000 ? 0x0 : 0xffff);
141 return (uint16_t) (shifted | ((repeat >> 11) & mask));

145 static uint32_t scaleTo32 (uint8_t word7Bit)
146 {
147 const auto shifted = (uint32_t) (word7Bit << 0x19);
148 const auto repeat = (uint32_t) (word7Bit & 0x3f);
149 const auto mask = (uint32_t) (word7Bit <= 0x40 ? 0x0 : 0xffffffff);
150 return (uint32_t) (shifted | (((repeat << 19)
151 | (repeat << 13)
152 | (repeat << 7)
153 | (repeat << 1)
154 | (repeat >> 5)) & mask));
155 }

162 const auto mask = (uint32_t) (word14Bit <= 0x2000 ? 0x0 : 0xffffffff);
163 return (uint32_t) (shifted | (((repeat << 5) | (repeat >> 8)) & mask));

190 template <typename Callback>
191 static void midi2ToMidi1DefaultTranslation (const View& v, Callback&& callback)
192 {
193 const auto firstWord = v[0];
194
195 if (Utils::getMessageType (firstWord) != 0x4)
196 {
197 callback (v);
198 return;
199 }
200
201 const auto status = Utils::getStatus (firstWord);
202 const auto typeAndGroup = (uint8_t) ((0x2 << 0x4) | Utils::getGroup (firstWord));
203
204 switch (status)
205 {
206 case 0x8: // note off
207 case 0x9: // note on
208 case 0xa: // poly pressure
209 case 0xb: // control change
210 {
211 const auto statusAndChannel = (uint8_t) ((firstWord >> 0x10) & 0xff);
212 const auto byte2 = (uint8_t) ((firstWord >> 0x08) & 0xff);
213 const auto byte3 = scaleTo7 (v[1]);
214
215 // If this is a note-on, and the scaled byte is 0,
216 // the scaled velocity should be 1 instead of 0
217 const auto needsCorrection = status == 0x9 && byte3 == 0;
218 const auto correctedByte = (uint8_t) (needsCorrection ? 1 : byte3);
219
220 const auto shouldIgnore = status == 0xb && [&]
221 {
222 switch (byte2)
223 {
224 case 0:
225 case 6:
226 case 32:
227 case 38:
228 case 98:
229 case 99:
230 case 100:
231 case 101:
232 return true;
233 }
234
235 return false;
236 }();
237
238 if (shouldIgnore)
239 return;
240
241 const PacketX1 packet { Utils::bytesToWord (typeAndGroup,
242 statusAndChannel,
243 byte2,
244 correctedByte) };
245 callback (View (packet.data()));
246 return;
247 }
248
249 case 0xd: // channel pressure
250 {
251 const auto statusAndChannel = (uint8_t) ((firstWord >> 0x10) & 0xff);
252 const auto byte2 = scaleTo7 (v[1]);
253
254 const PacketX1 packet { Utils::bytesToWord (typeAndGroup,
255 statusAndChannel,
256 byte2,
257 0) };
258 callback (View (packet.data()));
259 return;
260 }
261
262 case 0x2: // rpn
263 case 0x3: // nrpn
264 {
265 const auto ccX = (uint8_t) (status == 0x2 ? 101 : 99);
266 const auto ccY = (uint8_t) (status == 0x2 ? 100 : 98);
267 const auto statusAndChannel = (uint8_t) ((0xb << 0x4) | Utils::getChannel (firstWord));
268 const auto data = scaleTo14 (v[1]);
269
270 const PacketX1 packets[]
271 {
272 PacketX1 { Utils::bytesToWord (typeAndGroup, statusAndChannel, ccX, (uint8_t) ((firstWord >> 0x8) & 0x7f)) },
273 PacketX1 { Utils::bytesToWord (typeAndGroup, statusAndChannel, ccY, (uint8_t) ((firstWord >> 0x0) & 0x7f)) },
274 PacketX1 { Utils::bytesToWord (typeAndGroup, statusAndChannel, 6, (uint8_t) ((data >> 0x7) & 0x7f)) },
275 PacketX1 { Utils::bytesToWord (typeAndGroup, statusAndChannel, 38, (uint8_t) ((data >> 0x0) & 0x7f)) },
276 };
277
278 for (const auto& packet : packets)
279 callback (View (packet.data()));
280
281 return;
282 }
283
284 case 0xc: // program change / bank select
285 {
286 if (firstWord & 1)
287 {
288 const auto statusAndChannel = (uint8_t) ((0xb << 0x4) | Utils::getChannel (firstWord));
289 const auto secondWord = v[1];
290
291 const PacketX1 packets[]
292 {
293 PacketX1 { Utils::bytesToWord (typeAndGroup, statusAndChannel, 0, (uint8_t) ((secondWord >> 0x8) & 0x7f)) },
294 PacketX1 { Utils::bytesToWord (typeAndGroup, statusAndChannel, 32, (uint8_t) ((secondWord >> 0x0) & 0x7f)) },
295 };
296
297 for (const auto& packet : packets)
298 callback (View (packet.data()));
299 }
300
301 const auto statusAndChannel = (uint8_t) ((0xc << 0x4) | Utils::getChannel (firstWord));
302 const PacketX1 packet { Utils::bytesToWord (typeAndGroup,
303 statusAndChannel,
304 (uint8_t) ((v[1] >> 0x18) & 0x7f),
305 0) };
306 callback (View (packet.data()));
307 return;
308 }
309
310 case 0xe: // pitch bend
311 {
312 const auto data = scaleTo14 (v[1]);
313 const auto statusAndChannel = (uint8_t) ((firstWord >> 0x10) & 0xff);
314 const PacketX1 packet { Utils::bytesToWord (typeAndGroup,
315 statusAndChannel,
316 (uint8_t) (data & 0x7f),
317 (uint8_t) ((data >> 7) & 0x7f)) };
318 callback (View (packet.data()));
319 return;
320 }
321
322 default: // other message types do not translate
323 return;
324 }
325 }
