81 template <typename... Items>
82 void add (std::unique_ptr<Items>... items)
83 {
84 parameters.reserve (parameters.size() + sizeof... (items));
85
86 // We can replace this with some nicer code once generic lambdas become available. A
87 // sequential context like an array initialiser is required to ensure we get the correct
88 // order from the parameter pack.
89 int unused[] { (parameters.emplace_back (MakeContents() (std::move (items))), 0)... };
90 ignoreUnused (unused);
91 }
92
93 template <typename It, typename = ValidIfIterator<It>>
94 void add (It begin, It end)
95 {
96 parameters.reserve (parameters.size() + std::size_t (std::distance (begin, end)));
97 std::transform (std::make_move_iterator (begin),
98 std::make_move_iterator (end),
99 std::back_inserter (parameters),
100 MakeContents());
101 }
