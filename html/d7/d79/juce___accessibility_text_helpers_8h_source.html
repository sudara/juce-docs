45 static int findTextBoundary (const AccessibilityTextInterface& textInterface,
46 int currentPosition,
47 BoundaryType boundary,
48 Direction direction)
49 {
50 const auto numCharacters = textInterface.getTotalNumCharacters();
51 const auto isForwards = (direction == Direction::forwards);
52
53 const auto offsetWithDirecton = [isForwards] (int input) { return isForwards ? input : -input; };
54
55 switch (boundary)
56 {
57 case BoundaryType::character:
58 return jlimit (0, numCharacters, isForwards ? currentPosition + 1 : currentPosition);
59
60 case BoundaryType::word:
61 case BoundaryType::line:
62 {
63 const auto text = [&]() -> String
64 {
65 if (isForwards)
66 return textInterface.getText ({ currentPosition, textInterface.getTotalNumCharacters() });
67
68 const auto str = textInterface.getText ({ 0, currentPosition });
69
70 auto start = str.getCharPointer();
71 auto end = start.findTerminatingNull();
72 const auto size = getAddressDifference (end.getAddress(), start.getAddress());
73
74 String reversed;
75
76 if (size > 0)
77 {
78 reversed.preallocateBytes ((size_t) size);
79
80 auto destPtr = reversed.getCharPointer();
81
82 for (;;)
83 {
84 destPtr.write (*--end);
85
86 if (end == start)
87 break;
88 }
89
90 destPtr.writeNull();
91 }
92
93 return reversed;
94 }();
95
96 auto tokens = (boundary == BoundaryType::line ? StringArray::fromLines (text)
97 : StringArray::fromTokens (text, false));
98
99 return currentPosition + offsetWithDirecton (tokens[0].length());
100 }
101
102 case BoundaryType::document:
103 return isForwards ? numCharacters : 0;
104 }
105
106 jassertfalse;
107 return -1;
108 }
109}
