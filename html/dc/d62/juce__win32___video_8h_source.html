493 Result loadFile (const String& fileOrURLPath)
494 {
495 jassert (state == uninitializedState);
496
497 if (! createNativeWindow())
498 return Result::fail ("Can't create window");
499
500 HRESULT hr = graphBuilder.CoCreateInstance (ComTypes::CLSID_FilterGraph);
501
502 // basic playback interfaces
503 if (SUCCEEDED (hr)) hr = graphBuilder.QueryInterface (mediaControl);
504 if (SUCCEEDED (hr)) hr = graphBuilder.QueryInterface (mediaPosition);
505 if (SUCCEEDED (hr)) hr = graphBuilder.QueryInterface (mediaEvent);
506 if (SUCCEEDED (hr)) hr = graphBuilder.QueryInterface (basicAudio);
507
508 // video renderer interface
509 if (SUCCEEDED (hr))
510 {
511 if (SystemStats::getOperatingSystemType() >= SystemStats::WinVista)
512 {
513 videoRenderer.reset (new VideoRenderers::EVR());
514 hr = videoRenderer->create (graphBuilder, baseFilter, hwnd);
515
516 if (FAILED (hr))
517 videoRenderer = nullptr;
518 }
519
520 if (videoRenderer == nullptr)
521 {
522 videoRenderer.reset (new VideoRenderers::VMR7());
523 hr = videoRenderer->create (graphBuilder, baseFilter, hwnd);
524 }
525 }
526
527 // build filter graph
528 if (SUCCEEDED (hr))
529 {
530 hr = graphBuilder->RenderFile (fileOrURLPath.toWideCharPointer(), nullptr);
531
532 if (FAILED (hr))
533 {
534 #if JUCE_MODAL_LOOPS_PERMITTED
535 // Annoyingly, if we don't run the msg loop between failing and deleting the window, the
536 // whole OS message-dispatch system gets itself into a state, and refuses to deliver any
537 // more messages for the whole app. (That's what happens in Win7, anyway)
538 MessageManager::getInstance()->runDispatchLoopUntil (200);
539 #endif
540 }
541 }
542
543 // remove video renderer if not connected (no video)
544 if (SUCCEEDED (hr))
545 {
546 if (isRendererConnected())
547 {
548 hasVideo = true;
549 }
550 else
551 {
552 hasVideo = false;
553 graphBuilder->RemoveFilter (baseFilter);
554 videoRenderer = nullptr;
555 baseFilter = nullptr;
556 }
557 }
558
559 // set window to receive events
560 if (SUCCEEDED (hr))
561 {
562 mediaEvent->CancelDefaultHandling (ComTypes::EC_STATE_CHANGE);
563 hr = mediaEvent->SetNotifyWindow ((ComTypes::OAHWND) hwnd, graphEventID, 0);
564 }
565
566 if (SUCCEEDED (hr))
567 {
568 state = stoppedState;
569 pause();
570 return Result::ok();
571 }
572
573 // Note that if you're trying to open a file and this method fails, you may
574 // just need to install a suitable codec. It seems that by default DirectShow
575 // doesn't support a very good range of formats.
576 release();
577 return getErrorMessageFromResult (hr);
578 }
579
580 static Result getErrorMessageFromResult (HRESULT hr)
581 {
582 switch (hr)
583 {
584 case ComTypes::VFW_E_INVALID_FILE_FORMAT: return Result::fail ("Invalid file format");
585 case ComTypes::VFW_E_NOT_FOUND: return Result::fail ("File not found");
586 case ComTypes::VFW_E_UNKNOWN_FILE_TYPE: return Result::fail ("Unknown file type");
587 case ComTypes::VFW_E_UNSUPPORTED_STREAM: return Result::fail ("Unsupported stream");
588 case ComTypes::VFW_E_CANNOT_CONNECT: return Result::fail ("Cannot connect");
589 case ComTypes::VFW_E_CANNOT_LOAD_SOURCE_FILTER: return Result::fail ("Cannot load source filter");
590 }
591
592 TCHAR messageBuffer[512] = { 0 };
593
594 FormatMessage (FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
595 nullptr, (DWORD) hr, MAKELANGID (LANG_NEUTRAL, SUBLANG_DEFAULT),
596 messageBuffer, (DWORD) numElementsInArray (messageBuffer) - 1, nullptr);
597
598 return Result::fail (String (messageBuffer));
599 }

601 void release()
602 {
603 if (mediaControl != nullptr)
604 mediaControl->Stop();
605
606 if (mediaEvent != nullptr)
607 mediaEvent->SetNotifyWindow (0, 0, 0);
608
609 if (videoRenderer != nullptr)
610 videoRenderer->setVideoWindow (nullptr);
611
612 hasVideo = false;
613 videoRenderer = nullptr;
614 baseFilter = nullptr;
615 basicAudio = nullptr;
616 mediaEvent = nullptr;
617 mediaPosition = nullptr;
618 mediaControl = nullptr;
619 graphBuilder = nullptr;
620
621 state = uninitializedState;
622
623 if (nativeWindow != nullptr)

627 void graphEventProc()
628 {
629 LONG ec = 0;
630 LONG_PTR p1 = {}, p2 = {};
631
632 jassert (mediaEvent != nullptr);
633
634 while (SUCCEEDED (mediaEvent->GetEvent (&ec, &p1, &p2, 0)))
635 {
636 mediaEvent->FreeEventParams (ec, p1, p2);
637
638 switch (ec)
639 {
640 case ComTypes::EC_REPAINT:
641 component.repaint();
642 break;
643
644 case ComTypes::EC_COMPLETE:
645 component.stop();
646 component.setPosition (0.0);
647 break;
648
649 case ComTypes::EC_ERRORABORT:
650 case ComTypes::EC_ERRORABORTEX:
651 component.errorOccurred (getErrorMessageFromResult ((HRESULT) p1).getErrorMessage());
652 // intentional fallthrough
653 case ComTypes::EC_USERABORT:
654 component.close();
655 break;
656
657 case ComTypes::EC_STATE_CHANGE:
658 switch (p1)
659 {
660 case ComTypes::State_Paused: component.playbackStopped(); break;
661 case ComTypes::State_Running: component.playbackStarted(); break;
662 default: break;
663 }
664
665 default:
666 break;
667 }
668 }
669 }

727 // in DirectShow, full volume is 0, silence is -10000
728 static long convertToDShowVolume (float vol) noexcept
729 {
730 if (vol >= 1.0f) return 0;
731 if (vol <= 0.0f) return -10000;
732
733 return roundToInt ((vol * 10000.0f) - 10000.0f);
734 }

805 bool isRendererConnected()
806 {
807 ComSmartPtr<ComTypes::IEnumPins> enumPins;
808
809 HRESULT hr = baseFilter->EnumPins (enumPins.resetAndGetPointerAddress());
810
811 if (SUCCEEDED (hr))
812 hr = enumPins->Reset();
813
814 ComSmartPtr<ComTypes::IPin> pin;
815
816 while (SUCCEEDED (hr)
817 && enumPins->Next (1, pin.resetAndGetPointerAddress(), nullptr) == S_OK)
818 {
819 ComSmartPtr<ComTypes::IPin> otherPin;
820
821 hr = pin->ConnectedTo (otherPin.resetAndGetPointerAddress());
822
823 if (SUCCEEDED (hr))
824 {
825 ComTypes::PIN_DIRECTION direction;
826 hr = pin->QueryDirection (&direction);
827
828 if (SUCCEEDED (hr) && direction == ComTypes::PINDIR_INPUT)
829 return true;
830 }
831 else if (hr == ComTypes::VFW_E_NOT_CONNECTED)
832 {
833 hr = S_OK;
834 }
835 }
836
837 return false;
838 }

849 NativeWindowClass()

878 static LRESULT CALLBACK wndProc (HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam)
879 {
880 if (auto* c = (DirectShowContext*) GetWindowLongPtr (hwnd, GWLP_USERDATA))
881 {
882 switch (msg)
883 {
884 case WM_NCHITTEST: return HTTRANSPARENT;
885 case WM_ERASEBKGND: return 1;
886 case WM_DISPLAYCHANGE: c->displayResolutionChanged(); break;
887 case graphEventID: c->graphEventProc(); return 0;
888 default: break;
889 }
890 }
891
892 return DefWindowProc (hwnd, msg, wParam, lParam);
893 }
