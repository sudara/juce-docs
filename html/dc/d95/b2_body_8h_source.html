626inline void b2Body::SetAwake(bool flag)
627{
628 if (flag)
629 {
630 if ((m_flags & e_awakeFlag) == 0)
631 {
632 m_flags |= e_awakeFlag;
633 m_sleepTime = 0.0f;

639 m_sleepTime = 0.0f;
640 m_linearVelocity.SetZero();
641 m_angularVelocity = 0.0f;
642 m_force.SetZero();
643 m_torque = 0.0f;
644 }

790inline void b2Body::ApplyLinearImpulse(const b2Vec2& impulse, const b2Vec2& point)
791{

819inline void b2Body::SynchronizeTransform()
820{
821 m_xf.q.Set(m_sweep.a);
822 m_xf.p = m_sweep.c - b2Mul(m_xf.q, m_sweep.localCenter);
823}

825inline void b2Body::Advance(float32 alpha)
826{
827 // Advance to the new safe time. This doesn't sync the broad-phase.
828 m_sweep.Advance(alpha);
829 m_sweep.c = m_sweep.c0;
830 m_sweep.a = m_sweep.a0;
831 m_xf.q.Set(m_sweep.a);
832 m_xf.p = m_sweep.c - b2Mul(m_xf.q, m_sweep.localCenter);
833}
