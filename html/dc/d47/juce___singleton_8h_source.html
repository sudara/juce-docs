38template <typename Type, typename MutexType, bool onlyCreateOncePerRun>
39struct SingletonHolder : private MutexType // (inherited so we can use the empty-base-class optimisation)
40{
41 SingletonHolder() = default;
42
43 ~SingletonHolder()
44 {
45 /* The static singleton holder is being deleted before the object that it holds
46 has been deleted. This could mean that you've forgotten to call clearSingletonInstance()
47 in the class's destructor, or have failed to delete it before your app shuts down.
48 If you're having trouble cleaning up your singletons, perhaps consider using the
49 SharedResourcePointer class instead.
50 */
51 jassert (instance.load() == nullptr);
52 }
53
55 Type* get()
56 {
57 if (auto* ptr = instance.load())
58 return ptr;
59
60 typename MutexType::ScopedLockType sl (*this);
61
62 if (auto* ptr = instance.load())
63 return ptr;
64
65 auto once = onlyCreateOncePerRun; // (local copy avoids VS compiler warning about this being constant)
66
67 if (once)
68 {
69 static bool createdOnceAlready = false;
70
71 if (createdOnceAlready)
72 {
73 // This means that the doNotRecreateAfterDeletion flag was set
74 // and you tried to create the singleton more than once.
75 jassertfalse;
76 return nullptr;
77 }
78
79 createdOnceAlready = true;
80 }
81
82 static bool alreadyInside = false;
83
84 if (alreadyInside)
85 {
86 // This means that your object's constructor has done something which has
87 // ended up causing a recursive loop of singleton creation.
88 jassertfalse;
89 return nullptr;
90 }
91
92 const ScopedValueSetter<bool> scope (alreadyInside, true);
93 return getWithoutChecking();
94 }
95
99 Type* getWithoutChecking()
100 {
101 if (auto* p = instance.load())
102 return p;
103
104 auto* newObject = new Type(); // (create into a local so that instance is still null during construction)
105 instance.store (newObject);
106 return newObject;
107 }
108
110 void deleteInstance()
111 {
112 typename MutexType::ScopedLockType sl (*this);
113 delete instance.exchange (nullptr);
114 }
115
117 void clear (Type* expectedObject) noexcept
118 {
119 instance.compare_exchange_strong (expectedObject, nullptr);
120 }
121
122 // This must be atomic, otherwise a late call to get() may attempt to read instance while it is
123 // being modified by the very first call to get().
124 std::atomic<Type*> instance { nullptr };
125};
