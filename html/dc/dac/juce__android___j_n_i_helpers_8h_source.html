30template <typename JavaType>
31class LocalRef
32{
33public:
34 explicit inline LocalRef() noexcept : obj (nullptr) {}
35 explicit inline LocalRef (JavaType o) noexcept : obj (o) {}
36 inline LocalRef (const LocalRef& other) noexcept : obj (retain (other.obj)) {}
37 inline LocalRef (LocalRef&& other) noexcept : obj (nullptr) { std::swap (obj, other.obj); }
38 ~LocalRef() { clear(); }
39
40 void clear()
41 {
42 if (obj != nullptr)
43 {
44 getEnv()->DeleteLocalRef (obj);
45 obj = nullptr;
46 }
47 }
48
49 LocalRef& operator= (const LocalRef& other)
50 {
51 JavaType newObj = retain (other.obj);
52 clear();
53 obj = newObj;
54 return *this;
55 }
56
57 LocalRef& operator= (LocalRef&& other)
58 {
59 clear();
60 std::swap (other.obj, obj);
61 return *this;
62 }
63
64 inline operator JavaType() const noexcept { return obj; }
65 inline JavaType get() const noexcept { return obj; }
66
67private:
68 JavaType obj;
69
70 static JavaType retain (JavaType obj)
71 {
72 return obj == nullptr ? nullptr : (JavaType) getEnv()->NewLocalRef (obj);
73 }
74};

77class GlobalRef
78{
79public:
80 inline GlobalRef() noexcept : obj (nullptr) {}
81 inline explicit GlobalRef (const LocalRef<jobject>& o) : obj (retain (o.get(), getEnv())) {}
82 inline explicit GlobalRef (const LocalRef<jobject>& o, JNIEnv* env) : obj (retain (o.get(), env)) {}
83 inline GlobalRef (const GlobalRef& other) : obj (retain (other.obj, getEnv())) {}
84 inline GlobalRef (GlobalRef && other) noexcept : obj (nullptr) { std::swap (other.obj, obj); }
85 ~GlobalRef() { clear(); }
86
87
88 inline void clear() { if (obj != nullptr) clear (getEnv()); }
89 inline void clear (JNIEnv* env)
90 {
91 if (obj != nullptr)
92 {
93 env->DeleteGlobalRef (obj);
94 obj = nullptr;
95 }
96 }
97
98 inline GlobalRef& operator= (const GlobalRef& other)
99 {
100 jobject newObj = retain (other.obj, getEnv());
101 clear();
102 obj = newObj;
103 return *this;
104 }
105
106 inline GlobalRef& operator= (GlobalRef&& other)
107 {
108 clear();
109 std::swap (obj, other.obj);
110
111 return *this;
112 }
113
114 //==============================================================================
115 inline operator jobject() const noexcept { return obj; }
116 inline jobject get() const noexcept { return obj; }
117
118 //==============================================================================
119 #define DECLARE_CALL_TYPE_METHOD(returnType, typeName) \
120 returnType call##typeName##Method (jmethodID methodID, ... ) const \
121 { \
122 va_list args; \
123 va_start (args, methodID); \
124 returnType result = getEnv()->Call##typeName##MethodV (obj, methodID, args); \
125 va_end (args); \
126 return result; \
127 }
128
129 DECLARE_CALL_TYPE_METHOD (jobject, Object)
130 DECLARE_CALL_TYPE_METHOD (jboolean, Boolean)
131 DECLARE_CALL_TYPE_METHOD (jbyte, Byte)
132 DECLARE_CALL_TYPE_METHOD (jchar, Char)
133 DECLARE_CALL_TYPE_METHOD (jshort, Short)
134 DECLARE_CALL_TYPE_METHOD (jint, Int)
135 DECLARE_CALL_TYPE_METHOD (jlong, Long)
136 DECLARE_CALL_TYPE_METHOD (jfloat, Float)
137 DECLARE_CALL_TYPE_METHOD (jdouble, Double)
138 #undef DECLARE_CALL_TYPE_METHOD
139
140 void callVoidMethod (jmethodID methodID, ... ) const
141 {
142 va_list args;
143 va_start (args, methodID);
144 getEnv()->CallVoidMethodV (obj, methodID, args);
145 va_end (args);
146 }
147
148private:
149 //==============================================================================
150 jobject obj = nullptr;
151
152 static jobject retain (jobject obj, JNIEnv* env)
153 {
154 return obj == nullptr ? nullptr : env->NewGlobalRef (obj);
155 }
156};

165struct SystemJavaClassComparator;
166class JNIClassBase
167{
168public:
169 explicit JNIClassBase (const char* classPath, int minSDK, const void* byteCode, size_t byteCodeSize);
170 virtual ~JNIClassBase();
171
172 inline operator jclass() const noexcept { return classRef; }
173
174 static void initialiseAllClasses (JNIEnv*);
175 static void releaseAllClasses (JNIEnv*);
176
177 inline const char* getClassPath() const noexcept { return classPath; }
178
179protected:
180 virtual void initialiseFields (JNIEnv*) = 0;
181
182 jmethodID resolveMethod (JNIEnv*, const char* methodName, const char* params);
183 jmethodID resolveStaticMethod (JNIEnv*, const char* methodName, const char* params);
184 jfieldID resolveField (JNIEnv*, const char* fieldName, const char* signature);
185 jfieldID resolveStaticField (JNIEnv*, const char* fieldName, const char* signature);
186 void resolveCallbacks (JNIEnv*, const Array<JNINativeMethod>&);
187
188private:
189 friend struct SystemJavaClassComparator;
190
191 const char* const classPath;
192 const void* byteCode;
193 size_t byteCodeSize;
194
195 int minSDK;
196 jclass classRef = nullptr;
197
198 static Array<JNIClassBase*>& getClasses();
199 void initialise (JNIEnv*);
200 void release (JNIEnv*);
201 void tryLoadingClassWithClassLoader (JNIEnv* env, jobject classLoader);
202
203 JUCE_DECLARE_NON_COPYABLE (JNIClassBase)
204};

216#define DECLARE_JNI_CLASS_WITH_BYTECODE(CppClassName, javaPath, minSDK, byteCodeData, byteCodeSize) \
217 class CppClassName ## _Class : public JNIClassBase \
218 { \
219 public: \
220 CppClassName ## _Class() : JNIClassBase (javaPath, minSDK, byteCodeData, byteCodeSize) {} \
221 \
222 void initialiseFields (JNIEnv* env) \
223 { \
224 Array<JNINativeMethod> callbacks; \
225 JNI_CLASS_MEMBERS (CREATE_JNI_METHOD, CREATE_JNI_STATICMETHOD, CREATE_JNI_FIELD, CREATE_JNI_STATICFIELD, CREATE_JNI_CALLBACK); \
226 resolveCallbacks (env, callbacks); \
227 } \
228 \
229 JNI_CLASS_MEMBERS (DECLARE_JNI_METHOD, DECLARE_JNI_METHOD, DECLARE_JNI_FIELD, DECLARE_JNI_FIELD, DECLARE_JNI_CALLBACK) \
230 }; \
231 static CppClassName ## _Class CppClassName;

775 inline String juceString (JNIEnv* env, jstring s)
776 {
777 if (s == nullptr)
778 return {};
779
780 const char* const utf8 = env->GetStringUTFChars (s, nullptr);
781 CharPointer_UTF8 utf8CP (utf8);
782 const String result (utf8CP);
783 env->ReleaseStringUTFChars (s, utf8);
784 return result;
785 }
786
787 inline String juceString (jstring s)
788 {
789 return juceString (getEnv(), s);
790 }
791
792 inline LocalRef<jstring> javaString (const String& s)
793 {
794 return LocalRef<jstring> (getEnv()->NewStringUTF (s.toUTF8()));
795 }
796
797 inline LocalRef<jstring> javaStringFromChar (const juce_wchar c)
798 {
799 char utf8[8] = { 0 };
800 CharPointer_UTF8 (utf8).write (c);
801 return LocalRef<jstring> (getEnv()->NewStringUTF (utf8));
802 }
803
804 inline LocalRef<jobjectArray> juceStringArrayToJava (const StringArray& juceArray)
805 {
806 auto* env = getEnv();
807
808 LocalRef<jobjectArray> result (env->NewObjectArray ((jsize) juceArray.size(),
809 JavaString,
810 javaString ("").get()));
811
812 for (int i = 0; i < juceArray.size(); ++i)
813 env->SetObjectArrayElement (result, i, javaString (juceArray [i]).get());
814
815 return result;
816 }
817
818 inline StringArray javaStringArrayToJuce (const LocalRef<jobjectArray>& javaArray)
819 {
820 if (javaArray.get() == nullptr)
821 return {};
822
823 auto* env = getEnv();
824
825 StringArray result;
826
827 for (int i = 0; i < env->GetArrayLength (javaArray.get()); ++i)
828 {
829 LocalRef<jstring> javaString ((jstring) env->GetObjectArrayElement (javaArray.get(), i));
830 result.add (juceString (javaString.get()));
831 }
832
833 return result;
834 }

925struct SurfaceHolderCallback : AndroidInterfaceImplementer
926{
927 virtual ~SurfaceHolderCallback() override = default;
928
929 virtual void surfaceChanged (LocalRef<jobject> holder, int format, int width, int height) = 0;
930 virtual void surfaceCreated (LocalRef<jobject> holder) = 0;
931 virtual void surfaceDestroyed (LocalRef<jobject> holder) = 0;
932
933private:
934 jobject invoke (jobject proxy, jobject method, jobjectArray args) override
935 {
936 auto* env = getEnv();
937 auto methodName = juceString ((jstring) env->CallObjectMethod (method, JavaMethod.getName));
938 LocalRef<jobject> holder (env->GetArrayLength (args) > 0 ? env->GetObjectArrayElement (args, 0) : (jobject) nullptr);
939
940 if (methodName == "surfaceChanged")
941 {
942 int intArgs[3];
943
944 for (int i = 0; i < 3; ++i)
945 {
946 LocalRef<jobject> boxedType (env->GetObjectArrayElement (args, 1 + i));
947 intArgs[i] = env->CallIntMethod (boxedType.get(), JavaInteger.intValue);
948 }
949
950 surfaceChanged (std::move (holder), intArgs[0], intArgs[1], intArgs[2]);
951 }
952 else if (methodName == "surfaceCreated")
953 {
954 surfaceCreated (std::move (holder));
955 }
956 else if (methodName == "surfaceDestroyed")
957 {
958 surfaceDestroyed (std::move (holder));
959 }
960 else
961 {
962 return AndroidInterfaceImplementer::invoke (proxy, method, args);
963 }
964
965 return nullptr;
966 }
967};
