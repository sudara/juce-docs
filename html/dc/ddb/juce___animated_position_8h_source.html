152 Behaviour behaviour;

154private:
155 //==============================================================================
156 double position = 0.0, grabbedPos = 0.0, releaseVelocity = 0.0;
157 Range<double> range;
158 Time lastUpdate, lastDrag;
159 ListenerList<Listener> listeners;
160
161 static double getSpeed (const Time last, double lastPos,
162 const Time now, double newPos)
163 {
164 auto elapsedSecs = jmax (0.005, (now - last).inSeconds());
165 auto v = (newPos - lastPos) / elapsedSecs;
166 return std::abs (v) > 0.2 ? v : 0.0;
167 }
168
169 void moveTo (double newPos)
170 {
171 auto now = Time::getCurrentTime();
172 releaseVelocity = getSpeed (lastDrag, position, now, newPos);
173 behaviour.releasedWithVelocity (newPos, releaseVelocity);
174 lastDrag = now;
175
176 setPositionAndSendChange (newPos);
177 }
178
179 void setPositionAndSendChange (double newPosition)
180 {
181 newPosition = range.clipValue (newPosition);
182
183 if (position != newPosition)
184 {
185 position = newPosition;
186 listeners.call ([this, newPosition] (Listener& l) { l.positionChanged (*this, newPosition); });
187 }
188 }
189
190 void timerCallback() override
191 {
192 auto now = Time::getCurrentTime();
193 auto elapsed = jlimit (0.001, 0.020, (now - lastUpdate).inSeconds());
194 lastUpdate = now;
195 auto newPos = behaviour.getNextPosition (position, elapsed);
196
197 if (behaviour.isStopped (newPos))
198 stopTimer();
199 else
200 startTimerHz (60);
201
202 setPositionAndSendChange (newPos);
203 }
