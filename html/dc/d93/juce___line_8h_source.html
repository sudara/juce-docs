223 Point<ValueType> getPointAlongLine (ValueType distanceFromStart,
224 ValueType perpendicularDistance) const noexcept
225 {
226 auto delta = end - start;
227 auto length = juce_hypot ((double) delta.x,
228 (double) delta.y);
229 if (length <= 0)
230 return start;
231
232 return { start.x + static_cast<ValueType> ((delta.x * distanceFromStart - delta.y * perpendicularDistance) / length),
233 start.y + static_cast<ValueType> ((delta.y * distanceFromStart + delta.x * perpendicularDistance) / length) };
234 }

262 ValueType getDistanceFromPoint (Point<ValueType> targetPoint,
263 Point<ValueType>& pointOnLine) const noexcept
264 {
265 auto delta = end - start;
266 auto length = delta.x * delta.x + delta.y * delta.y;
267
268 if (length > 0)
269 {
270 auto prop = ((targetPoint.x - start.x) * delta.x
271 + (targetPoint.y - start.y) * delta.y) / (double) length;
272
273 if (prop >= 0 && prop <= 1.0)
274 {
275 pointOnLine = start + delta * prop;
276 return targetPoint.getDistanceFrom (pointOnLine);
277 }
278 }
279
280 auto fromStart = targetPoint.getDistanceFrom (start);
281 auto fromEnd = targetPoint.getDistanceFrom (end);
282
283 if (fromStart < fromEnd)
284 {
285 pointOnLine = start;
286 return fromStart;
287 }
288
289 pointOnLine = end;
290 return fromEnd;
291 }

360 static bool findIntersection (const Point<ValueType> p1, const Point<ValueType> p2,
361 const Point<ValueType> p3, const Point<ValueType> p4,
362 Point<ValueType>& intersection) noexcept
363 {
364 if (p2 == p3)
365 {
366 intersection = p2;
367 return true;
368 }
369
370 auto d1 = p2 - p1;
371 auto d2 = p4 - p3;
372 auto divisor = d1.x * d2.y - d2.x * d1.y;
373
374 if (divisor == 0)
375 {
376 if (! (d1.isOrigin() || d2.isOrigin()))
377 {
378 if (d1.y == 0 && d2.y != 0)
379 {
380 auto along = (p1.y - p3.y) / d2.y;
381 intersection = p1.withX (p3.x + along * d2.x);
382 return isZeroToOne (along);
383 }
384
385 if (d2.y == 0 && d1.y != 0)
386 {
387 auto along = (p3.y - p1.y) / d1.y;
388 intersection = p3.withX (p1.x + along * d1.x);
389 return isZeroToOne (along);
390 }
391
392 if (d1.x == 0 && d2.x != 0)
393 {
394 auto along = (p1.x - p3.x) / d2.x;
395 intersection = p1.withY (p3.y + along * d2.y);
396 return isZeroToOne (along);
397 }
398
399 if (d2.x == 0 && d1.x != 0)
400 {
401 auto along = (p3.x - p1.x) / d1.x;
402 intersection = p3.withY (p1.y + along * d1.y);
403 return isZeroToOne (along);
404 }
405 }
406
407 intersection = (p2 + p3) / static_cast<ValueType> (2);
408 return false;
409 }
410
411 auto along1 = ((p1.y - p3.y) * d2.x - (p1.x - p3.x) * d2.y) / divisor;
412 intersection = p1 + d1 * along1;
413
414 if (! isZeroToOne (along1))
415 return false;
416
417 auto along2 = ((p1.y - p3.y) * d1.x - (p1.x - p3.x) * d1.y) / divisor;
418 return isZeroToOne (along2);
419 }
