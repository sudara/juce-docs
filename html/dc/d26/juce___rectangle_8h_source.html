665 Rectangle getIntersection (Rectangle other) const noexcept
666 {
667 auto nx = jmax (pos.x, other.pos.x);
668 auto ny = jmax (pos.y, other.pos.y);
669 auto nw = jmin (pos.x + w, other.pos.x + other.w) - nx;

671 if (nw >= ValueType())

673 auto nh = jmin (pos.y + h, other.pos.y + other.h) - ny;

679 return {};

739 bool enlargeIfAdjacent (Rectangle other) noexcept

768 bool reduceIfPartlyContainedIn (Rectangle other) noexcept
769 {
770 int inside = 0;
771 auto otherR = other.getRight();
772 if (pos.x >= other.pos.x && pos.x < otherR) inside = 1;
773 auto otherB = other.getBottom();
774 if (pos.y >= other.pos.y && pos.y < otherB) inside |= 2;
775 auto r = pos.x + w;
776 if (r >= other.pos.x && r < otherR) inside |= 4;
777 auto b = pos.y + h;
778 if (b >= other.pos.y && b < otherB) inside |= 8;
779
780 switch (inside)
781 {
782 case 1 + 2 + 8: w = r - otherR; pos.x = otherR; return true;
783 case 1 + 2 + 4: h = b - otherB; pos.y = otherB; return true;
784 case 2 + 4 + 8: w = other.pos.x - pos.x; return true;
785 case 1 + 4 + 8: h = other.pos.y - pos.y; return true;
786 default: break;
787 }
788
789 return false;
790 }

815 Rectangle transformedBy (const AffineTransform& transform) const noexcept
816 {
817 using FloatType = typename TypeHelpers::SmallestFloatType<ValueType>::type;
818
819 auto x1 = static_cast<FloatType> (pos.x), y1 = static_cast<FloatType> (pos.y);
820 auto x2 = static_cast<FloatType> (pos.x + w), y2 = static_cast<FloatType> (pos.y);
821 auto x3 = static_cast<FloatType> (pos.x), y3 = static_cast<FloatType> (pos.y + h);
822 auto x4 = static_cast<FloatType> (x2), y4 = static_cast<FloatType> (y3);
823
824 transform.transformPoints (x1, y1, x2, y2);
825 transform.transformPoints (x3, y3, x4, y4);
826
827 auto rx1 = jmin (x1, x2, x3, x4);
828 auto rx2 = jmax (x1, x2, x3, x4);
829 auto ry1 = jmin (y1, y2, y3, y4);
830 auto ry2 = jmax (y1, y2, y3, y4);
831
832 Rectangle r;
833 Rectangle<FloatType> (rx1, ry1, rx2 - rx1, ry2 - ry1).copyWithRounding (r);
834 return r;
835 }

909 auto minY = points[0].y;
910 auto maxY = minY;
911

913 {
914 minX = jmin (minX, points[i].x);
915 maxX = jmax (maxX, points[i].x);
916 minY = jmin (minY, points[i].y);
917 maxY = jmax (maxY, points[i].y);
918 }
919
920 return { minX, minY, maxX - minX, maxY - minY };
921 }

930 {
931 auto x = jmax (x1, x2);
932 w1 = jmin (x1 + w1, x2 + w2) - x;
933
934 if (w1 > ValueType())
935 {
936 auto y = jmax (y1, y2);
937 h1 = jmin (y1 + h1, y2 + h2) - y;
938
939 if (h1 > ValueType())
940 {
941 x1 = x; y1 = y;
942 return true;
943 }
944 }
945
946 return false;
947 }
